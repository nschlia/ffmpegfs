97

103  m_angle_idx = virtualfile->m_dvd.m_angle_no - 1;
104  m_duration = virtualfile->m_duration;
105  }
106  else

113  }

116 
117  // Open the disc.
118  m_dvd = DVDOpen(path().c_str());
119  if (m_dvd == nullptr)
120  {
121  Logging::error(path(), "Couldn't open DVD.");
122  return EINVAL;
123  }

126  m_vmg_file = ifoOpen(m_dvd, 0);
127  if (m_vmg_file == nullptr)
128  {

166  }

175  return EINVAL;

189  }
190  else
191  {
192  m_end_cell = m_cur_pgc->nr_of_cells;
193  }
194

196  m_dvd_title = DVDOpenFile(m_dvd, tt_srpt->title[m_title_idx].title_set_nr, DVD_READ_TITLE_VOBS);
197  if (m_dvd_title == nullptr)
198  {
199  Logging::error(path(), "Can't open title VOBS (VTS_%<%02d>1_X.VOB).", tt_srpt->title[m_title_idx].title_set_nr);
200  ifoClose(m_vts_file);
201  ifoClose(m_vmg_file);
202  DVDClose(m_dvd);
203  return EINVAL;
204  }

206  rewind();
207 
208  // Determine the net file size
209  m_size = 0;
210  size_t bytes_read;
211 
212  while ((bytes_read = read(nullptr, ULONG_MAX)) != 0 && !eof())
213  {
214  m_size += bytes_read;
215  }
216 
217  rewind();
218

239 bool DvdIO::get_packet_size(const uint8_t *p, size_t peek, size_t *size) const
240 {
241  if (peek < 4)
242  {
243  return false; // Invalid size
244  }
245 
246  switch (p[PS_STREAM_ID])
247  {
248  case PS_STREAM_ID_END_STREAM:
249  {
250  *size = 4;
251  return true;
252  }
253  case PS_STREAM_ID_PACK_HEADER:
254  {
255  // MPEG-2 pack header, see http://stnsoft.com/DVD/packhdr.html
256  if (peek > 4)
257  {
258  if (peek >= 14 && (p[4] >> 6) == 0x01)
259  {
260  *size = static_cast<size_t>(14 + (p[13] & 0x07)); // Byte 13 Bit 0..2: Pack stuffing length
261  return true;
262  }
263  else if (peek >= 12 && (p[4] >> 4) == 0x02)
264  {
265  *size = 12; // unclear what this is for
266  return true;
267  }
268  }
269  break;
270  }
271  case PS_STREAM_ID_SYSTEM_HEADER: // http://stnsoft.com/DVD/sys_hdr.html, see http://stnsoft.com/DVD/sys_hdr.html
272  case PS_STREAM_ID_MAP: // ???
273  case PS_STREAM_ID_DIRECTORY: // ???
274  default:
275  {
276  if (peek >= 6)
277  {
278  *size = (6 + ((static_cast<unsigned int>(p[4]) << 8) | p[5])); // Byte 4/5: header length
279  return true;
280  }
281  break;
282  }
283  }
284  return false; // unknown ID
285 }

287 int DvdIO::get_pes_id(const uint8_t *buffer, size_t size) const
288 {
289  if (buffer[PS_STREAM_ID] == PS_STREAM_ID_PRIVATE_STREAM1)
290  {
291  uint8_t sub_id = 0;
292  if (size >= 9 && size >= static_cast<unsigned int>(9 + buffer[8]))
293  {
294  const size_t start = static_cast<size_t>(9 + buffer[8]);
295  sub_id = buffer[start];
296 
297  if ((sub_id & 0xfe) == 0xa0 &&
298  size >= start + 7 &&
299  (buffer[start + 5] >= 0xc0 ||
300  buffer[start + 6] != 0x80))
301  {
302  // AOB LPCM/MLP extension
303  // XXX for MLP I think that the !=0x80 test is not good and
304  // will fail for some valid files
305  return (0xa000 | (sub_id & 0x01));
306  }
307  }
308 
309  // VOB extension
310  return (0xbd00 | sub_id);
311  }
312  else if (buffer[PS_STREAM_ID] == PS_STREAM_ID_EXTENDED &&
313  size >= 9 &&
314  (buffer[6] & 0xC0) == 0x80 && // mpeg2
315  (buffer[7] & 0x01) == 0x01) // extension_flag
316  {
317  // ISO 13818 amendment 2 and SMPTE RP 227
318  const uint8_t flags = buffer[7];
319  size_t skip = 9;
320 
321  // Find PES extension
322  if ((flags & 0x80))
323  {
324  skip += 5; // pts
325  if ((flags & 0x40))
326  {
327  skip += 5; // dts
328  }
329  }
330  if ((flags & 0x20))
331  {
332  skip += 6;
333  }
334  if ((flags & 0x10))
335  {
336  skip += 3;
337  }
338  if ((flags & 0x08))
339  {
340  skip += 1;
341  }
342  if ((flags & 0x04))
343  {
344  skip += 1;
345  }
346  if ((flags & 0x02))
347  {
348  skip += 2;
349  }
350 
351  if (skip < size && (buffer[skip] & 0x01))
352  {
353  const uint8_t flags2 = buffer[skip];
354 
355  // Find PES extension 2
356  skip += 1;
357  if (flags2 & 0x80)
358  {
359  skip += 16;
360  }
361  if ((flags2 & 0x40) && skip < size)
362  {
363  skip += static_cast<size_t>(1 + buffer[skip]);
364  }
365  if (flags2 & 0x20)
366  {
367  skip += 2;
368  }
369  if (flags2 & 0x10)
370  {
371  skip += 2;
372  }
373 
374  if (skip + 1 < size)
375  {
376  const int i_extension_field_length = buffer[skip] & 0x7f;
377  if (i_extension_field_length >=1)
378  {
379  int i_stream_id_extension_flag = (buffer[skip+1] >> 7) & 0x1;
380  if (i_stream_id_extension_flag == 0)
381  {
382  return (0xfd00 | (buffer[skip+1] & 0x7f));
383  }
384  }
385  }
386  }
387  }
388  return buffer[PS_STREAM_ID];
389 }

391 size_t DvdIO::demux_pes(uint8_t *out, const uint8_t *in, size_t len) const
392 {
393  size_t netsize = 0;
394  while (len > 0)
395  {
396  size_t size = 0;
397  if (!get_packet_size(in, len, &size) || size > len)
398  {
399  break;
400  }
401 
402  // Parse block and copy to buffer
403  switch (0x100 | in[PS_STREAM_ID])
404  {
405  // Ignore thesse...
406  case (0x100 | PS_STREAM_ID_END_STREAM):
407  case (0x100 | PS_STREAM_ID_SYSTEM_HEADER):
408  case (0x100 | PS_STREAM_ID_MAP):
409  {
410  break;
411  }
412  case (0x100 | PS_STREAM_ID_PACK_HEADER): // MPEG-2 Pack Header
413  {
414  memcpy(out, in, size);
415  out += size;
416  netsize += size;
417  break;
418  }
419  default:
420  {
421  int id = get_pes_id(in, size);
422  if (id >= PS_STREAM_ID_AUDIO) // Audio/Video/Extended or Directory
423  {
424  // Probably this is sufficient here:
425  // 110x xxxx 0xC0 - 0xDF MPEG-1 or MPEG-2 audio stream number x xxxx
426  // 1110 xxxx 0xE0 - 0xEF MPEG-1 or MPEG-2 video stream number xxxx
427  memcpy(out, in, size);
428  out += size;
429  netsize += size;
430  }
431  break;
432  }
433  }
434 
435  in += size;
436  len -= size;
437  }
438 
439  return netsize;
440 }

442 DvdIO::DSITYPE DvdIO::handle_DSI(void *_dsi_pack, size_t * cur_output_size, unsigned int *next_block, uint8_t *data)
443 {
444  dsi_t * dsi_pack = reinterpret_cast<dsi_t*>(_dsi_pack);
445  DSITYPE dsitype = DSITYPE_CONTINUE;
446  bool end_of_cell;
447 
448  navRead_DSI(dsi_pack, &data[DSI_START_BYTE]);
449 
450  // Determine where we go next. These values are the ones we mostly
451  // care about.
452  m_cur_block = dsi_pack->dsi_gi.nv_pck_lbn;
453  *cur_output_size = dsi_pack->dsi_gi.vobu_ea;
454 
455  // If we're not at the end of this cell, we can determine the next
456  // VOBU to display using the VOBU_SRI information section of the
457  // DSI. Using this value correctly follows the current angle,
458  // avoiding the doubled scenes in The Matrix, and makes our life
459  // really happy.
460 
461  *next_block = m_cur_block + (dsi_pack->vobu_sri.next_vobu & 0x7fffffff);
462  end_of_cell = (dsi_pack->vobu_sri.next_vobu == SRI_END_OF_CELL || *next_block >= m_cur_pgc->cell_playback[m_cur_cell].last_sector); // Double check end of cell: DVD transcoding stops in the middle of the chapter #48
463 
464  if (!end_of_cell && m_angle_idx > 1)
465  {
466  switch ((dsi_pack->sml_pbi.category & 0xf000) >> 12)
467  {
468  case 0x4:
469  {
470  // Interleaved unit with no angle
471  // dsi_pack->sml_pbi.ilvu_sa
472  // relative offset to the next ILVU block (not VOBU) for this angle or scene.
473  // 00 00 00 00 for PREU and non-interleaved blocks
474  // ff ff ff ff for the last interleaved block, indicating the end of interleaving
475  if (dsi_pack->sml_pbi.ilvu_sa != 0 && dsi_pack->sml_pbi.ilvu_sa != 0xffffffff)
476  {
477  *next_block = m_cur_block + dsi_pack->sml_pbi.ilvu_sa;
478  *cur_output_size = dsi_pack->sml_pbi.ilvu_ea;
479  }
480  else
481  {
482  *next_block = m_cur_block + dsi_pack->dsi_gi.vobu_ea + 1;
483  }
484  break;
485  }
486  case 0x5:
487  {
488  // vobu is end of ilvu
489  if (dsi_pack->sml_agli.data[m_angle_idx].address)
490  {
491  *next_block = m_cur_block + dsi_pack->sml_agli.data[m_angle_idx].address;
492  *cur_output_size = dsi_pack->sml_pbi.ilvu_ea;
493  break;
494  }
495  }
496  // fall through
497  case 0x6: // vobu is beginning of ilvu
498  case 0x9: // next scr is 0
499  case 0xa: // entering interleaved section
500  case 0x8: // non interleaved cells in interleaved section
501  default:
502  {
503  *next_block = m_cur_block + (dsi_pack->vobu_sri.next_vobu & 0x7fffffff);
504  break;
505  }
506  }
507  }
508  else if (end_of_cell)
509  {
510  if (m_next_cell >= m_cur_pgc->nr_of_cells)
511  {
512  *next_block = 0;
513  dsitype = DSITYPE_EOF_TITLE;
514  }
515  else
516  {
517  m_cur_cell = m_next_cell;
518 
519  next_cell();
520 
521  if (m_cur_cell >= m_end_cell)
522  {
523  dsitype = DSITYPE_EOF_CHAPTER;
524  }
525 
526  *next_block = m_cur_pgc->cell_playback[m_cur_cell].first_sector;
527  }
528  }
529 
530  return dsitype;
531 }

533 void DvdIO::next_cell()
534 {
535  // Check if we're entering an angle block
536  if (m_cur_pgc->cell_playback[m_cur_cell].block_type == BLOCK_TYPE_ANGLE_BLOCK)
537  {
538  m_cur_cell += m_angle_idx;
539 
540  for (int i = 0;; ++i)
541  {
542  if (m_cur_pgc->cell_playback[m_cur_cell + i].block_mode == BLOCK_MODE_LAST_CELL)
543  {
544  m_next_cell = m_cur_cell + i + 1;
545  break;
546  }
547  }
548  }
549  else
550  {
551  m_next_cell = m_cur_cell + 1;
552  }

555 size_t DvdIO::read(void * data, size_t size)
556 {
557  size_t cur_output_size;
558  size_t result_len = 0;
559  DSITYPE dsitype;
560 
561  if (m_rest_size)
562  {
563  size_t rest_size = m_rest_size;
564 
565  if (m_rest_size > size)
566  {
567  errno = EINVAL;
568  return 0;
569  }
570 
571  if (data != nullptr)
572  {
573  memcpy(data, &m_data[m_rest_pos], rest_size);
574  }
575 
576  m_rest_size = m_rest_pos = 0;
577 
578  return rest_size;
579  }
580 
581  // Playback by cell in this pgc, starting at the cell for our chapter.
582  //while (next_cell < last_cell)
583  {
584  if (m_goto_next_cell)
585  {
586  m_goto_next_cell = false;
587 
588  m_cur_cell = m_next_cell;
589 
590  next_cell();
591 
592  m_cur_block = m_cur_pgc->cell_playback[m_cur_cell].first_sector;
593  }
594 
595  if (m_cur_block >= m_cur_pgc->cell_playback[m_cur_cell].last_sector)
596  {
597  m_is_eof = false;
598  return 0;
599  }
600 
601  // We loop until we're out of this cell.
602  //for(cur_pack = cur_pgc->cell_playback[cur_cell].first_sector;
603  // cur_pack < cur_pgc->cell_playback[cur_cell].last_sector;)
604  {
605  dsi_t dsi_pack;
606  unsigned int next_block;
607  ssize_t maxlen;
608 
609  // Read NAV packet.
610  maxlen = DVDReadBlocks(m_dvd_title, static_cast<int>(m_cur_block), 1, m_buffer);
611  if (maxlen != 1)
612  {
613  Logging::error(path(), "Read failed for block at %1", m_cur_block);
614  m_errno = EIO;
615  return 0;
616  }
617 
618  if (!is_nav_pack(m_buffer))
619  {
620  Logging::warning(path(), "Block at %1 is probably not a NAV packet. Transcode may fail.", m_cur_block);
621  }
622 
623  // Parse the contained dsi packet.
624  dsitype = handle_DSI(&dsi_pack, &cur_output_size, &next_block, m_buffer);
625  if (m_cur_block != dsi_pack.dsi_gi.nv_pck_lbn)
626  {
627  Logging::error(path(), "Read failed at %1 because current block != dsi_pack.dsi_gi.nv_pck_lbn", m_cur_block);
628  m_errno = EIO;
629  return 0;
630  }
631 
632  if (cur_output_size >= 1024)
633  {
634  Logging::error(path(), "Read failed at %1 because current output size %2 >= 1024", m_cur_block, cur_output_size);
635  m_errno = EIO;
636  return 0;
637  }
638 
639  m_cur_block++;
640 
641  // Read in and output cur_output_size packs.
642  maxlen = DVDReadBlocks(m_dvd_title, static_cast<int>(m_cur_block), cur_output_size, m_buffer);
643 
644  if (maxlen != static_cast<int>(cur_output_size))
645  {
646  Logging::error(path(), "Read failed for %1 blocks at %2", cur_output_size, m_cur_block);
647  m_errno = EIO;
648  return 0;
649  }
650 
651  size_t netsize = cur_output_size * DVD_VIDEO_LB_LEN;
652 
653  netsize = demux_pes(m_data, m_buffer, netsize);
654 
655  if (data != nullptr)
656  {
657  if (netsize > size)
658  {
659  result_len = size;
660 
661  memcpy(data, m_data, result_len);
662 
663  m_rest_size = netsize - size;
664  m_rest_pos = size;
665  }
666  else
667  {
668  result_len = netsize;
669 
670  memcpy(data, m_data, result_len);
671  }
672  }
673  else
674  {
675  if (netsize > size)
676  {
677  result_len = size;
678  }
679  else
680  {
681  result_len = netsize;
682  }

737 int DvdIO::seek(int64_t offset, int whence)
738 {
739  errno = 0;
740 
741  if (whence == SEEK_SET && !offset)
742  {
743  // Only rewind (seek(0, SEEK_SET) is implemented yet
744  rewind();
745  return 0;
746  }
747  size_t cur_pos = tell();
748  size_t abs_offset = 0;
749 
750  switch (whence)
751  {
752  case SEEK_SET:
753  assert(offset >= 0);
754  abs_offset = static_cast<size_t>(offset);
755  break;
756  case SEEK_CUR:
757  abs_offset += cur_pos;
758  break;
759  case SEEK_END:
760  assert(offset >= 0);
761  abs_offset = size() - abs_offset;
762  break;
763  default:
764  errno = EINVAL;
765  return (EOF);
766  }
767 
768  if (cur_pos == abs_offset)
769  {
770  // Already at right position
771  return static_cast<int>(abs_offset);
772  }
773 
774  if (cur_pos > abs_offset)
775  {
776  // Need to start from beginning to find byte position
777  rewind();
778  cur_pos = 0;
779  }
780 
781  size_t total_read = 0;
782  size_t bytes_read;
783  while ((bytes_read = read(nullptr, abs_offset - total_read - cur_pos)) != 0 && total_read < abs_offset && !eof())
784  {
785  total_read += bytes_read;
786  }
787 
788  if (total_read)
789  {
790  return static_cast<int>(cur_pos + total_read);
791  }
792 
793  errno = EPERM;
794  return (EOF);
795 }

835 // Code nicked from Handbrake (https://github.com/HandBrake/HandBrake/blob/master/libhb/dvd.c)
836 bool DvdIO::is_nav_pack(const unsigned char *buffer) const
837 {
838  /*
839  * The NAV Pack is comprised of the PCI Packet and DSI Packet, both
840  * of these start at known offsets and start with a special identifier.
841  *
842  * NAV = {
843  * PCI = { 00 00 01 bf # private stream header
844  * ?? ?? # length
845  * 00 # substream
846  * ...
847  * }
848  * DSI = { 00 00 01 bf # private stream header
849  * ?? ?? # length
850  * 01 # substream
851  * ...
852  * }
853  *
854  * The PCI starts at offset 0x26 into the sector, and the DSI starts at 0x400
855  *
856  * This information from: http://dvd.sourceforge.net/dvdinfo/
857  */
858  if ((buffer[0x26] == 0x00 && // PCI
859  buffer[0x27] == 0x00 &&
860  buffer[0x28] == 0x01 &&
861  buffer[0x29] == 0xbf &&
862  buffer[0x2c] == 0x00) &&
863  (buffer[0x400] == 0x00 && // DSI
864  buffer[0x401] == 0x00 &&
865  buffer[0x402] == 0x01 &&
866  buffer[0x403] == 0xbf &&
867  buffer[0x406] == 0x01))
868  {
869  return (1);
870  }
871  else
872  {
873  return (0);
874  }
875 }
