82 int VcdEntries::load_file(const std::string & path)
83 {
84  FILE * fpi = nullptr;
85  std::string fullname;
86  bool is_vcd = false;
87 
88  clear();
89 
90  if (!VCDUTILS::locate_file(path, "ENTRIES", fullname, is_vcd))
91  {
92  return ENOENT;
93  }
94 
95  VCDUTILS::get_directory(path, &m_disk_path);
96 
97  try
98  {
99  VCDENTRY vcdentry;
100  struct stat stbuf;
101  int num_entries = 0;
102 
103  fpi = fopen(fullname.c_str(), "rb");
104  if (fpi == nullptr)
105  {
106  throw static_cast<int>(errno);
107  }
108 
109  if (fstat(fileno(fpi), &stbuf) != 0)
110  {
111  throw static_cast<int>(ferror(fpi));
112  }
113 
114  m_file_date = stbuf.st_mtime;
115 
116  memset(&vcdentry, 0, sizeof(vcdentry));
117 
118  if (fread(reinterpret_cast<char *>(&vcdentry), 1, sizeof(vcdentry), fpi) != sizeof(vcdentry))
119  {
120  throw static_cast<int>(ferror(fpi));
121  }
122 
123  m_id = VCDUTILS::convert_txt2string(reinterpret_cast<const char *>(vcdentry.m_ID), sizeof(vcdentry.m_ID));
124  m_type = static_cast<VCDTYPE>(vcdentry.m_type);
125  m_profile_tag = static_cast<VCDPROFILETAG>(vcdentry.m_profile_tag);
126  num_entries = htons(vcdentry.m_num_entries);
127  m_duration = 0;
128 
129  int sec = BCD2DEC(vcdentry.m_chapter[0].m_msf.m_min) * 60 + BCD2DEC(vcdentry.m_chapter[0].m_msf.m_sec);
130  for (int chapter_no = 0, total = num_entries; chapter_no < total; chapter_no++)
131  {
132  if (chapter_no && BCD2DEC(vcdentry.m_chapter[chapter_no].m_msf.m_min) * 60 + BCD2DEC(vcdentry.m_chapter[chapter_no].m_msf.m_sec) - sec < 1)
133  {
134  // Skip chapters shorter than 1 second
135  sec = BCD2DEC(vcdentry.m_chapter[chapter_no].m_msf.m_min) * 60 + BCD2DEC(vcdentry.m_chapter[chapter_no].m_msf.m_sec);
136  --num_entries;
137  continue;
138  }
139 
140  VcdChapter chapter(vcdentry.m_chapter[chapter_no], is_vcd);
141 
142  m_chapters.push_back(chapter);
143  }
144 
145  // Calculate durations of all chapters until last. This will be done later as we do not yet know the duration of the stream
146  for (size_t chapter_no = 0; chapter_no < m_chapters.size() - 1; chapter_no++)
147  {
148  VcdChapter & chapter1 = m_chapters[chapter_no];
149  const VcdChapter & chapter2 = m_chapters[chapter_no + 1];
150  int64_t chapter_duration = chapter2.get_start_time() - chapter1.get_start_time();
151 
152  // Chapter duration
153  chapter1.m_duration = chapter_duration;
154  // Total duration
155  m_duration += chapter_duration;
156  }
157  }
158  catch (int orgerrno)
159  {
160  if (fpi != nullptr)
161  {
162  fclose(fpi);
163  }
164  return orgerrno;
165  }
166 
167  fclose(fpi);
168 
169  return scan_chapters();
170 }

172 int VcdEntries::scan_chapters()
173 {
174  FILE * fpi = nullptr;
175  std::string fullname;
176  struct stat stbuf;
177 
178  if (!m_chapters.size())
179  {
180  return EIO; // Fail safe only: Should not happen, at least 1 chapter is required.
181  }
182 
183  try
184  {
185  int last_track_no = -1;
186  int64_t first_sync = -1;
187 
188  // Build list of chapters
189  for (size_t chapter_no = 0; chapter_no < m_chapters.size(); chapter_no++)
190  {
191  if (last_track_no != m_chapters[chapter_no].get_track_no())
192  {
193  last_track_no = m_chapters[chapter_no].get_track_no();
194 
195  int orgerrno = VCDUTILS::locate_video(m_disk_path, last_track_no, fullname);
196  if (orgerrno != 0)
197  {
198  throw static_cast<int>(orgerrno);
199  }
200 
201  if (chapter_no)
202  {
203  m_chapters[chapter_no - 1].m_end_pos = static_cast<uint64_t>(stbuf.st_size);
204  }
205 
206  if (fpi != nullptr)
207  {
208  fclose(fpi);
209  }
210 
211  fpi = fopen(fullname.c_str(), "rb");
212 
213  if (fpi == nullptr)
214  {
215  throw static_cast<int>(errno);
216  }
217 
218  if (fstat(fileno(fpi), &stbuf) != 0)
219  {
220  throw static_cast<int>(ferror(fpi));
221  }
222 
223  // Locate the first sync bytes
224  SEEKRES res = seek_sync(fpi, SYNC, sizeof(SYNC));
225 
226  if (res != SEEKRES_FOUND)
227  {
228  throw static_cast<int>(EIO);
229  }
230 
231  first_sync = ftell(fpi) - static_cast<int64_t>(sizeof(SYNC));
232  }
233 
234  int64_t total_chunks = (stbuf.st_size - first_sync) / VCD_SECTOR_SIZE;
235  int64_t first = 0;
236  int64_t last = total_chunks - 1;
237  int64_t middle = (first + last) / 2;
238 
239  // Locate sector with correct start time
240  while (first <= last)
241  {
242  VcdChapter buffer(m_chapters[chapter_no].get_is_svcd());
243  __off_t file_pos = static_cast<__off_t>(first_sync + middle * VCD_SECTOR_SIZE);
244 
245  if (fseek(fpi, file_pos, SEEK_SET))
246  {
247  throw static_cast<int>(ferror(fpi));
248  }
249 
250  int orgerrno = buffer.read(fpi, last_track_no);
251  if (orgerrno)
252  {
253  throw static_cast<int>(orgerrno);
254  }
255 
256  if (buffer < m_chapters[chapter_no])
257  {
258  first = middle + 1;
259  }
260  else if (buffer == m_chapters[chapter_no])
261  {
262  m_chapters[chapter_no].m_start_pos = static_cast<uint64_t>(file_pos);
263 
264  if (chapter_no)
265  {
266  m_chapters[chapter_no - 1].m_end_pos = static_cast<uint64_t>(file_pos);
267  }
268  break;
269  }
270  else
271  {
272  last = middle - 1;
273  }
274 
275  middle = (first + last) / 2;
276  }
277  }
278 
279  {
280  VcdChapter buffer(m_chapters[m_chapters.size() - 1].get_is_svcd());
281  int64_t total_chunks = (stbuf.st_size - first_sync) / VCD_SECTOR_SIZE;
282 
283  // Read time stamp of last sector
284  if (fseek(fpi, static_cast<long int>(first_sync + (total_chunks - 1) * VCD_SECTOR_SIZE), SEEK_SET))
285  {
286  throw static_cast<int>(ferror(fpi));
287  }
288 
289  int orgerrno = buffer.read(fpi, last_track_no);
290  if (orgerrno)
291  {
292  throw static_cast<int>(orgerrno);
293  }
294 
295  VcdChapter & chapter1 = m_chapters[m_chapters.size() - 1];
296  int64_t chapter_duration = buffer.get_start_time() - chapter1.get_start_time();
297 
298  // Chapter duration
299  chapter1.m_duration = chapter_duration;
300  // Total duration
301  m_duration += chapter_duration;
302  }
303  }
304  catch (int orgerrno)
305  {
306  if (fpi != nullptr)
307  {
308  fclose(fpi);
309  }
310  return orgerrno;
311  }
312 
313  // End of last chapter
314  m_chapters[m_chapters.size() - 1].m_end_pos = static_cast<uint64_t>(stbuf.st_size);
315 
316  if (fpi != nullptr)
317  {
318  fclose(fpi);
319  }
320 
321  return 0;
322 }

324 VcdEntries::SEEKRES VcdEntries::seek_sync(FILE *fpi, const char * sync, int len) const
325 {
326  char ch;
327 
328  // Read first char
329  if (fread(&ch, 1, 1, fpi) != 1)
330  {
331  return SEEKRES_NOTFOUND;
332  }
333 
334  for (int n = 0; n < len; n++)
335  {
336  if (ch != *(sync + n))
337  {
338  if (n > 0)
339  {
340  // Restart check
341  n = -1;
342  continue;
343  }
344 
345  n = -1;
346  }
347 
348  if (n == len - 1)
349  {
350  // Found!
351  break;
352  }
353 
354  if (fread(&ch, 1, 1, fpi) != 1)
355  {
356  return SEEKRES_NOTFOUND;
357  }
358  }
359 
360  return SEEKRES_FOUND;
361 }
