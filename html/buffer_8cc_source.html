77 bool Buffer::open_file(uint32_t segment_no, uint32_t flags, size_t defaultsize)
78 {
79  uint32_t index = segment_no;
80  if (index)
81  {
82  index--;
83  }
84 
85  m_ci[index].m_flags |= flags;
86 
87  if (m_ci[index].m_fd != -1)
88  {
89  Logging::trace(m_ci[index].m_cachefile, "Cache file is already open.");
90 
91  if (defaultsize)
92  {
93  // Make sure the requested size is available
94  reserve(defaultsize);
95  }
96  // Already open
97  return true;
98  }
99 
100  if (flags & CACHE_FLAG_RW)
101  {
102  Logging::debug(m_ci[index].m_cachefile, "Writing to cache file.");
103  }
104  else
105  {
106  Logging::debug(m_ci[index].m_cachefile, "Reading from cache file.");
107  }
108 
109  size_t filesize = 0;
110  bool isdefaultsize = false;
111  uint8_t *p = nullptr;
112 
113  if (!map_file(m_ci[index].m_cachefile, &m_ci[index].m_fd, &p, &filesize, &isdefaultsize, defaultsize, (flags & CACHE_FLAG_RW) ? true : false))
114  {
115  return false;
116  }
117 
118  if (!isdefaultsize)
119  {
120  m_ci[index].m_buffer_pos = m_ci[index].m_buffer_watermark = filesize;
121  }
122 
123  m_ci[index].m_buffer_size = filesize;
124  m_ci[index].m_buffer = static_cast<uint8_t*>(p);
125  m_ci[index].m_buffer_write_size = 0;
126  m_ci[index].m_buffer_writes = 0;
127 
128  ++m_cur_open; // track open files
129 
130  return true;
131 }

134 {
135  uint32_t index = segment_no;
136  if (index)
137  {
138  index--;
139  }
140 
141  m_ci[index].m_flags &= ~flags;
142 
143  if (m_ci[index].m_flags)
144  {
145  Logging::trace(m_ci[index].m_cachefile, "While attempting to close, the cache file is still in use. Currently open: %1", m_cur_open);
146  return true;
147  }
148 
149  if (m_ci[index].m_fd == -1)
150  {

157 
158  bool success = unmap_file(m_ci[index].m_cachefile, &m_ci[index].m_fd, &m_ci[index].m_buffer, m_ci[index].m_buffer_size, &m_ci[index].m_buffer_watermark);

161  m_ci[index].m_buffer_size = 0;
162

166  }

171 bool Buffer::init(bool erase_cache)
172 {
173  std::lock_guard<std::recursive_mutex> lock_mutex(m_mutex);
174 
175  if (is_open())
176  {
177  return true;
178  }
179 
180  bool success = true;
181 
182  try
183  {
184  if ((virtualfile()->m_flags & VIRTUALFLAG_HLS))
185  {
186  // HLS format: create several segments
187  if (virtualfile()->get_segment_count())
188  {
189  m_ci.resize(virtualfile()->get_segment_count());
190 
191  for (uint32_t segment_no = 1; segment_no <= virtualfile()->get_segment_count(); segment_no++)
192  {
193  make_cachefile_name(m_ci[segment_no - 1].m_cachefile, filename() + "." + make_filename(segment_no, params.current_format(virtualfile())->fileext()), params.current_format(virtualfile())->fileext(), false);
194  }
195  }
196  else
197  {
198  Logging::error(filename(), "INTERNAL ERROR: Buffer::init()! Segment count is 0.");
199  errno = EINVAL;
200  throw false;
201  }
202  }
203  else
204  {
205  // All other formats: create just a single segment.
206  m_ci.resize(1);
207 
208  make_cachefile_name(m_ci[0].m_cachefile, filename(), params.current_format(virtualfile())->fileext(), false);
209  if ((virtualfile()->m_flags & VIRTUALFLAG_FRAME))
210  {
211  // Create extra index cash for frame sets only
212  make_cachefile_name(m_ci[0].m_cachefile_idx, filename(), params.current_format(virtualfile())->fileext(), true);
213  }
214  }
215 
216  // Set current segment
217  m_cur_ci = &m_ci[0];
218 
219  // Create the path to the cache file. All paths are the same, so this is required only once.
220  char *cachefile = new_strdup(m_ci[0].m_cachefile);
221 
222  if (cachefile == nullptr)
223  {
224  Logging::error(m_ci[0].m_cachefile, "Error opening the cache file: out of memory.");
225  errno = ENOMEM;
226  throw false;
227  }
228 
229  if (mktree(dirname(cachefile), S_IRWXU | S_IRWXG | S_IROTH | S_IXOTH) && errno != EEXIST)
230  {
231  Logging::error(m_ci[0].m_cachefile, "Error creating cache directory: (%1) %2", errno, strerror(errno));
232  delete [] cachefile;
233  throw false;
234  }
235  errno = 0; // reset EEXIST, error can safely be ignored here
236 
237  delete [] cachefile;
238 
239  for (uint32_t index = 0; index < segment_count(); index++)
240  {
241  m_ci[index].m_fd = -1;
242  m_ci[index].m_buffer = nullptr;
243  m_ci[index].m_buffer_pos = 0;
244  m_ci[index].m_buffer_watermark = 0;
245  m_ci[index].m_buffer_size = 0;
246  m_ci[index].m_buffer_write_size = 0;
247  m_ci[index].m_buffer_writes = 0;
248 
249  if (erase_cache)
250  {
251  remove_cachefile(index + 1);
252  errno = 0; // ignore this error
253  }
254  }
255 
256  // Index only required for frame sets and there is only one.
257  if (!m_ci[0].m_cachefile_idx.empty())
258  {
259  if (virtualfile()->m_video_frame_count == 0)
260  {
261  errno = EINVAL;
262  Logging::error(m_ci[0].m_cachefile, "INTERNAL ERROR: Buffer::init()! Frame count is zero (%1) %2", errno, strerror(errno));
263  throw false;
264  }
265  static_assert(sizeof(IMAGE_FRAME) == 32, "sizeof(IMAGE_FRAME) must be 32 bytes");
266 
267  size_t filesize = 0;
268  bool isdefaultsize = false;
269  uint8_t *p = nullptr;
270 
271  if (!map_file(m_ci[0].m_cachefile_idx, &m_ci[0].m_fd_idx, &p, &filesize, &isdefaultsize, sizeof(IMAGE_FRAME) * virtualfile()->m_video_frame_count, false))
272  {
273  throw false;
274  }
275 
276  m_ci[0].m_buffer_size_idx = filesize;
277  m_ci[0].m_buffer_idx = static_cast<uint8_t*>(p);
278  }
279  }
280  catch (bool _success)
281  {
282  success = _success;
283 
284  if (!success)
285  {
286  for (uint32_t index = 0; index < segment_count(); index++)
287  {
288  m_ci[index].m_fd = -1;
289  m_ci[index].m_buffer = nullptr;
290  m_ci[index].m_buffer_pos = 0;
291  m_ci[index].m_buffer_watermark = 0;
292  m_ci[index].m_buffer_size = 0;
293  m_ci[index].m_buffer_write_size = 0;
294  m_ci[index].m_buffer_writes = 0;
295  }
296  }
297  }
298 
299  return success;

357 bool Buffer::map_file(const std::string & filename, int *fd, uint8_t **p, size_t *filesize, bool *isdefaultsize, size_t defaultsize, bool truncate) const
358 {
359  bool success = true;
360 
361  if (!defaultsize)
362  {
363  Logging::trace(filename, "Mapping cache file.");
364  }
365  else
366  {
367  Logging::trace(filename, "Mapping cache file with %1.", format_size(defaultsize).c_str());
368  }
369 
370  try
371  {
372  struct stat sb;
373 
374  *fd = ::open(filename.c_str(), O_CREAT | O_RDWR | (truncate ? O_TRUNC : 0), static_cast<mode_t>(0644));
375  if (*fd == -1)
376  {
377  Logging::error(filename, "The cache file could not be opened due to an error: (%1) %2", errno, strerror(errno));
378  throw false;
379  }
380 
381  if (fstat(*fd, &sb) == -1)
382  {
383  Logging::error(filename, "File stat failed: (%1) %2 (fd = %3)", errno, strerror(errno), *fd);
384  throw false;
385  }
386 
387  if (!S_ISREG(sb.st_mode))
388  {
389  Logging::error(filename, "Not a file.");
390  throw false;
391  }
392 
393  if (!sb.st_size || defaultsize)
394  {
395  // If file is empty or did not exist set file size to default
396 
397  if (!defaultsize)
398  {
399  defaultsize = static_cast<size_t>(sysconf(_SC_PAGESIZE));
400  }
401 
402  if (ftruncate(*fd, static_cast<off_t>(defaultsize)) == -1)
403  {
404  Logging::error(filename, "Error calling ftruncate() to 'stretch' the file: (%1) %2 (fd = %3)", errno, strerror(errno), *fd);
405  throw false;
406  }
407 
408  *filesize = defaultsize;
409  *isdefaultsize = true;
410  }
411  else
412  {
413  // Keep size
414  *filesize = static_cast<size_t>(sb.st_size);
415  *isdefaultsize = false;
416  }
417 
418  *p = static_cast<uint8_t *>(mmap(nullptr, *filesize, PROT_READ | PROT_WRITE, MAP_SHARED, *fd, 0));
419  if (*p == MAP_FAILED)
420  {
421  Logging::error(filename, "File mapping failed: (%1) %2 (fd = %3)", errno, strerror(errno), *fd);
422  *p = nullptr;
423  throw false;
424  }
425  }
426  catch (bool _success)
427  {
428  success = _success;
429 
430  if (!success && *fd != -1)
431  {
432  ::close(*fd);
433  *fd = -1;
434  }
435  }
436 
437  return success;
438 }

440 bool Buffer::unmap_file(const std::string &filename, int * fd, uint8_t **p, size_t len, size_t * filesize) const
441 {
442  bool success = true;
443 
444  Logging::trace(filename, "Unmapping cache file.");
445 
446  void * _p = *p;
447  size_t _filesize = *filesize;
448  int _fd = *fd;
449 
450  // Clear all variables
451  *p = nullptr;
452  *filesize = 0;
453  *fd = -1;
454 
455  if (_p != nullptr)
456  {
457  if (munmap(_p, len ? len : static_cast<size_t>(sysconf(_SC_PAGESIZE))) == -1) // Make sure we do not unmap a zero size file (spits EINVAL error)
458  {
459  Logging::error(filename, "Unmapping cache file failed: (%1) %2 Size: %3", errno, strerror(errno), len);
460  success = false;
461  }
462  }
463 
464  if (_fd != -1)
465  {
466  if (_filesize)
467  {
468  if (ftruncate(_fd, static_cast<off_t>(_filesize)) == -1)
469  {
470  Logging::error(filename, "Error calling ftruncate() to resize and close the cache file: (%1) %2 (fd = %3) Size: %5", errno, strerror(errno), _fd, _filesize);
471  success = false;
472  }
473  ::close(_fd);
474  }
475  else
476  {
477  ::close(_fd);
478 
479  if (unlink(filename.c_str()) && errno != ENOENT) // Ignore if file does not exist
480  {
481  Logging::error(filename, "Error removing the cache file: (%1) %2 (fd = %3)", errno, strerror(errno), _fd);
482  success = false;
483  }
484  }
485  }
486 
487  return success;
488 }

490 bool Buffer::release(int flags /*= CACHE_CLOSE_NOOPT*/)
491 {
492  std::lock_guard<std::recursive_mutex> lock_mutex(m_mutex);
493 
494  bool success = true;
495 
496  if (!is_open())
497  {
498  if (CACHE_CHECK_BIT(CACHE_CLOSE_DELETE, flags))
499  {
500  for (uint32_t n = 1; n <= segment_count(); n++)
501  {
502  remove_cachefile(n);
503  }
504  errno = 0; // ignore this error
505  }
506 
507  return true;
508  }
509 
510  // Write active cache to disk
511  flush();
512 
513  // Close anything that's still open
514  for (uint32_t index = 0; index < segment_count(); index++)
515  {
516  if (!close_file(index + 1, CACHE_FLAG_RO | CACHE_FLAG_RW))
517  {
518  success = false;
519  }
520 
521  if (CACHE_CHECK_BIT(CACHE_CLOSE_DELETE, flags))
522  {
523  remove_cachefile(index + 1);
524  errno = 0; // ignore this error
525  }
526  }
527 
528  // Remove index for frame sets. There is only one.
529  if (!m_ci[0].m_cachefile_idx.empty())
530  {
531  if (!unmap_file(m_ci[0].m_cachefile_idx, &m_ci[0].m_fd_idx, &m_ci[0].m_buffer_idx, m_ci[0].m_buffer_size_idx, &m_ci[0].m_buffer_size_idx))
532  {
533  success = false;
534  }
535  }
536 
537  return success;
538 }

689 size_t Buffer::write_frame(const uint8_t *data, size_t length, uint32_t frame_no)
690 {
691  std::lock_guard<std::recursive_mutex> lock_mutex(m_mutex);
692 
693  if (data == nullptr || m_cur_ci == nullptr || m_cur_ci->m_buffer_idx == nullptr || frame_no < 1 || frame_no > virtualfile()->m_video_frame_count)
694  {
695  // Invalid parameter
696  errno = EINVAL;
697  return 0;
698  }
699 
700  LPIMAGE_FRAME old_image_frame;
701  IMAGE_FRAME new_image_frame;
702  size_t bytes_written;
703  size_t start = static_cast<size_t>(frame_no - 1) * sizeof(IMAGE_FRAME);
704 
705  old_image_frame = reinterpret_cast<LPIMAGE_FRAME>(m_cur_ci->m_buffer_idx + start);
706 
707  if (old_image_frame->m_frame_no && (old_image_frame->m_size <= static_cast<uint32_t>(length)))
708  {
709  // Frame already exists and has enough space
710  old_image_frame->m_size = static_cast<uint32_t>(length);
711 
712  // Write image
713  seek(static_cast<long>(old_image_frame->m_offset), SEEK_SET);
714  bytes_written = writeio(data, old_image_frame->m_size);
