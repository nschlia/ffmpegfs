59 static bool create_vcd_virtualfile(const VcdEntries & vcd, const struct stat * statbuf, void * buf, fuse_fill_dir_t filler, bool full_title, int chapter_no)
60 {
61  const VcdChapter * chapter1 = vcd.get_chapter(chapter_no);
62  char title_buf[PATH_MAX + 1];
63  size_t size;
64  int64_t duration;
65 
66  if (!full_title)
67  {
68  size = static_cast<size_t>(chapter1->get_size());
69  duration = chapter1->get_duration();
70  snprintf(title_buf, sizeof(title_buf) - 1, "%02d. Chapter %03d [%s].%s", chapter1->get_track_no(), chapter_no + 1, replace_all(format_duration(duration), ":", "-").c_str(), params.m_format[0].fileext().c_str()); // can safely assume this a video
71  }
72  else
73  {
74  size = static_cast<size_t>(vcd.get_size());
75  duration = vcd.get_duration();
76  snprintf(title_buf, sizeof(title_buf) - 1, "%02d. Title [%s].%s", chapter1->get_track_no(), replace_all(format_duration(duration), ":", "-").c_str(), params.m_format[0].fileext().c_str()); // can safely assume this a video
77  }
78 
79  std::string filename(title_buf);
80 
81  LPVIRTUALFILE virtualfile = nullptr;
82  if (!params.m_format[0].is_multiformat())
83  {
84  virtualfile = insert_file(VIRTUALTYPE_VCD, vcd.get_disk_path() + filename, statbuf);
85  }
86  else
87  {
88  virtualfile = insert_dir(VIRTUALTYPE_VCD, vcd.get_disk_path() + filename, statbuf);
89  }
90 
91  if (virtualfile == nullptr)
92  {
93  Logging::error(vcd.get_disk_path(), "Failed to create virtual path: %1", (vcd.get_disk_path() + filename).c_str());
94  errno = EIO;
95  return false;
96  }
97 
98  if (add_fuse_entry(buf, filler, title_buf, &virtualfile->m_st, 0))
99  {
100  // break;
101  }
102 
103  // Video CD is video format anyway
104  virtualfile->m_format_idx = 0;
105  // Mark title/chapter/angle
106  virtualfile->m_full_title = full_title;
107  virtualfile->m_vcd.m_track_no = chapter1->get_track_no();
108  virtualfile->m_vcd.m_chapter_no = chapter_no;
109  virtualfile->m_vcd.m_start_pos = chapter1->get_start_pos();
110  if (!full_title)
111  {
112  virtualfile->m_vcd.m_end_pos = chapter1->get_end_pos();
113  }
114  else
115  {
116  virtualfile->m_vcd.m_end_pos = size;
117  }
118  virtualfile->m_duration = duration;
119  AVRational framerate = av_make_q(25000, 1000); //*** @todo check disk which framerate is correct, can be 25 or 29.996 fps!
120  virtualfile->m_video_frame_count = static_cast<uint32_t>(av_rescale_q(duration, av_get_time_base_q(), av_inv_q(framerate)));
121  virtualfile->m_predicted_size = size;
122 
123  return true;
124 }

134 static int parse_vcd(const std::string & path, const struct stat * statbuf, void * buf, fuse_fill_dir_t filler)
135 {
136  VcdEntries vcd;
137  bool success = true;
138 
139  vcd.load_file(path);
140 
141  Logging::debug(path, "Parsing Video CD.");
142 
143  for (int chapter_no = 0; chapter_no < vcd.get_number_of_chapters() && success; chapter_no++)
144  {
145  success = create_vcd_virtualfile(vcd, statbuf, buf, filler, false, chapter_no);
146  }
147 
148  if (success && vcd.get_number_of_chapters() > 1)
149  {
150  success = create_vcd_virtualfile(vcd, statbuf, buf, filler, true, 0);
151  }
152 
153  if (success)
154  {
155  return vcd.get_number_of_chapters();
156  }
157  else
158  {
159  return -errno;
160  }
161 }

163 int check_vcd(const std::string & _path, void *buf, fuse_fill_dir_t filler)
164 {

175  Logging::trace(path, "SVCD detected.");
176  res = parse_vcd(path, &stbuf, buf, filler);
177  Logging::trace(nullptr, "Found %1 titles.", res);
178  }
179  else

181  res = load_path(path, &stbuf, buf, filler);
182  }
183 
184  add_dotdot(buf, filler, &stbuf, 0);
185  }

187  {

190  Logging::trace(path, "VCD detected.");

197  }
198
