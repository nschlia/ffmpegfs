40std::string convert_txt2string(const char * txt, int size, bool trimmed)
41{
42 char * buffer = new(std::nothrow) char[static_cast<size_t>(size + 1)];
43 std::string ret_value;
44
45 if (buffer != nullptr)
46 {
47 std::memcpy(buffer, txt, static_cast<size_t>(size));
48 *(buffer + size) = '\0';
49
50 ret_value = buffer;
51
52 delete [] buffer;
53 }
54
55 if (trimmed)
56 {
57 return trim(ret_value);
58 }
59 else
60 {
61 return ret_value;
62 }
63}

152void get_directory(const std::string & fullname, std::string *directory)
153{
154 struct stat stbuf;
155
156 stat(fullname.c_str(), &stbuf);
157
158 if (S_ISDIR(stbuf.st_mode))
159 {
160 // Already a directory
161 *directory = fullname;
162 append_sep(directory);
163 }
164
165 else
166 {
167 // Make it a directory
168 *directory = fullname;
169 remove_filename(directory);
170 }
