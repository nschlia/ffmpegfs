299  template <typename T, typename... Args>
300  static std::string format_helper(
301  const std::string &string_to_search,
302  const size_t index_to_replace,
303  T &&val,
304  Args &&...args)
305  {
306  // Match %# exactly (e.g. %12 and %123 literally)
307 
308  std::regex exp("%(<([^>]+)>)*" + std::to_string(index_to_replace) + "(?=[^0-9]|$)");
309  std::smatch res;
310  std::string string_to_update(string_to_search);
311  std::string::const_iterator searchStart(string_to_search.cbegin());
312  size_t offset = 0;
313 
314  while (std::regex_search(searchStart, string_to_search.cend(), res, exp))
315  {
316  std::ostringstream ostr;
317 
318  if (res[2].length())
319  {
320  // Found match with printf format in res[2]
321  size_t size = static_cast<size_t>(std::snprintf(nullptr, 0, res[2].str().c_str(), val)) + 1;
322  std::vector<char> buffer;
323  buffer.resize(size);
324  std::snprintf(buffer.data(), size, res[2].str().c_str(), val);
325  ostr << buffer.data();
326  }
327  else
328  {
329  // No printf format, replace literally
330  ostr << val;
331  }
332 
333  string_to_update.replace(static_cast<size_t>(res.position()) + offset, static_cast<size_t>(res[0].length()), ostr.str());
334 
335  offset += static_cast<size_t>(res.position()) + ostr.str().length();
336 
337  searchStart = res.suffix().first;
338  }
339 
340  return format_helper(
341  string_to_update,
342  index_to_replace + 1,
343  std::forward<Args>(args)...);
344  }
