75 int BlurayIO::open(LPVIRTUALFILE virtualfile)
76 {
77  const char *bdpath = nullptr;
78  uint32_t title_count;
79  uint32_t chapter_end;
80  char *keyfile = nullptr;
81  BLURAY_TITLE_INFO *ti;
82 
83  set_virtualfile(virtualfile);
84 
85  bdpath = path().c_str();
86 
87  if (virtualfile != nullptr)
88  {
89  m_full_title = virtualfile->m_full_title;
90  m_title_idx = virtualfile->m_bluray.m_title_no - 1;
91  m_chapter_idx = virtualfile->m_bluray.m_chapter_no - 1;
92  m_angle_idx = virtualfile->m_bluray.m_angle_no - 1;
93  m_duration = virtualfile->m_duration;
94  }
95  else
96  {
97  m_full_title = false;
98  m_title_idx = 0;
99  m_chapter_idx = 0;
100  m_angle_idx = 0;
101  m_duration = AV_NOPTS_VALUE;
102  }
103 
104  chapter_end = m_chapter_idx + 1;
105 
106  Logging::debug(bdpath, "Opening input Blu-ray.");
107 
108  m_bd = bd_open(bdpath, keyfile);
109  if (m_bd == nullptr)
110  {
111  Logging::error(bdpath, "Failed to open disc.");
112  return 1;
113  }
114 
115  title_count = bd_get_titles(m_bd, TITLES_RELEVANT, 0);
116  if (title_count == 0)
117  {
118  Logging::error(bdpath, "There were no titles found.");
119  return 1;
120  }
121 
122  if (!bd_select_title(m_bd, m_title_idx))
123  {
124  Logging::error(bdpath, "The Blu-ray title no. %1 could not be opened.", m_title_idx);
125  return 1;
126  }
127  ti = bd_get_title_info(m_bd, m_title_idx, m_angle_idx);
128 
129  if (m_angle_idx >= ti->angle_count)
130  {
131  Logging::warning(bdpath, "The angle %1 is greate than the angle count %2. Using angle 1.", m_angle_idx + 1, ti->angle_count);
132  m_angle_idx = 0;
133  }
134 
135  bd_select_angle(m_bd, m_angle_idx);
136 
137  if (m_chapter_idx >= ti->chapter_count)
138  {
139  Logging::error(bdpath, "The first chapter %1 is greater than the chapter count %2.", m_chapter_idx + 1, ti->chapter_count);
140  return 1;
141  }
142 
143  if (chapter_end >= ti->chapter_count)
144  {
145  chapter_end = 0;
146  }
147 
148  if (chapter_end > 0 && !m_full_title)
149  {
150  m_end_pos = bd_chapter_pos(m_bd, chapter_end) - 1;
151  }
152  else
153  {
154  m_end_pos = static_cast<int64_t>(bd_get_title_size(m_bd));
155  }
156 
157  if (m_full_title)
158  {
159  m_duration = static_cast<int64_t>(ti->duration * AV_TIME_BASE / 90000);
160  }
161  else
162  {
163  BLURAY_TITLE_CHAPTER *chapter = &ti->chapters[m_chapter_idx];
164  m_duration = static_cast<int64_t>(chapter->duration * AV_TIME_BASE / 90000);
165  }
166 
167  bd_free_title_info(ti);
168 
169  m_start_pos = bd_seek_chapter(m_bd, m_chapter_idx);
170 
171  m_rest_size = 0;
172  m_rest_pos = 0;
173 
174  return 0;
175 }

177 size_t BlurayIO::read(void * data, size_t size)
178 {
179  size_t result_len = 0;
180 
181  if (m_rest_size)
182  {
183  result_len = m_rest_size;
184 
185  if (m_rest_size > size)
186  {
187  errno = EINVAL;
188  return 0;
189  }
190 
191  memcpy(data, &m_data[m_rest_pos], m_rest_size);
192 
193  m_rest_size = m_rest_pos = 0;
194 
195  return result_len;
196  }
197 
198  m_cur_pos = static_cast<int64_t>(bd_tell(m_bd));
199  if (m_end_pos < 0 || m_cur_pos < m_end_pos)
200  {
201  int maxsize = sizeof(m_data);
202 
203  if (maxsize > (m_end_pos - m_cur_pos))
204  {
205  maxsize = static_cast<int>(m_end_pos - m_cur_pos);
206  }
207 
208  int res = bd_read(m_bd, m_data, maxsize);
209  if (res < 0)
210  {
211  Logging::error(path(), "bd_read has failed.");
212  return 0;
213  }
214 
215  size_t bytes = static_cast<size_t>(res);
216 
217  m_cur_pos = static_cast<int64_t>(bd_tell(m_bd));
218 
219  if (bytes > size)
220  {
221  result_len = size;
222  memcpy(data, m_data, result_len);
223 
224  m_rest_size = bytes - size;
225  m_rest_pos = size;
226  }
227  else
228  {
229  result_len = bytes;
230  memcpy(data, m_data, result_len);
231  }
232  }
233 
234  return result_len;
235 }

257 int BlurayIO::seek(int64_t offset, int whence)
258 {
259  int64_t seek_pos;
260 
261  switch (whence)
262  {
263  case SEEK_SET:
264  {
265  seek_pos = m_start_pos + offset;
266  break;
267  }
268  case SEEK_CUR:
269  {
270  seek_pos = m_start_pos + offset + static_cast<int64_t>(bd_tell(m_bd));
271  break;
272  }
273  case SEEK_END:
274  {
275  seek_pos = m_end_pos + offset;
276  break;
277  }
278  default:
279  {
280  errno = EINVAL;
281  return (EOF);
282  }
283  }
284 
285  if (seek_pos > m_end_pos)
286  {
287  m_cur_pos = m_end_pos; // Cannot go beyond EOF. Set position to end, leave errno untouched.
288  return 0;
289  }
290 
291  if (seek_pos < 0) // Cannot go before head, set errno.
292  {
293  errno = EINVAL;
294  return (EOF);
295  }
296 
297  m_cur_pos = bd_seek(m_bd, static_cast<uint64_t>(seek_pos));
298 
299  return (m_cur_pos == seek_pos ? 0 : -1);
300 }
