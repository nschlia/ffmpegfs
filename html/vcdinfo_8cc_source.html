127int VcdInfo::load_file(const std::string & path)
128{
129 std::string fullname;
130 bool is_svcd = false;
131
132 clear();
133
134 if (!VCDUTILS::locate_file(path, "INFO", fullname, is_svcd))
135 {
136 return errno;
137 }
138
139 VCDUTILS::get_directory(path, &m_disk_path);
140
141 FILE *fpi;
142 VCDINFO vi;
143 bool success = true;
144
145 fpi = fopen(fullname.c_str(), "rb");
146 if (fpi == nullptr)
147 {
148 return errno;
149 }
150
151 struct stat stbuf;
152
153 if (fstat(fileno(fpi), &stbuf) != 0)
154 {
155 return ferror(fpi);
156 }
157
158 m_file_date = stbuf.st_mtime;
159
160 std::memset(&vi, 0, sizeof(vi));
161
162 if (fread(reinterpret_cast<char *>(&vi), 1, sizeof(vi), fpi) == sizeof(vi))
163 {
164 m_id = VCDUTILS::convert_txt2string(vi.m_ID, sizeof(vi.m_ID));
165 m_type = static_cast<VCDTYPE>(vi.m_type);
166 m_profile_tag = static_cast<VCDPROFILETAG>(vi.m_profile_tag);
167 m_album_id = VCDUTILS::convert_txt2string(vi.m_albumid, sizeof(vi.m_albumid));
168 m_number_of_cds = htons(static_cast<uint16_t>(vi.m_numberof_cds));
169 m_cd_number = htons(static_cast<uint16_t>(vi.m_cd_number));
170 }
171
172 else
173 {
174 success = false;
175 }
176
177 int orgerrno = 0;
178
179 if (success)
180 {
181 orgerrno = ferror(fpi);
182 }
183
184 fclose(fpi);
185
186 return orgerrno;
187}
