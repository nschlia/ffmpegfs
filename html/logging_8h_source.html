287  template <typename T, typename... Args>
288  static std::string format_helper(
289  const std::string &string_to_search,
290  const size_t index_to_replace,
291  T &&val,
292  Args &&...args)
293  {
294  // Match %# exactly (e.g. %12 and %123 literally)
295 
296  std::regex exp("%(<([^>]+)>)*" + std::to_string(index_to_replace) + "(?=[^0-9]|$)");
297  std::smatch res;
298  std::string string_to_update(string_to_search);
299  std::string::const_iterator searchStart(string_to_search.cbegin());
300  size_t offset = 0;
301 
302  while (std::regex_search(searchStart, string_to_search.cend(), res, exp))
303  {
304  std::ostringstream ostr;
305 
306  if (res[2].length())
307  {
308  // Found match with printf format in res[2]
309  size_t size = static_cast<size_t>(std::snprintf(nullptr, 0, res[2].str().c_str(), val)) + 1;
310  std::vector<char> buffer;
311  buffer.resize(size);
312  std::snprintf(buffer.data(), size, res[2].str().c_str(), val);
313  ostr << buffer.data();
314  }
315  else
316  {
317  // No printf format, replace literally
318  ostr << val;
319  }
320 
321  string_to_update.replace(static_cast<size_t>(res.position()) + offset, static_cast<size_t>(res[0].length()), ostr.str());
322 
323  offset += static_cast<size_t>(res.position()) + ostr.str().length();
324 
325  searchStart = res.suffix().first;
326  }
327 
328  return format_helper(
329  string_to_update,
330  index_to_replace + 1,
331  std::forward<Args>(args)...);
332  }
