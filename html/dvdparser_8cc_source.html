78 static int dvd_find_best_audio_stream(const vtsi_mat_t *vtsi_mat, int *best_channels, int *best_sample_frequency)
79 {
80  int best_stream = -1;
81  int best_application_mode = INT_MAX;
82  int best_lang_extension = INT_MAX;
83  int best_quantization = 0;
84 
85  *best_channels = 0;
86  *best_sample_frequency = 0;
87 
88  for(int i = 0; i < vtsi_mat->nr_of_vts_audio_streams; i++)
89  {
90  const audio_attr_t *attr = &vtsi_mat->vts_audio_attr[i];
91 
92  if (attr->audio_format == 0
93  && attr->multichannel_extension == 0
94  && attr->lang_type == 0
95  && attr->application_mode == 0
96  && attr->quantization == 0
97  && attr->sample_frequency == 0
98  && attr->unknown1 == 0
99  && attr->channels == 0
100  && attr->lang_extension == 0
101  && attr->unknown3 == 0)
102  {
103  // Unspecified
104  continue;
105  }
106 
107  // Preference in this order, if higher value is same, compare next and so on.
108  //
109  // application_mode: prefer not specified.
110  // 0: not specified
111  // 1: karaoke mode
112  // 2: surround sound mode
113  // lang_extension: prefer not specified or normal audio
114  // 0: Not specified
115  // 1: Normal audio/Caption
116  // 2: visually impaired
117  // 3: Director's comments 1
118  // 4: Director's comments 2
119  // sample_frequency: maybe 48K only
120  // 0: 48kHz
121  // 1: ??kHz
122  // quantization prefer highest bit width or drc
123  // 0: 16bit
124  // 1: 20bit
125  // 2: 24bit
126  // 3: drc
127  // channels: prefer no extension
128  // multichannel_extension
129 
130  // if ((best_multiframe > multiframe) ||
131  // (best_multiframe == multiframe && best_bitrate > bitrate) ||
132  // (best_multiframe == multiframe && best_bitrate == bitrate && best_count >= count))
133 
134  // Specs keep the meaning of the values of this field other than 0 secret, so we nail it to 48 kHz.
135  int sample_frequency = 48000;
136 
137  if ((best_application_mode < attr->application_mode) ||
138  (best_application_mode == attr->application_mode && best_lang_extension < attr->lang_extension) ||
139  (best_application_mode == attr->application_mode && best_lang_extension == attr->lang_extension && *best_sample_frequency > sample_frequency) ||
140  (best_application_mode == attr->application_mode && best_lang_extension == attr->lang_extension && *best_sample_frequency == sample_frequency && *best_channels > attr->channels) ||
141  (best_application_mode == attr->application_mode && best_lang_extension == attr->lang_extension && *best_sample_frequency == sample_frequency && *best_channels == attr->channels && best_quantization > attr->quantization)
142  )
143  {
144  continue;
145  }
146 
147  best_stream = i;
148  best_application_mode = attr->application_mode;
149  best_lang_extension = attr->lang_extension;
150  *best_sample_frequency = sample_frequency;
151  *best_channels = attr->channels;
152  best_quantization = attr->quantization;
153  }
154 
155  if (best_stream > -1)
156  {
157  ++*best_channels;
158  }
159 
160  return best_stream;
161 }

168 static AVRational dvd_frame_rate(const uint8_t * ptr)
169 {
170  AVRational framerate = { 0, 0 };
171 
172  // 11 = 30 fps, 10 = illegal, 01 = 25 fps, 00 = illegal
173  unsigned fps = ((ptr[3] & 0xC0) >> 6) & 0x03;
174 
175  switch (fps)
176  {
177  case 3: // PAL
178  {
179  framerate = av_make_q(25000, 1000);
180  break;
181  }
182  case 1: // NTSC
183  {
184  framerate = av_make_q(30000, 10001);
185  break;
186  }
187  default:
188  {
189  // Frame rate is illegal, so we need to set anything. Assume PAL.
190  framerate = av_make_q(25000, 1000);
191  break;
192  }
193  }
194 
195  return framerate;
196 }

203 static int64_t BCDtime(const dvd_time_t * dvd_time)
204 {
205  std::array<int64_t, 4> time;
206  AVRational framerate = dvd_frame_rate(&dvd_time->frame_u);
207 
208  if (!framerate.den)
209  {
210  framerate = av_make_q(25000, 1000); // Avoid divisions by 0
211  }
212  time[0] = dvd_time->hour;
213  time[1] = dvd_time->minute;
214  time[2] = dvd_time->second;
215  time[3] = dvd_time->frame_u & 0x3F; // Number of frame
216 
217  // convert BCD (two digits) to binary
218  for (int64_t & tm : time)
219  {
220  tm = ((tm & 0xf0) >> 4) * 10 + (tm & 0x0f);
221  }
222 
223  return (AV_TIME_BASE * (time[0] * 3600 + time[1] * 60 + time[2]) + static_cast<int64_t>(static_cast<double>(AV_TIME_BASE * time[3]) / av_q2d(framerate)));
224 }

248  int pgcnum = vts_ptt_srpt->title[ttnnum - 1].ptt[chapter_idx].pgcn;

254  bool interleaved = false;
255  int start_cell = cur_pgc->program_map[pgn - 1] - 1;
256  int end_cell = 0;
257 
258  if (pgn < cur_pgc->nr_of_programs && !full_title)
259  {
260  end_cell = cur_pgc->program_map[pgn] - 1;
261  }
262  else
263  {
264  end_cell = cur_pgc->nr_of_cells;
265  }
266 
267  interleaved = cur_pgc->cell_playback[start_cell].interleaved ? true : false;
268  framerate = dvd_frame_rate(&cur_pgc->cell_playback[start_cell].playback_time.frame_u);
269 
270  bool has_angles = false;
271 
272  for (int cell_no = start_cell; cell_no < end_cell; cell_no++)
273  {
274  cell_playback_t *cell_playback = &cur_pgc->cell_playback[cell_no];
275 
276  // Only count normal cells and the first of an angle to avoid duplicate sizes
277  if (cell_playback->block_mode == static_cast<unsigned int>(BLOCK_MODE_NOT_IN_BLOCK) || cell_playback->block_mode == static_cast<unsigned int>(BLOCK_MODE_FIRST_CELL))
278  {
279  size += (cell_playback->last_sector - cell_playback->first_sector) * 2048;
280  duration += BCDtime(&cell_playback->playback_time);
281  }
282 
283  if (cell_playback->block_type == static_cast<unsigned int>(BLOCK_TYPE_ANGLE_BLOCK))
284  {
285  has_angles = true;
286  }
287  }
288 
289  if (duration < params.m_min_dvd_chapter_duration * AV_TIME_BASE)
290  {
291  Logging::debug(nullptr, "Skipping short DVD chapter.");
292  return true;
293  }
294 
295  if (!has_angles)
296  {
297  // If this chapter has no angle cells, reset angles to 1
298  angles = 1;
299  }
300 
301  // Split file if chapter has several angles
302  for (int angle_idx = 0; angle_idx < angles; angle_idx++)
303  {
304  std::string title_buf;
305  int angle_no = angle_idx + 1;
306 
307  // can safely assume this a video
308  if (!full_title)
309  {
310  // Single chapter
311  if (angles > 1)
312  {
313  strsprintf(&title_buf, "%02d. Chapter %03d (Angle %d) [%s].%s",
314  title_no,
315  chapter_no,
316  angle_no,
317  replace_all(format_duration(duration), ":", "-").c_str(),
318  ffmpeg_format[0].fileext().c_str());
319  }
320  else
321  {
322  strsprintf(&title_buf, "%02d. Chapter %03d [%s].%s",
323  title_no,
324  chapter_no,
325  replace_all(format_duration(duration), ":", "-").c_str(),
326  ffmpeg_format[0].fileext().c_str());
327  }
328  }
329  else
330  {
331  // Full title
332  if (angles > 1)
333  {
334  strsprintf(&title_buf, "%02d. Title (Angle %d) [%s].%s",
335  title_no,
336  angle_no,
337  replace_all(format_duration(duration), ":", "-").c_str(),
338  ffmpeg_format[0].fileext().c_str());
339  }
340  else
341  {
342  strsprintf(&title_buf, "%02d. Title [%s].%s",
343  title_no,
344  replace_all(format_duration(duration), ":", "-").c_str(),
345  ffmpeg_format[0].fileext().c_str());
346  }
347  }
348 
349  LPVIRTUALFILE virtualfile = nullptr;
350  if (!ffmpeg_format[0].is_multiformat())
351  {
352  virtualfile = insert_file(VIRTUALTYPE_DVD, path + title_buf, statbuf);
353  }
354  else
355  {
356  virtualfile = insert_dir(VIRTUALTYPE_DVD, path + title_buf, statbuf);
357  }
358 
359  if (virtualfile == nullptr)
360  {
361  Logging::error(path, "Failed to create virtual path: %1", (path + title_buf).c_str());
362  errno = EIO;
363  return false;
364  }
365 
366  if (add_fuse_entry(buf, filler, title_buf, &virtualfile->m_st, 0))
367  {
368  // break;
369  }
370 
371  // DVD is video format anyway
372  virtualfile->m_format_idx = 0;
373  // Mark title/chapter/angle
374  virtualfile->m_full_title = full_title;
375  virtualfile->m_dvd.m_title_no = title_no;
376  virtualfile->m_dvd.m_chapter_no = chapter_no;
377  virtualfile->m_dvd.m_angle_no = angle_no;
378 
379  if (!transcoder_cached_filesize(virtualfile, &virtualfile->m_st))
380  {
381  virtualfile->m_duration = duration;
382 
383  BITRATE video_bit_rate = 8*1024*1024; // In case the real bitrate cannot be calculated later, assume 8 Mbit video bitrate
384  if (duration)
385  {
391  video_bit_rate = static_cast<BITRATE>(size * 8LL * AV_TIME_BASE / static_cast<uint64_t>(duration)); // calculate bitrate in bps
392  }
393 
394  Logging::trace(virtualfile->m_destfile, "Video %1 %2x%3@%<%5.2f>4%5 fps %6 [%7]", format_bitrate(video_settings.m_video_bit_rate).c_str(), video_settings.m_width, video_settings.m_height, av_q2d(framerate), interleaved ? "i" : "p", format_size(size).c_str(), format_duration(duration).c_str());
395 
396  virtualfile->m_width = video_settings.m_width;
397  virtualfile->m_height = video_settings.m_height;
398  virtualfile->m_framerate = framerate;
399 
400  if (audio_stream > -1)
401  {
402  Logging::trace(virtualfile->m_destfile, "Audio Channels %1 Sample Rate %2", audio_settings.m_channels, audio_settings.m_sample_rate);
403 
404  virtualfile->m_channels = audio_settings.m_channels;
405  virtualfile->m_sample_rate = audio_settings.m_sample_rate;
406  }
407 
408  transcoder_set_filesize(virtualfile, duration, audio_settings.m_audio_bit_rate, audio_settings.m_channels, audio_settings.m_sample_rate, AV_SAMPLE_FMT_NONE, video_bit_rate, video_settings.m_width, video_settings.m_height, interleaved, framerate);
409 
410  virtualfile->m_video_frame_count = static_cast<uint32_t>(av_rescale_q(duration, av_get_time_base_q(), av_inv_q(framerate)));
411  virtualfile->m_predicted_size = static_cast<size_t>(size);
412  }
413  }
414 
415  return true;

426 static int parse_dvd(const std::string & path, const struct stat *statbuf, void *buf, fuse_fill_dir_t filler)
427 {
428  dvd_reader_t *dvd;
429  ifo_handle_t *ifo_file;
430  tt_srpt_t *tt_srpt;
431  int titles;
432  bool success = true;
433 
434  Logging::debug(path, "Parsing the DVD.");
435 
436  dvd = DVDOpen(path.c_str());
437  if (dvd == nullptr)
438  {
439  Logging::error(path, "Couldn't open the DVD.");
440  return ENOENT;
441  }
442 
443  ifo_file = ifoOpen(dvd, 0);
444  if (ifo_file == nullptr)
445  {
446  Logging::error(path, "Can't open VMG info for the DVD.");
447  DVDClose(dvd);
448  return -EINVAL;
449  }
450  tt_srpt = ifo_file->tt_srpt;
451 
452  titles = tt_srpt->nr_of_srpts;
453 
454  Logging::debug(path, "There are %1 titles on this DVD.", titles);
455 
456  for (int title_idx = 0; title_idx < titles && success; ++title_idx)
457  {
458  ifo_handle_t *vts_file;
459  int vtsnum = tt_srpt->title[title_idx].title_set_nr;
460  int ttnnum = tt_srpt->title[title_idx].vts_ttn;
461  int chapters = tt_srpt->title[title_idx].nr_of_ptts;
462  int angles = tt_srpt->title[title_idx].nr_of_angles;
463 
464  Logging::trace(path, "Title: %1 VTS: %2 TTN: %3", title_idx + 1, vtsnum, ttnnum);
465  Logging::trace(path, "DVD title has %1 chapters and %2 angles.", chapters, angles);
466 
467  vts_file = ifoOpen(dvd, vtsnum);
468  if (vts_file == nullptr)
469  {
470  Logging::error(path, "Can't open info file for title %1.", vtsnum);
471  DVDClose(dvd);
472  return -EINVAL;
473  }
474 
475  // Set reasonable defaults
476  AUDIO_SETTINGS audio_settings;
477  audio_settings.m_audio_bit_rate = 256000;
478  audio_settings.m_channels = 2;
479  audio_settings.m_sample_rate = 48000;
480  int audio_stream = 0;
481 
482  VIDEO_SETTINGS video_settings;
483  video_settings.m_video_bit_rate = 8000000;
484  video_settings.m_width = 720;
485  video_settings.m_height = 576;
486 
487  if (vts_file->vtsi_mat)
488  {
489  audio_stream = dvd_find_best_audio_stream(vts_file->vtsi_mat, &audio_settings.m_channels, &audio_settings.m_sample_rate);
490 
491  video_settings.m_height = (vts_file->vtsi_mat->vts_video_attr.video_format != 0) ? 576 : 480;
492 
493  switch(vts_file->vtsi_mat->vts_video_attr.picture_size)
494  {
495  case 0:
496  {
497  video_settings.m_width = 720;
498  break;
499  }
500  case 1:
501  {
502  video_settings.m_width = 704;
503  break;
504  }
505  case 2:
506  {
507  video_settings.m_width = 352;
508  break;
509  }
510  case 3:
511  {
512  video_settings.m_width = 352;
513  video_settings.m_height /= 2;
514  break;
515  }
516  default:
517  {
518  Logging::warning(path, "DVD video contains invalid picture size attribute.");
519  }
520  }
521  }
522 
523  // Add separate chapters
524  for (int chapter_idx = 0; chapter_idx < chapters && success; ++chapter_idx)
525  {
526  success = create_dvd_virtualfile(vts_file, path, statbuf, buf, filler, false, title_idx, chapter_idx, angles, ttnnum, audio_stream, audio_settings, video_settings);
527  }
528 
529  if (success && chapters > 1)
530  {
531  // If more than 1 chapter, add full title as well
532  success = create_dvd_virtualfile(vts_file, path, statbuf, buf, filler, true, title_idx, 0, 1, ttnnum, audio_stream, audio_settings, video_settings);
533  }
534 
535  ifoClose(vts_file);
536  }
537 
538  ifoClose(ifo_file);
539  DVDClose(dvd);
540 
541  if (success)
542  {
543  return titles; // Number of titles on disk
544  }
545  else
546  {
547  return -errno;
548  }
549 }

551 int check_dvd(const std::string & path, void *buf, fuse_fill_dir_t filler)
552 {
553  std::string _path(path);
554  struct stat stbuf;
555  int res = 0;
556 
557  append_sep(&_path);
558 
559  if (stat((_path + "VIDEO_TS.IFO").c_str(), &stbuf) == 0 || stat((_path + "VIDEO_TS/VIDEO_TS.IFO").c_str(), &stbuf) == 0)
560  {
561  if (!check_path(_path))
562  {
563  Logging::trace(_path, "DVD detected.");
564  res = parse_dvd(_path, &stbuf, buf, filler);
565  Logging::trace(_path, "%1 titles were discovered.", res);
566  }
567  else
568  {
569  res = load_path(_path, &stbuf, buf, filler);
570  }
571 
572  add_dotdot(buf, filler, &stbuf, 0);
573  }
574  return res;
