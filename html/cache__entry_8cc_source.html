57 Cache_Entry::~Cache_Entry()
58 {
59  if (m_thread_id && m_thread_id != pthread_self())
60  {
61  pthread_t thread_id = m_thread_id;
62  // If not same thread, wait for other to finish
63  Logging::warning(filename(), "Waiting for thread id 0x%<%" FFMPEGFS_FORMAT_PTHREAD_T ">1 to terminate.", thread_id);
64 
65  int s = pthread_join(m_thread_id, nullptr);
66  if (s != 0)
67  {
68  Logging::error(filename(), "Error joining thread id 0x%<%" FFMPEGFS_FORMAT_PTHREAD_T ">1 : %2", thread_id, strerror(s));
69  }
70  else
71  {
72  Logging::info(filename(), "Thread id 0x%<%" FFMPEGFS_FORMAT_PTHREAD_T ">1 has terminated.", thread_id);
73  }
74  }
75 
76  delete m_buffer;
77 
78  unlock();
79 
80  Logging::trace(filename(), "Deleted buffer.");
81 }

175 bool Cache_Entry::open(bool create_cache /*= true*/)
176 {
177  if (m_buffer == nullptr)
178  {
179  errno = EINVAL;
180  return false;
181  }
182 
183  if (__sync_fetch_and_add(&m_ref_count, 1) > 0)
184  {
185  return true;
186  }
187 
188  bool erase_cache = !read_info(); // If read_info fails, rebuild cache entry
189 
190  if (!create_cache)
191  {
192  return true;
193  }
194 
195  if (!m_cache_info.m_finished)
196  {
197  // If no database entry found (database is not consistent),
198  // or file was not completely transcoded last time,
199  // simply create a new file.
200  erase_cache = true;
201  }
202 
203  Logging::trace(filename(), "Last transcode finished: %1 Erase cache: %2.", m_cache_info.m_finished, erase_cache);
204 
205  // Store access time
206  update_access(true);
207 
208  // Open the cache
209  if (m_buffer->init(erase_cache))
210  {
211  return true;
212  }
213  else
214  {
215  clear(false);
216  return false;
217  }
218 }

231 bool Cache_Entry::close(int flags)
232 {
233  write_info();
234 
235  if (m_buffer == nullptr)
236  {
237  errno = EINVAL;
238  return false;
239  }
240 
241  if (!m_ref_count)
242  {
243  close_buffer(flags);
244 
245  return true;
246  }
247 
248  if (__sync_sub_and_fetch(&m_ref_count, 1) > 0)
249  {
250  // Just flush to disk
251  flush();
252  return false;
253  }
254 
255  close_buffer(flags);
256 
257  return true;

345 bool Cache_Entry::outdated() const
346 {
347  struct stat sb;
348 
349  if (m_cache_info.m_audiobitrate != params.m_audiobitrate)
350  {
351  Logging::debug(filename(), "Triggering re-transcode: Selected audio bitrate changed from %1 to %2.", m_cache_info.m_audiobitrate, params.m_audiobitrate);
352  return true;
353  }
354 
355  if (m_cache_info.m_audiosamplerate != params.m_audiosamplerate)
356  {
357  Logging::debug(filename(), "Triggering re-transcode: Selected audio samplerate changed from %1 to %2.", m_cache_info.m_audiosamplerate, params.m_audiosamplerate);
358  return true;
359  }
360 
361  if (m_cache_info.m_videobitrate != params.m_videobitrate)
362  {
363  Logging::debug(filename(), "Triggering re-transcode: Selected video bitrate changed from %1 to %2.", m_cache_info.m_audiobitrate, params.m_audiobitrate);
364  return true;
365  }
366 
367  if (m_cache_info.m_videowidth != params.m_videowidth || m_cache_info.m_videoheight != params.m_videoheight)
368  {
369  Logging::debug(filename(), "Triggering re-transcode: Selected video witdh/height changed.");
370  return true;
371  }
372 
373 #ifndef USING_LIBAV
374  if (m_cache_info.m_deinterlace != params.m_deinterlace)
375  {
376  Logging::debug(filename(), "Triggering re-transcode: Selected video deinterlace changed from %1 to %2.", m_cache_info.m_deinterlace, params.m_deinterlace);
377  return true;
378  }
379 #endif // !USING_LIBAV
380 
381  if (stat(filename().c_str(), &sb) != -1)
382  {
383  // If source file exists, check file date/size
384  if (m_cache_info.m_file_time < sb.st_mtime)
385  {
386  Logging::debug(filename(), "Triggering re-transcode: File time has gone forward.");
387  return true;
388  }
389 
390  if (m_cache_info.m_file_size != static_cast<size_t>(sb.st_size))
391  {
392  Logging::debug(filename(), "Triggering re-transcode: File size has changed.");
393  return true;
394  }
395  }
396 
397  return false;
398 }
