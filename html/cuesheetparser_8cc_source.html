122 static bool create_cuesheet_virtualfile(Track *track, int titleno, const std::string & filename, const std::string & path, const struct stat * statbuf, int trackcount, int trackno, const std::string & album, const std::string & genre, const std::string & date, int channels, int sample_rate, int64_t *remainingduration, LPVIRTUALFILE *lastfile)
123 {
124  Cdtext *cuesheetcdtext = track_get_cdtext(track);
125  if (cuesheetcdtext == nullptr)
126  {
127  Logging::error(filename, "Unable to get track cd text from cue sheet.");
128  errno = EIO;
129  return false;
130  }
131 
132  std::string performer = VAL_OR_EMPTY(cdtext_get(PTI_PERFORMER, cuesheetcdtext));
133  std::string title = VAL_OR_EMPTY(cdtext_get(PTI_TITLE, cuesheetcdtext));
134 
135  int64_t start = AV_TIME_BASE * static_cast<int64_t>(track_get_start(track)) / FPS;
136  int64_t length = track_get_length(track);
137  int64_t duration;
138 
139  if (length > -1)
140  {
141  duration = AV_TIME_BASE * length / FPS;
142  *remainingduration -= duration;
143  }
144  else
145  {
146  // Length of
147  duration = *remainingduration;
148  }
149 
150  char title_buf[PATH_MAX + 1];
151 
152  snprintf(title_buf, sizeof(title_buf) - 1, "%02d. %s - %s [%s].%s",
153  titleno,
154  performer.c_str(),
155  title.c_str(),
156  replace_all(format_duration(duration), ":", "-").c_str(),
157  ffmpeg_format[0].fileext().c_str());
158 
159  std::string virtfilename(title_buf);
160  // Filenames can't contain '/' in POSIX etc.
161  std::replace(virtfilename.begin(), virtfilename.end(), '/', '-');
162 
163  LPVIRTUALFILE virtualfile = nullptr;
164  if (!ffmpeg_format[0].is_multiformat())
165  {
166  virtualfile = insert_file(VIRTUALTYPE_DISK, path + virtfilename, filename, statbuf, VIRTUALFLAG_CUESHEET);
167  }
168  else
169  {
170  virtualfile = insert_dir(VIRTUALTYPE_DISK, path + virtfilename, statbuf, VIRTUALFLAG_CUESHEET);
171  }
172 
173  if (virtualfile == nullptr)
174  {
175  Logging::error(path, "Failed to create virtual path: %1", (path + virtfilename).c_str());
176  errno = EIO;
177  return false;
178  }
179 
180  // We do not add the file to fuse here because it's in a sub directory.
181  // Will be done later on request by load_path()
182 
183  virtualfile->m_format_idx = 0;
184 
185  if (!transcoder_cached_filesize(virtualfile, &virtualfile->m_st))
186  {
187  BITRATE video_bit_rate = 0;
188  BITRATE audio_bit_rate = 0;
189 
190  int width = 0;
191  int height = 0;
192  AVRational framerate = { 0, 0 };
193  int interleaved = 0;
194 
195  virtualfile->m_duration = duration;
196  virtualfile->m_cuesheet.m_duration = duration;
197  virtualfile->m_cuesheet.m_start = start;
198  virtualfile->m_cuesheet.m_tracktotal = trackcount;
199  virtualfile->m_cuesheet.m_trackno = trackno;
200  virtualfile->m_cuesheet.m_artist = performer;
201  virtualfile->m_cuesheet.m_title = title;
202  virtualfile->m_cuesheet.m_album = album;
203  virtualfile->m_cuesheet.m_genre = genre;
204  virtualfile->m_cuesheet.m_date = date;
205  if (*lastfile != nullptr)
206  {
207  (*lastfile)->m_cuesheet.m_nextfile = virtualfile;
208  }
209  *lastfile = virtualfile;
210 
211  transcoder_set_filesize(virtualfile, duration, audio_bit_rate, channels, sample_rate, AV_SAMPLE_FMT_NONE, video_bit_rate, width, height, interleaved, framerate);
212 
213  stat_set_size(&virtualfile->m_st, virtualfile->m_predicted_size);
214  }
215 
216  return true;
217 }

255 static int parse_cuesheet(const std::string & filename, Cd *cd, const struct stat *statbuf, void *buf, fuse_fill_dir_t filler, AVFormatContext *fmt_ctx)
256 {
257  int res = 0;
258 
259  try
260  {
261  if (cd == nullptr)
262  {
263  Logging::error(filename, "Unable to parse cue sheet.");
264  throw AVERROR(EIO);
265  }
266 
267  Rem *rem = cd_get_rem(cd);
268  if (rem == nullptr)
269  {
270  Logging::error(filename, "Unable to parse remarks from cue sheet.");
271  throw AVERROR(EIO);
272  }
273 
274  Cdtext *cdtext = cd_get_cdtext(cd);
275  if (cdtext == nullptr)
276  {
277  Logging::error(filename, "Unable to get cd text from cue sheet.");
278  throw AVERROR(EIO);
279  }
280 
281  std::string performer = VAL_OR_EMPTY(cdtext_get(PTI_PERFORMER, cdtext));
282  std::string album = VAL_OR_EMPTY(cdtext_get(PTI_TITLE, cdtext));
283  std::string genre = VAL_OR_EMPTY(cdtext_get(PTI_GENRE, cdtext));
284  std::string date = VAL_OR_EMPTY(rem_get(REM_DATE, rem));
285 
286  int trackcount = static_cast<int>(cd_get_ntrack(cd));
287  if (trackcount)
288  {
289  LPVIRTUALFILE virtualfile = nullptr;
290  std::string subbdir(filename);
291 
292  append_ext(&subbdir, TRACKDIR);
293 
294  std::string dirname(subbdir);
295 
296  append_sep(&subbdir);
297  remove_path(&dirname);
298 
299  virtualfile = insert_dir(VIRTUALTYPE_DISK, subbdir, statbuf, VIRTUALFLAG_CUESHEET);
300 
301  if (virtualfile == nullptr)
302  {
303  Logging::error(filename, "Failed to create virtual path: %1", subbdir.c_str());
304  errno = EIO;
305  return -errno;
306  }
307 
308  if (buf != nullptr && filler(buf, dirname.c_str(), &virtualfile->m_st, 0))
309  {
310  // break;
311  }
312 
313  std::string path(filename);
314 
315  remove_filename(&path);
316 
317  LPVIRTUALFILE lastfile = nullptr;
318  int64_t remainingduration = fmt_ctx->duration;
319  int channels = 0;
320  int samplerate = 0;
321 
322  int ret = get_audio_props(fmt_ctx, &channels, &samplerate);
323  if (ret < 0)
324  {
325  Logging::error(filename, "Could not find audio stream in input file (error '%1').", ffmpeg_geterror(ret).c_str());
326 
327  errno = EIO;
328  throw -errno;
329  }
330 
331  for (int trackno = 1; trackno <= trackcount; trackno++)
332  {
333  Track *track = cd_get_track(cd, trackno);
334  if (track == nullptr)
335  {
336  Logging::error(filename, "Unable to get track no. %1 from cue sheet.", trackno);
337  errno = EIO;
338  throw -errno;
339  }
340 
341  if (!create_cuesheet_virtualfile(track, trackno, filename, path + dirname + "/", statbuf, trackcount, trackno, album, genre, date, channels, samplerate, &remainingduration, &lastfile))
342  {
343  throw -errno;
344  }
345  }
346  }
347 
348  res = trackcount;
349  }
350  catch (int _res)
351  {
352  res = _res;
353  }
354 
355  if (cd != nullptr)
356  {
357  cd_delete(cd);
358  }
359 
360  return res;
361 }

371 static int parse_embedded_cuesheet(const std::string & filename, void *buf, fuse_fill_dir_t filler, AVFormatContext *fmt_ctx)
372 {
373  int res = 0;
374 
375  try
376  {
377  struct stat stbuf;
378 
379  if (lstat(filename.c_str(), &stbuf) != 0)
380  {
381  // Can be silently ignored
382  throw 0;
383  }
384 
385  AVDictionaryEntry *tag = av_dict_get(fmt_ctx->metadata, "CUESHEET", nullptr, AV_DICT_IGNORE_SUFFIX);
386  if (tag != nullptr)
387  {
388  // Found cue sheet
389  Logging::trace(filename, "Found embedded cue sheet.");
390 
391  std::string result(tag->value);
392 
393  result += "\r\n"; // cue_parse_string() reports syntax error if string does not end with newline
394  replace_all(&result, "\r\n", "\n"); // Convert all to unix
395 
396  res = parse_cuesheet(filename, cue_parse_string(result.c_str()), &stbuf, buf, filler, fmt_ctx);
397  }
398  }
399  catch (int _res)
400  {
401  res = _res;
402  }
403 
404  return res;

407 int check_cuesheet(const std::string & filename, void *buf, fuse_fill_dir_t filler, AVFormatContext *fmt_ctx)
408 {
409  std::string _filename(filename);
410  std::string trackdir(filename);
411  std::string cuesheet(filename);
412  struct stat stbuf;
413  int res = 0;
414 
415  std::string ext;
416  if (find_ext(&ext, _filename) && ext == TRACKDIR)
417  {
418  remove_ext(&cuesheet); // remove TRACKDIR extension from virtual directory
419  remove_ext(&_filename);
420  }
421  else
422  {
423  append_ext(&trackdir, TRACKDIR);
424  append_sep(&trackdir);
425  }
426  replace_ext(&cuesheet, "cue");
427 
428  AVFormatContext *_fmt_ctx = nullptr;
429  try
430  {
431  if (lstat(_filename.c_str(), &stbuf) == -1)
432  {
433  // Can be silently ignored
434  throw 0;
435  }
436 
437  if (fmt_ctx == nullptr)
438  {
439  // If no format context was passed, we'll have to create a new one
440  Logging::debug(_filename, "Creating new format context and checking for cue sheet");
441  res = avformat_open_input(&_fmt_ctx, _filename.c_str(), nullptr, nullptr);
442  if (res)
443  {
444  Logging::warning(_filename, "Unable to scan file for embedded queue sheet: %1", ffmpeg_geterror(res).c_str());
445  errno = -ENOENT;
446  throw -errno;
447  }
448 
449  res = avformat_find_stream_info(_fmt_ctx, nullptr);
450  if (res < 0)
451  {
452  Logging::warning(_filename, "Cannot find stream information: %1", ffmpeg_geterror(res).c_str());
453  errno = -EINVAL;
454  throw -errno;
455  }
456 
457  fmt_ctx = _fmt_ctx;
458  }
459 
460  if (lstat(cuesheet.c_str(), &stbuf) != -1)
461  {
462  if (!check_path(trackdir))
463  {
464  // Not a virtual directory
465  res = parse_cuesheet(_filename, cuesheet, &stbuf, buf, filler, fmt_ctx);
466  Logging::trace(cuesheet, "Found %1 titles.", res);
467  }
468  else
469  {
470  // Obviously a virtual directory
471  std::string dirname(trackdir);
472 
473  remove_path(&dirname);
474 
475  LPVIRTUALFILE virtualfile = find_file(trackdir);
476 
477  if (virtualfile == nullptr)
478  {
479  Logging::error(_filename, "Failed to find virtual path: %1", trackdir.c_str());
480  errno = EIO;
481  throw -errno;
482  }
483 
484  if (buf != nullptr && filler(buf, dirname.c_str(), &virtualfile->m_st, 0))
485  {
486  // break;
487  }
488 
489  res = 0;
490  }
491  }
492  else
493  {
494  // No cue sheet file, check if one is embedded in media file
495  res = parse_embedded_cuesheet(_filename, buf, filler, fmt_ctx);
496  }
497  }
498  catch (int _res)
499  {
500  res = _res;
501  }
502 
503  if (_fmt_ctx != nullptr)
504  {
505  // If we created a format context we also have to close it.
506  avformat_close_input(&_fmt_ctx);
507  }
508 
509  return res;
510 }
