80 bool Buffer::init(bool erase_cache)
81 {
82  std::lock_guard<std::recursive_mutex> lck (m_mutex);
83 
84  if (is_open())
85  {
86  return true;
87  }
88 
89  bool success = true;
90 
91  make_cachefile_name(m_cachefile, filename(), params.current_format(virtualfile())->fileext());
92 
93  try
94  {
95  // Create the path to the cache file
96  char *cachefile = new_strdup(m_cachefile);
97 
98  if (cachefile == nullptr)
99  {
100  Logging::error(m_cachefile, "Error opening cache file: Out of memory");
101  errno = ENOMEM;
102  throw false;
103  }
104 
105  if (mktree(dirname(cachefile), S_IRWXU | S_IRWXG | S_IROTH | S_IXOTH) && errno != EEXIST)
106  {
107  Logging::error(m_cachefile, "Error creating cache directory: (%1) %2", errno, strerror(errno));
108  delete [] cachefile;
109  throw false;
110  }
111  errno = 0; // reset EEXIST, error can safely be ignored here
112 
113  delete [] cachefile;
114 
115  m_fd = -1;
116  m_buffer = nullptr;
117  m_buffer_pos = 0;
118  m_buffer_watermark = 0;
119  m_buffer_size = 0;
120 
121  if (erase_cache)
122  {
123  remove_cachefile();
124  errno = 0; // ignore this error
125  }
126 
127  size_t filesize = 0;
128  bool isdefaultsize = false;
129  uint8_t *p = nullptr;
130 
131  if (!map_file(m_cachefile, &m_fd, &p, &filesize, &isdefaultsize, 0))
132  {
133  throw false;
134  }
135 
136  if (!isdefaultsize)
137  {
138  m_buffer_pos = m_buffer_watermark = filesize;
139  }
140 
141  m_buffer_size = filesize;
142  m_buffer = static_cast<uint8_t*>(p);
143  }
144  catch (bool _success)
145  {
146  success = _success;
147 
148  if (!success)
149  {
150  m_fd = -1;
151  m_buffer = nullptr;
152  m_buffer_pos = 0;
153  m_buffer_watermark = 0;
154  m_buffer_size = 0;
155  }
156  }
157 
158  return success;
159 }

161 bool Buffer::map_file(const std::string & filename, int *fd, uint8_t **p, size_t *filesize, bool *isdefaultsize, off_t defaultsize) const
162 {
163  bool success = true;
164 
165  Logging::trace(filename, "Mapping cache file.");
166 
167  try
168  {
169  struct stat sb;
170 
171  *fd = ::open(filename.c_str(), O_CREAT | O_RDWR, static_cast<mode_t>(0644));
172  if (*fd == -1)
173  {
174  Logging::error(filename, "Error opening cache file: (%1) %2", errno, strerror(errno));
175  throw false;
176  }
177 
178  if (fstat(*fd, &sb) == -1)
179  {
180  Logging::error(filename, "File stat failed: (%1) %2 (fd = %3)", errno, strerror(errno), *fd);
181  throw false;
182  }
183 
184  if (!S_ISREG(sb.st_mode))
185  {
186  Logging::error(filename, "Not a file.");
187  throw false;
188  }
189 
190  if (!sb.st_size || *isdefaultsize)
191  {
192  // If file is empty or did not exist set file size to default
193 
194  if (!defaultsize)
195  {
196  defaultsize = sysconf(_SC_PAGESIZE);
197  }
198 
199  if (ftruncate(*fd, defaultsize) == -1)
200  {
201  Logging::error(filename, "Error calling ftruncate() to 'stretch' the file: (%1) %2 (fd = %3)", errno, strerror(errno), *fd);
202  throw false;
203  }
204 
205  *filesize = static_cast<size_t>(defaultsize);
206  *isdefaultsize = true;
207  }
208  else
209  {
210  // Keep size
211  *filesize = static_cast<size_t>(sb.st_size);
212  *isdefaultsize = false;
213  }
214 
215  *p = static_cast<uint8_t *>(mmap(nullptr, *filesize, PROT_READ | PROT_WRITE, MAP_SHARED, *fd, 0));
216  if (*p == MAP_FAILED)
217  {
218  Logging::error(filename, "File mapping failed: (%1) %2 (fd = %3)", errno, strerror(errno), *fd);
219  throw false;
220  }
221  }
222  catch (bool _success)
223  {
224  success = _success;
225 
226  if (!success && *fd != -1)
227  {
228  ::close(*fd);
229  *fd = -1;
230  }
231  }
232 
233  return success;
234 }

236 bool Buffer::unmap_file(const std::string &filename, int * fd, uint8_t **p, size_t * filesize, size_t *buffer_pos) const
237 {
238  bool success = true;
239 
240  Logging::trace(filename, "Unmapping cache file.");
241 
242  void * __p = *p;
243  size_t __filesize = *filesize;
244  int __fd = *fd;
245 
246  // Clear all variables
247  p = nullptr;
248  *filesize = 0;
249  if (buffer_pos != nullptr)
250  {
251  *buffer_pos = 0;
252  }
253  *fd = -1;
254 
255  if (__p != nullptr)
256  {
257  if (munmap(__p, __filesize ? __filesize : static_cast<size_t>(sysconf(_SC_PAGESIZE))) == -1) // Make sure we do not unmap a zero size file (spitzs EINVBAL error)
258  {
259  Logging::error(filename, "Unmapping cache file failed: (%1) %2 %3", errno, strerror(errno), __filesize);
260  fprintf(stderr, "P = %p size = %zi\n", __p, __filesize);
261  success = false;
262  }
263  }
264 
265  if (__fd != -1)
266  {
267  if (__filesize)
268  {
269  if (ftruncate(__fd, static_cast<off_t>(__filesize)) == -1)
270  {
271  Logging::error(filename, "Error calling ftruncate() to resize and close the cache file: (%1) %2 (fd = %3)", errno, strerror(errno), __fd);
272  success = false;
273  }
274  ::close(__fd);
275  }
276  else
277  {
278  ::close(__fd);
279 
280  if (unlink(filename.c_str()))
281  {
282  Logging::error(filename, "Error removing the cache file: (%1) %2 (fd = %3)", errno, strerror(errno), __fd);
283  success = false;
284  }
285  }
286  }
287 
288  return success;

457 int Buffer::seek(int64_t offset, int whence)
458 {
459  if (m_buffer == nullptr)
460  {
461  errno = EBADF;
462  return (EOF);
463  }
464 
465  off_t seek_pos;
466 
467  switch (whence)
468  {
469  case SEEK_SET:
470  {
471  seek_pos = offset;
472  break;
473  }
474  case SEEK_CUR:
475  {
476  seek_pos = static_cast<off_t>(tell()) + offset;
477  break;
478  }
479  case SEEK_END:
480  {
481  seek_pos = static_cast<off_t>(size()) + offset;
482  break;
483  }
484  default:
485  {
486  errno = EINVAL;
487  return (EOF);
488  }
489  }
490 
491  if (seek_pos > static_cast<off_t>(size()))
492  {
493  m_buffer_pos = size(); // Cannot go beyond EOF. Set position to end, leave errno untouched.
494  return 0;
495  }
496 
497  if (seek_pos < 0) // Cannot go before head, leave position untouched, set errno.
498  {
499  errno = EINVAL;
500  return (EOF);
501  }
502 
503  m_buffer_pos = static_cast<size_t>(seek_pos);
504  return 0;
505 }
