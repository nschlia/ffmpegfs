68int VcdIO::openio(LPVIRTUALFILE virtualfile)
69{
70 std::string src_filename;
71
72 set_virtualfile(virtualfile);
73
74 if (virtualfile != nullptr)
75 {
76 m_full_title = virtualfile->m_full_title;
77 m_track_no = virtualfile->m_vcd.m_track_no;
78 m_chapter_no = virtualfile->m_vcd.m_chapter_no;
79 m_start_pos = virtualfile->m_vcd.m_start_pos;
80 m_end_pos = virtualfile->m_vcd.m_end_pos;
81 }
82 else
83 {
84 m_full_title = false;
85 m_track_no = 1;
86 m_chapter_no = 0;
87 m_start_pos = 0;
88 m_end_pos = size();
89 }
90
91 VCDUTILS::locate_video(path(), m_track_no, src_filename);
92
93 Logging::info(src_filename.c_str(), "Opening the input VCD.");
94
95 m_fpi = fopen(src_filename.c_str(), "rb");
96
97 if (m_fpi == nullptr)
98 {
99 return errno;
100 }
101
102 return seek(0, SEEK_SET);
103}

105size_t VcdIO::readio(void * data, size_t size)
106{
107 if (static_cast<size_t>(ftell(m_fpi)) + size > m_end_pos)
108 {
109 size = static_cast<size_t>(m_end_pos - static_cast<uint64_t>(ftell(m_fpi)));
110 }
111
112 if (!size)
113 {
114 return 0;
115 }
116
117 return fread(data, 1, size, m_fpi);
118}

130size_t VcdIO::size() const
131{
132 if (m_fpi == nullptr)
133 {
134 errno = EINVAL;
135 return 0;
136 }
137
138 if (m_end_pos)

153int VcdIO::seek(int64_t offset, int whence)
154{
155 off_t seek_pos;
156 switch (whence)
157 {
158 case SEEK_SET:
159 {
160 seek_pos = static_cast<off_t>(m_start_pos) + offset;
161 break;
162 }
163 case SEEK_CUR:
164 {
165 seek_pos = static_cast<off_t>(m_start_pos) + ftell(m_fpi) + offset;
166 break;
167 }
168 case SEEK_END:
169 {
170 seek_pos = static_cast<off_t>(m_end_pos) - offset;
171 break;
172 }
173 default:
174 {
175 errno = EINVAL;
176 return (EOF);
177 }
178 }
179
180 if (static_cast<uint64_t>(seek_pos) > m_end_pos)
181 {
182 seek_pos = static_cast<off_t>(m_end_pos); // Cannot go beyond EOF. Set position to end, leave errno untouched.
183 }
184
185 if (static_cast<uint64_t>(seek_pos) < m_start_pos) // Cannot go before head, leave position untouched, set errno.
186 {
187 errno = EINVAL;
188 return (EOF);
189 }
190
191 return fseek(m_fpi, static_cast<long int>(seek_pos), SEEK_SET);
192}
