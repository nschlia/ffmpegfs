284 AVCodecID Format_Options::subtitle_codec(AVCodecID codec_id) const
285 {
286  FORMAT_MAP::const_iterator it = m_format_map.find(params.m_sample_fmt);
287  if (it == m_format_map.cend())
288  {
289  // Output supports no subtitles. End of story.
290  return AV_CODEC_ID_NONE;
291  }
292 
293  // Try to find direct match, prefer same as input stream
294  for (const AVCodecID & subtitle_codec : it->second.m_subtitle_codec)
295  {
296  // Also match AV_CODEC_ID_DVD_SUBTITLE to AV_CODEC_ID_DVB_SUBTITLE
297  if (subtitle_codec == codec_id || (codec_id == AV_CODEC_ID_DVD_SUBTITLE && subtitle_codec == AV_CODEC_ID_DVB_SUBTITLE))
298  {
299  return subtitle_codec;
300  }
301  }
302 
303  // No direct match, try to find a text/text or bitmap/bitmap pair
304  if (is_text_codec(codec_id))
305  {
306  // Find a text based codec in the list
307  for (const AVCodecID & subtitle_codec : it->second.m_subtitle_codec)
308  {
309  if (is_text_codec(subtitle_codec))
310  {
311  return subtitle_codec;
312  }
313  }
314  }
315  else
316  {
317  // Find a bitmap based codec in the list
318  for (const AVCodecID & subtitle_codec : it->second.m_subtitle_codec)
319  {
320  if (!is_text_codec(subtitle_codec))
321  {
322  return subtitle_codec;
323  }
324  }
325  }
326 
327  // No matching codec support
328  return AV_CODEC_ID_NONE;
329 }
