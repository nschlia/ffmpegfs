436 /*
437  * *******************************************************************************************************************
438  *
439  * webm
440  * ====
441  *
442  * https://trac.ffmpeg.org/wiki/Encode/VP9
443  *
444  * Controlling Speed and Quality
445  * =============================
446  *
447  * libvpx-vp9 has two main control knobs for speed and quality:
448  *
449  * Deadline / Quality
450  *
451  * -deadline can be set to realtime, good, or best. For legacy reasons, the option is also accessible with -quality in ffmpeg.
452  *
453  * good is the default and recommended for most applications.
454  * best is recommended if you have lots of time and want the best compression efficiency.
455  * realtime is recommended for live / fast encoding.
456  *
457  * CPU Utilization / Speed
458  * =======================
459  *
460  * -cpu-used sets how efficient the compression will be. For legacy reasons, the option is also accessible with -speed in ffmpeg.
461  *
462  * When the deadline/quality parameter is good or best, values for -cpu-used can be set between 0 and 5. The default is 0.
463  * Using 1 or 2 will increase encoding speed at the expense of having some impact on quality and rate control accuracy.
464  * 4 or 5 will turn off rate distortion optimization, having even more of an impact on quality.
465  *
466  * When the deadline/quality is set to realtime, the encoder will try to hit a particular CPU utilization target, and encoding quality
467  * will depend on CPU speed and the complexity of the clip that you are encoding. See â€‹the vpx documentation for more info. In this case,
468  * the valid values for -cpu-used are between 0 and 15, where the CPU utilization target (in per cent) is calculated as
469  * (100*(16-cpu-used)/16)%.
470  *
471  * *******************************************************************************************************************
472  *
473  */

484 // ffmpeg -i <source> -c:v libvpx-vp9 -pass 2 -b:v 1000K -threads 8 -speed 1
485 // -tile-columns 6 -frame-parallel 1 -auto-alt-ref 1 -lag-in-frames 25
486 // -c:a libopus -b:a 64k -f webm out.webm
487 
488 
489 // Most of the current VP9 decoders use tile-based, multi-threaded decoding. In order for the decoders to take advantage
490 // of multiple cores, the encoder must set tile-columns and frame-parallel.
491 
492 // Setting auto-alt-ref and lag-in-frames >= 12 will turn on VP9's alt-ref frames, a VP9 feature that enhances quality.
493 
494 // speed 4 tells VP9 to encode really fast, sacrificing quality. Useful to speed up the first pass.
495 
496 // speed 1 is a good speed vs. quality compromise. Produces output quality typically very close to speed 0, but usually encodes much faster.
497 
498 // Multi-threaded encoding may be used if -threads > 1 and -tile-columns > 0.
499 
500 
501 // { "threads", "8", 0, 0 },
502 // { "speed", "4", 0, 0 },
503  { "tile-columns", "6", 0, 0 },
504  { "frame-parallel", "1", 0, 0 },
505  { "auto-alt-ref", "1", 0, 0 },
506  { "lag-in-frames", "25", 0, 0 },
507 
508  { nullptr, nullptr, 0, 0 }
