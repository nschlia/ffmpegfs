303 static int ffmpegfs_readdir(const char *path, void *buf, fuse_fill_dir_t filler, off_t /*offset*/, struct fuse_file_info * /*fi*/)
304 {
305  std::string origpath;
306 
307  Logging::trace(path, "readdir");
308 
309  append_basepath(&origpath, path);
310  append_sep(&origpath);
311 
312  // Add a virtual script if enabled
313  if (params.m_enablescript)
314  {
315  LPVIRTUALFILE virtualfile = make_file(buf, filler, VIRTUALTYPE_SCRIPT, origpath, params.m_scriptfile, script_file.size());
316  virtualfile->m_file_contents = script_file;
317  }
318 
319  LPVIRTUALFILE virtualfile = find_original(origpath);
320 
321  if (virtualfile == nullptr)
322  {
323 #if defined(USE_LIBBLURAY) || defined(USE_LIBDVD) || defined(USE_LIBVCD)
324  int res;
325 #endif
326 
327 #ifdef USE_LIBVCD
328  res = check_vcd(origpath, buf, filler);
329  if (res != 0)
330  {
331  // Found VCD or error reading VCD
332  return (res >= 0 ? 0 : res);
333  }
334 #endif // USE_LIBVCD
335 #ifdef USE_LIBDVD
336  res = check_dvd(origpath, buf, filler);
337  if (res != 0)
338  {
339  // Found DVD or error reading DVD
340  return (res >= 0 ? 0 : res);
341  }
342 #endif // USE_LIBDVD
343 #ifdef USE_LIBBLURAY
344  res = check_bluray(origpath, buf, filler);
345  if (res != 0)
346  {
347  // Found Bluray or error reading Bluray
348  return (res >= 0 ? 0 : res);
349  }
350 #endif // USE_LIBBLURAY
351  }
352 
353  if (virtualfile == nullptr || !(virtualfile->m_flags & VIRTUALFLAG_FILESET))
354  {
355  DIR *dp = opendir(origpath.c_str());
356  if (dp != nullptr)
357  {
358  try
359  {
360  std::map<const std::string, struct stat> files;
361 
362  // Read directory contents
363  for (struct dirent *de = readdir(dp); de != nullptr; de = readdir(dp))
364  {
365  struct stat stbuf;
366 
367  if (lstat((origpath + de->d_name).c_str(), &stbuf) == -1)
368  {
369  // Should actually not happen, file listed by readdir, so it should exist
370  throw ENOENT;
371  }
372 
373  files.insert({ de->d_name, stbuf });
374  }
375 
376  // Process files
377  for (std::map<const std::string, struct stat>::iterator it = files.begin(); it != files.end(); ++it)
378  {
379  std::string origname(it->first);
380 
381  if (is_blocked(origname))
382  {
383  continue;
384  }
385 
386  std::string origfile;
387  std::string filename(it->first);
388  struct stat & stbuf = it->second;
389  int flags = 0;
390 
391  origfile = origpath + origname;
392 
393  std::string origext;
394  find_ext(&origext, filename);
395 
396  if (S_ISREG(stbuf.st_mode) || S_ISLNK(stbuf.st_mode))
397  {
398  const FFmpegfs_Format *current_format = nullptr;
399  // Check if file can be transcoded
400  if (virtual_name(&filename, origpath, &current_format))
401  {
402  AVFormatContext *fmt_ctx = nullptr;
403  int res;
404 
405  res = avformat_open_input(&fmt_ctx, origfile.c_str(), nullptr, nullptr);
406  if (res)
407  {
408  Logging::warning(origfile, "Unable to open file: %1", ffmpeg_geterror(res).c_str());
409  throw ENOENT;
410  }
411 
412  res = avformat_find_stream_info(fmt_ctx, nullptr);
413  if (res < 0)
414  {
415  Logging::warning(origfile, "Cannot find stream information: %1", ffmpeg_geterror(res).c_str());
416  throw EINVAL;
417  }
418 
419  if (current_format->video_codec() == AV_CODEC_ID_NONE)
420  {
421  // If target supports no video, we need to do some extra work and check
422  // the input file to actually have an audio stream. If not, hide the file,
423  // makes no sense to transcode anyway.
424  res = av_find_best_stream(fmt_ctx, AVMEDIA_TYPE_AUDIO, -1, -1, nullptr, 0);
425  if (res < 0 && res != AVERROR_STREAM_NOT_FOUND)
426  {
427  Logging::warning(origfile, "Could not find %1 stream in input file (error '%2').", get_media_type_string(AVMEDIA_TYPE_AUDIO), ffmpeg_geterror(res).c_str());
428  throw EINVAL;
429  }
430 
431  if (res == AVERROR_STREAM_NOT_FOUND && current_format->video_codec() == AV_CODEC_ID_NONE)
432  {
433  Logging::debug(origfile, "Unable to transcode, source has no audio stream, but target just supports audio.");
434  flags |= VIRTUALFLAG_HIDDEN;
435  }
436  }
437 
438  if (!(flags & VIRTUALFLAG_HIDDEN))
439  {
440  // Check if we have a cue sheet
441  res = check_cuesheet(origfile, buf, filler, fmt_ctx);
442  if (res < 0)
443  {
444  throw EINVAL;
445  }
446  }
447 
448  if (fmt_ctx != nullptr)
449  {
450  avformat_close_input(&fmt_ctx);
451  }
452 
453  std::string newext;
454  find_ext(&newext, filename);
455 
456  if (!current_format->is_multiformat())
457  {
458  if (origext != newext || params.m_recodesame == RECODESAME_YES)
459  {
460  insert_file(VIRTUALTYPE_DISK, origpath + filename, origfile, &stbuf, flags);
461  }
462  else
463  {
464  insert_file(VIRTUALTYPE_DISK, origpath + filename, origfile, &stbuf, flags | VIRTUALFLAG_PASSTHROUGH);
465  }
466  }
467  else
468  {
469  flags |= VIRTUALFLAG_FILESET | VIRTUALFLAG_DIRECTORY;
470 
471  // Change file to directory for the frame set
472  stbuf.st_mode &= ~static_cast<mode_t>(S_IFREG | S_IFLNK);
473  stbuf.st_mode |= S_IFDIR;
474  stbuf.st_size = stbuf.st_blksize;
475 
476  filename = origname; // Restore original name
477 
478  if (current_format->is_frameset())
479  {
480  flags |= VIRTUALFLAG_FRAME;
481  }
482  else if (current_format->is_hls())
483  {
484  flags |= VIRTUALFLAG_HLS;
485  }
486 
487  insert_file(VIRTUALTYPE_DISK, origfile, &stbuf, flags);
488  }
489  }
490  }
491 
492  if (!(flags & VIRTUALFLAG_HIDDEN))
493  {
494  if (add_fuse_entry(buf, filler, filename.c_str(), &stbuf, 0))
495  {
496  break;
497  }
498  }
499  }
500 
501  closedir(dp);
502 
503  errno = 0; // Just to make sure - reset any error
504  }
505  catch (int _errno)
506  {
507  closedir(dp);
508 
509  errno = _errno;
510  }
511  }
512  }
513  else
514  {
515  try
516  {
517  if (virtualfile->m_flags & (VIRTUALFLAG_FILESET | VIRTUALFLAG_FRAME | VIRTUALFLAG_HLS | VIRTUALFLAG_DIRECTORY))
518  {
519  add_dotdot(buf, filler, &virtualfile->m_st, 0);
520  }
521 
522  const FFmpegfs_Format *ffmpegfs_format = params.current_format(virtualfile);
523 
524  if (ffmpegfs_format->is_frameset())
525  {
526  // Generate set of all frames
527  if (!virtualfile->m_video_frame_count)
528  {
529  int res = get_source_properties(origpath, virtualfile);
530  if (res < 0)
531  {
532  throw EINVAL;
533  }
534  }
535 
536  //Logging::debug(origpath, "readdir: Creating frame set of %1 frames. %2", virtualfile->m_video_frame_count, virtualfile->m_origfile);
537 
538  for (uint32_t frame_no = 1; frame_no <= virtualfile->m_video_frame_count; frame_no++)
539  {
540  make_file(buf, filler, virtualfile->m_type, origpath, make_filename(frame_no, params.current_format(virtualfile)->fileext()), virtualfile->m_predicted_size, virtualfile->m_st.st_ctime, VIRTUALFLAG_FRAME);
541  }
542  }
543  else if (ffmpegfs_format->is_hls())
544  {
545  int res = make_hls_fileset(buf, filler, origpath, virtualfile);
546  if (res < 0)
547  {
548  throw EINVAL;
549  }
550  }
551  else if (/*virtualfile != nullptr && */virtualfile->m_flags & VIRTUALFLAG_CUESHEET)
552  {
553  // Fill in list for cue sheet
554  load_path(origpath, nullptr, buf, filler);
555  }
556 
557  errno = 0; // Just to make sure - reset any error
558  }
559  catch (int _errno)
560  {
561  errno = _errno;
562  }
563  }
564 
565  return -errno;
566 }

574 static int ffmpegfs_getattr(const char *path, struct stat *stbuf)
575 {
576  Logging::trace(path, "getattr");
577 
578  if (is_blocked(path))
579  {
580  errno = ENOENT;
581  return -errno;
582  }
583 
584  std::string origpath;
585 
586  append_basepath(&origpath, path);
587 
588  LPVIRTUALFILE virtualfile = find_original(&origpath);
589  VIRTUALTYPE type = (virtualfile != nullptr) ? virtualfile->m_type : VIRTUALTYPE_DISK;
590  int flags = (virtualfile != nullptr) ? virtualfile->m_flags : VIRTUALFLAG_NONE;
591 
592  if (virtualfile != nullptr && (virtualfile->m_flags & VIRTUALFLAG_HIDDEN))
593  {
594  errno = ENOENT;
595  return -errno;
596  }
597 
598  if (virtualfile == nullptr && lstat(origpath.c_str(), stbuf) == 0)
599  {
600  // File was not yet created as virtual file, but physically exists
601  const FFmpegfs_Format *current_format = nullptr;
602  std::string filename(origpath);
603 
604  if (virtual_name(&filename, "", &current_format))
605  {
606  if (!current_format->is_multiformat())
607  {
608  // Pass-through for regular files
609  Logging::trace(origpath, "getattr: Not transcoding existing file.");
610  errno = 0;
611  return 0;
612  }
613  else
614  {
615  Logging::trace(origpath, "getattr: Creating frame set directory of file.");
616 
617  flags |= VIRTUALFLAG_FILESET;
618 
619  if (current_format->is_frameset())
620  {
621  flags |= VIRTUALFLAG_FRAME;
622  }
623  else if (current_format->is_hls())
624  {
625  flags |= VIRTUALFLAG_HLS;
626  }
627  }
628  }
629  else
630  {
631  // Pass-through for regular files
632  Logging::trace(origpath, "getattr: Treating existing file as passthrough.");
633  errno = 0;
634  return 0;
635  }
636  }
637  else if ((flags & VIRTUALFLAG_PASSTHROUGH) && lstat(origpath.c_str(), stbuf) == 0)
638  {
639  // File physically exists and is marked as passthrough
640  Logging::debug(origpath, "getattr: File not recoded because --recodesame=NO.");
641  errno = 0;
642  return 0;
643  }
644  else
645  {
646  // Not really an error.
647  errno = 0;
648  }
649 
650  // This is a virtual file
651  bool no_check = false;
652 
653  switch (type)
654  {
655  case VIRTUALTYPE_SCRIPT:
