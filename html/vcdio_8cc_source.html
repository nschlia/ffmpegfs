153 int VcdIO::seek(int64_t offset, int whence)
154 {
155  off_t seek_pos;
156  switch (whence)
157  {
158  case SEEK_SET:
159  {
160  seek_pos = static_cast<off_t>(m_start_pos) + offset;
161  break;
162  }
163  case SEEK_CUR:
164  {
165  seek_pos = static_cast<off_t>(m_start_pos) + ftell(m_fpi) + offset;
166  break;
167  }
168  case SEEK_END:
169  {
170  seek_pos = static_cast<off_t>(m_end_pos) - offset;
171  break;
172  }
173  default:
174  {
175  errno = EINVAL;
176  return (EOF);
177  }
178  }
179 
180  if (static_cast<uint64_t>(seek_pos) > m_end_pos)
181  {
182  seek_pos = static_cast<off_t>(m_end_pos); // Cannot go beyond EOF. Set position to end, leave errno untouched.
183  }
184 
185  if (static_cast<uint64_t>(seek_pos) < m_start_pos) // Cannot go before head, leave position untouched, set errno.
186  {
187  errno = EINVAL;
188  return (EOF);
189  }
190 
191  return fseek(m_fpi, static_cast<long int>(seek_pos), SEEK_SET);
192 }
