132 int VcdInfo::load_file(const std::string & path)
133 {
134  std::string fullname;
135  bool is_svcd = false;
136 
137  clear();
138 
139  if (!VCDUTILS::locate_file(path, "INFO", fullname, is_svcd))
140  {
141  return errno;
142  }
143 
144  VCDUTILS::get_directory(path, &m_disk_path);
145 
146  FILE *fpi;
147  VCDINFO vi;
148  bool success = true;
149 
150  fpi = fopen(fullname.c_str(), "rb");
151  if (fpi == nullptr)
152  {
153  return errno;
154  }
155 
156  struct stat stbuf;
157 
158  if (fstat(fileno(fpi), &stbuf) != 0)
159  {
160  return ferror(fpi);
161  }
162 
163  m_file_date = stbuf.st_mtime;
164 
165  memset(&vi, 0, sizeof(vi));
166 
167  if (fread(reinterpret_cast<char *>(&vi), 1, sizeof(vi), fpi) == sizeof(vi))
168  {
169  m_id = VCDUTILS::convert_txt2string(vi.m_ID, sizeof(vi.m_ID));
170  m_type = static_cast<VCDTYPE>(vi.m_type);
171  m_profile_tag = static_cast<VCDPROFILETAG>(vi.m_profile_tag);
172  m_album_id = VCDUTILS::convert_txt2string(vi.m_albumid, sizeof(vi.m_albumid));
173  m_number_of_cds = htons(static_cast<uint16_t>(vi.m_numberof_cds));
174  m_cd_number = htons(static_cast<uint16_t>(vi.m_cd_number));
175  }
176 
177  else
178  {
179  success = false;
180  }
181 
182  int orgerrno = 0;
183 
184  if (success)
185  {
186  orgerrno = ferror(fpi);
187  }
188 
189  fclose(fpi);
190 
191  return orgerrno;
192 }
