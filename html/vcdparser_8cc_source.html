59 static bool create_vcd_virtualfile(const VcdEntries & vcd, const struct stat * statbuf, void * buf, fuse_fill_dir_t filler, bool full_title, int chapter_no)
60 {
61  const VcdChapter * chapter1 = vcd.get_chapter(chapter_no);
62  std::string title_buf;
63  size_t size;
64  int64_t duration;
65 
66  if (!full_title)
67  {
68  size = static_cast<size_t>(chapter1->get_size());
69  duration = chapter1->get_duration();
70  strsprintf(&title_buf, "%02d. Chapter %03d [%s].%s", chapter1->get_track_no(), chapter_no + 1, replace_all(format_duration(duration), ":", "-").c_str(), ffmpeg_format[0].fileext().c_str()); // can safely assume this a video
71  }
72  else
73  {
74  size = static_cast<size_t>(vcd.get_size());
75  duration = vcd.get_duration();
76  strsprintf(&title_buf, "%02d. Title [%s].%s", chapter1->get_track_no(), replace_all(format_duration(duration), ":", "-").c_str(), ffmpeg_format[0].fileext().c_str()); // can safely assume this a video
77  }
78 
79  LPVIRTUALFILE virtualfile = nullptr;
80  if (!ffmpeg_format[0].is_multiformat())
81  {
82  virtualfile = insert_file(VIRTUALTYPE_VCD, vcd.get_disk_path() + title_buf, statbuf);
83  }
84  else
85  {
86  virtualfile = insert_dir(VIRTUALTYPE_VCD, vcd.get_disk_path() + title_buf, statbuf);
87  }
88 
89  if (virtualfile == nullptr)
90  {
91  Logging::error(vcd.get_disk_path(), "Failed to create virtual path: %1", (vcd.get_disk_path() + title_buf).c_str());
92  errno = EIO;
93  return false;
94  }
95 
96  if (add_fuse_entry(buf, filler, title_buf, &virtualfile->m_st, 0))
97  {
98  // break;
99  }
100 
101  // Video CD is video format anyway
102  virtualfile->m_format_idx = 0;
103  // Mark title/chapter/angle
104  virtualfile->m_full_title = full_title;
105  virtualfile->m_vcd.m_track_no = chapter1->get_track_no();
106  virtualfile->m_vcd.m_chapter_no = chapter_no;
107  virtualfile->m_vcd.m_start_pos = chapter1->get_start_pos();
108  if (!full_title)
109  {
110  virtualfile->m_vcd.m_end_pos = chapter1->get_end_pos();
111  }
112  else
113  {
114  virtualfile->m_vcd.m_end_pos = size;
115  }
116  virtualfile->m_duration = duration;
117  AVRational framerate = av_make_q(25000, 1000); //*** @todo check disk which framerate is correct, can be 25 or 29.996 fps!
118  virtualfile->m_video_frame_count = static_cast<uint32_t>(av_rescale_q(duration, av_get_time_base_q(), av_inv_q(framerate)));
119  virtualfile->m_predicted_size = size;
120 
121  return true;
122 }

132 static int parse_vcd(const std::string & path, const struct stat * statbuf, void * buf, fuse_fill_dir_t filler)
133 {
134  VcdEntries vcd;
135  bool success = true;
136 
137  vcd.load_file(path);
138 
139  Logging::debug(path, "Parsing the Video CD.");
140 
141  for (int chapter_no = 0; chapter_no < vcd.get_number_of_chapters() && success; chapter_no++)
142  {
143  success = create_vcd_virtualfile(vcd, statbuf, buf, filler, false, chapter_no);
144  }
145 
146  if (success && vcd.get_number_of_chapters() > 1)
147  {
148  success = create_vcd_virtualfile(vcd, statbuf, buf, filler, true, 0);
149  }
150 
151  if (success)
152  {
153  return vcd.get_number_of_chapters();
154  }
155  else
156  {
157  return -errno;
158  }
159 }

161 int check_vcd(const std::string & path, void *buf, fuse_fill_dir_t filler)
162 {
163  std::string _path(path);
164  struct stat stbuf;
165  int res = 0;
166 
167  append_sep(&_path);
168 
169  if (stat((_path + "SVCD/INFO.SVD").c_str(), &stbuf) == 0)
170  {
171  if (!check_path(_path))
172  {
173  Logging::trace(_path, "SVCD detected.");
174  res = parse_vcd(_path, &stbuf, buf, filler);
175  Logging::trace(nullptr, "%1 titles were discovered.", res);
176  }
177  else
178  {
179  res = load_path(_path, &stbuf, buf, filler);
180  }
181 
182  add_dotdot(buf, filler, &stbuf, 0);
183  }
184  else if (stat((_path + "VCD/INFO.VCD").c_str(), &stbuf) == 0)
185  {
186  if (!check_path(_path))
187  {
188  Logging::trace(_path, "VCD detected.");
189  res = parse_vcd(_path, &stbuf, buf, filler);
190  Logging::trace(nullptr, "%1 titles were discovered.", res);
191  }
192  else
193  {
194  res = load_path(_path, &stbuf, buf, filler);
195  }
196 
197  add_dotdot(buf, filler, &stbuf, 0);
198  }
199  return res;
200 }
