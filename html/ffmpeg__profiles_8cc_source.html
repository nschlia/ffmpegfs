476 /*
477  * *******************************************************************************************************************
478  *
479  * webm
480  * ====
481  *
482  * https://trac.ffmpeg.org/wiki/Encode/VP9
483  *
484  * Controlling Speed and Quality
485  * =============================
486  *
487  * libvpx-vp9 has two main control knobs for speed and quality:
488  *
489  * Deadline / Quality
490  *
491  * -deadline can be set to realtime, good, or best. For legacy reasons, the option is also accessible with -quality in ffmpeg.
492  *
493  * good is the default and recommended for most applications.
494  * best is recommended if you have lots of time and want the best compression efficiency.
495  * realtime is recommended for live / fast encoding.
496  *
497  * CPU Utilization / Speed
498  * =======================
499  *
500  * -cpu-used sets how efficient the compression will be. For legacy reasons, the option is also accessible with -speed in ffmpeg.
501  *
502  * When the deadline/quality parameter is good or best, values for -cpu-used can be set between 0 and 5. The default is 0.
503  * Using 1 or 2 will increase encoding speed at the expense of having some impact on quality and rate control accuracy.
504  * 4 or 5 will turn off rate distortion optimization, having even more of an impact on quality.
505  *
506  * When the deadline/quality is set to realtime, the encoder will try to hit a particular CPU utilization target, and encoding quality
507  * will depend on CPU speed and the complexity of the clip that you are encoding. See the vpx documentation for more info. In this case,
508  * the valid values for -cpu-used are between 0 and 15, where the CPU utilization target (in per cent) is calculated as
509  * (100*(16-cpu-used)/16)%.
510  *
511  * *******************************************************************************************************************
512  *
513  */
514 
518 static const FFmpeg_Profiles::PROFILE_OPTION m_option_webm_codec_default[] =
519 {
520  { "deadline", "realtime", 0, OPT_SW_ONLY },

522  { "cpu-used", "8", 0, OPT_SW_ONLY },
523 
524  // ffmpeg -i <source> -c:v libvpx-vp9 -pass 2 -b:v 1000K -threads 8 -speed 1
525  // -tile-columns 6 -frame-parallel 1 -auto-alt-ref 1 -lag-in-frames 25
526  // -c:a libopus -b:a 64k -f webm out.webm
527 
528  // Most of the current VP9 decoders use tile-based, multi-threaded decoding. In order for the decoders to take advantage
529  // of multiple cores, the encoder must set tile-columns and frame-parallel.
530  // Setting auto-alt-ref and lag-in-frames >= 12 will turn on VP9's alt-ref frames, a VP9 feature that enhances quality.
531  // speed 4 tells VP9 to encode really fast, sacrificing quality. Useful to speed up the first pass.
532  // speed 1 is a good speed vs. quality compromise. Produces output quality typically very close to speed 0, but usually encodes much faster.
533  // Multi-threaded encoding may be used if -threads > 1 and -tile-columns > 0.
534 
535  //{ "threads", "8", 0, OPT_SW_ONLY },
536  //{ "speed", "4", 0, OPT_SW_ONLY },
537  { "tile-columns", "6", 0, OPT_SW_ONLY },
538  { "frame-parallel", "1", 0, OPT_SW_ONLY },
539  { "auto-alt-ref", "1", 0, OPT_SW_ONLY },
540  { "lag-in-frames", "25", 0, OPT_SW_ONLY },
