107 Logging::Logger::~Logger()
108 {
109  std::lock_guard<std::recursive_mutex> lck (m_mutex);
110 
111  // Construct string containing time
112  time_t now = time(nullptr);
113  struct tm buffer;
114  std::string time_string(30, '\0');
115  std::string loglevel;
116  std::string filename;
117  std::string msg;
118 
119  time_string.resize(strftime(&time_string[0], time_string.size(), "%F %T", localtime_r(&now, &buffer))); // Mind the blank at the end
120 
121  loglevel = m_level_name_map.at(m_loglevel) + ":";
122 
123  msg = str();
124  trim(msg);
125 
126  if (!m_filename.empty())
127  {
128  filename = m_filename;
129 
130  if (replace_start(&filename, params.m_basepath))
131  {
132  filename = "INPUT [" + filename + "] ";
133  }
134 
135  else if (replace_start(&filename, params.m_mountpath))
136  {
137  filename = "OUTPUT [" + filename + "] ";
138  }
139  else
140  {
141  std::string cachepath;
142 
143  transcoder_cache_path(cachepath);
144 
145  if (replace_start(&filename, cachepath + params.m_mountpath + params.m_basepath))
146  {
147  filename = "CACHE [" + filename + "] ";
148  }
149  else
150  {
151  filename = "OTHER [" + filename + "] ";
152  }
153  }
154  }
155 
156  if (m_logging->m_to_syslog)
157  {
158  syslog(m_syslog_level_map.at(m_loglevel), "%s %s%s", loglevel.c_str(), filename.c_str(), msg.c_str());
159  }
160 
161  if (m_logging->m_logfile.is_open())
162  {
163  m_logging->m_logfile << time_string << " " << loglevel << " " << filename << msg << std::endl;
164  }
165 
166  if (m_logging->m_to_stderr)
167  {
168  if (!filename.empty())
169  {
170  filename = COLOUR_LIGHT_PURPLE + filename + COLOUR_RESET;
171  }
172 
173  if (m_loglevel <= LOGERROR)
174  {
175  msg = COLOUR_LIGHT_RED + msg + COLOUR_RESET;
176  }
177 
178  std::clog << COLOUR_DARK_GRAY << time_string << " " << loglevel << COLOUR_RESET << " " << filename << msg << std::endl;
179  }
180 }
