87 static int dvd_find_best_audio_stream(const vtsi_mat_t *vtsi_mat, int *best_channels, int *best_sample_frequency)
88 {
89  int best_stream = -1;
90  int best_application_mode = INT_MAX;
91  int best_lang_extension = INT_MAX;
92  int best_quantization = 0;
93 
94  *best_channels = 0;
95  *best_sample_frequency = 0;
96 
97  for(int i = 0; i < vtsi_mat->nr_of_vts_audio_streams; i++)
98  {
99  const audio_attr_t *attr = &vtsi_mat->vts_audio_attr[i];
100 
101  if (attr->audio_format == 0
102  && attr->multichannel_extension == 0
103  && attr->lang_type == 0
104  && attr->application_mode == 0
105  && attr->quantization == 0
106  && attr->sample_frequency == 0
107  && attr->unknown1 == 0
108  && attr->channels == 0
109  && attr->lang_extension == 0
110  && attr->unknown3 == 0)
111  {
112  // Unspecified
113  continue;
114  }
115 
116  // Preference in this order, if higher value is same, compare next and so on.
117  //
118  // application_mode: prefer not specified.
119  // 0: not specified
120  // 1: karaoke mode
121  // 2: surround sound mode
122  // lang_extension: prefer not specified or normal audio
123  // 0: Not specified
124  // 1: Normal audio/Caption
125  // 2: visually impaired
126  // 3: Director's comments 1
127  // 4: Director's comments 2
128  // sample_frequency: maybe 48K only
129  // 0: 48kHz
130  // 1: ??kHz
131  // quantization prefer highest bit width or drc
132  // 0: 16bit
133  // 1: 20bit
134  // 2: 24bit
135  // 3: drc
136  // channels: prefer no extension
137  // multichannel_extension
138 
139  // if ((best_multiframe > multiframe) ||
140  // (best_multiframe == multiframe && best_bitrate > bitrate) ||
141  // (best_multiframe == multiframe && best_bitrate == bitrate && best_count >= count))
142 
143  // Specs keep the meaning of the values of this field other than 0 secret, so we nail it to 48 kHz.
144  int sample_frequency = 48000;
145 
146  if ((best_application_mode < attr->application_mode) ||
147  (best_application_mode == attr->application_mode && best_lang_extension < attr->lang_extension) ||
148  (best_application_mode == attr->application_mode && best_lang_extension == attr->lang_extension && *best_sample_frequency > sample_frequency) ||
149  (best_application_mode == attr->application_mode && best_lang_extension == attr->lang_extension && *best_sample_frequency == sample_frequency && *best_channels > attr->channels) ||
150  (best_application_mode == attr->application_mode && best_lang_extension == attr->lang_extension && *best_sample_frequency == sample_frequency && *best_channels == attr->channels && best_quantization > attr->quantization)
151  )
152  {
153  continue;
154  }
155 
156  best_stream = i;
157  best_application_mode = attr->application_mode;
158  best_lang_extension = attr->lang_extension;
159  *best_sample_frequency = sample_frequency;
160  *best_channels = attr->channels;
161  best_quantization = attr->quantization;
162  }
163 
164  if (best_stream > -1)
165  {
166  ++*best_channels;
167  }
168 
169  return best_stream;
170 }

172 static AVRational dvd_frame_rate(const uint8_t * ptr)
173 {
174  AVRational framerate = { 0, 0 };
175 
176  // 11 = 30 fps, 10 = illegal, 01 = 25 fps, 00 = illegal
177  unsigned fps = ((ptr[3] & 0xC0) >> 6) & 0x03;
178 
179  switch (fps)
180  {
181  case 3: // PAL
182  {
183  framerate = av_make_q(30000, 1000);
184  break;
185  }
186  case 1: // NTSC
187  {
188  framerate = av_make_q(30000, 10001);
189  break;
190  }
191  default:
192  {
193  framerate = { 0, 0 };
194  break;
195  }
196  }
197 
198  return framerate;
199 }

201 static int64_t BCDtime(const dvd_time_t * dvd_time)
202 {
203  int64_t time[4];
204  AVRational framerate = dvd_frame_rate(&dvd_time->frame_u);
205 
206  if (!framerate.den)
207  {
208  framerate = av_make_q(25000, 1000); // Avoid divisions by 0
209  }
210  time[0] = dvd_time->hour;
211  time[1] = dvd_time->minute;
212  time[2] = dvd_time->second;
213  time[3] = dvd_time->frame_u & 0x3F; // Number of frame
214 
215  // convert BCD (two digits) to binary
216  for (int i = 0; i < 4; i++)
217  {
218  time[i] = ((time[i] & 0xf0) >> 4) * 10 + (time[i] & 0x0f);
219  }
220 
221  return (AV_TIME_BASE * (time[0] * 3600 + time[1] * 60 + time[2]) + static_cast<int64_t>(static_cast<double>(AV_TIME_BASE * time[3]) / av_q2d(framerate)));
222 }

224 static bool create_dvd_virtualfile(const ifo_handle_t *vts_file, const std::string & path, const struct stat *statbuf, void *buf, fuse_fill_dir_t filler, bool full_title, int title_idx, int chapter_idx, int angles, int ttnnum, int audio_stream, const AUDIO_SETTINGS & audio_settings, const VIDEO_SETTINGS & video_settings)

227  int title_no = title_idx + 1;
228  int chapter_no = chapter_idx + 1;
229  int pgcnum = vts_ptt_srpt->title[ttnnum - 1].ptt[chapter_idx].pgcn;
230  int pgn = vts_ptt_srpt->title[ttnnum - 1].ptt[chapter_idx].pgn;
231  const pgc_t *cur_pgc = vts_file->vts_pgcit->pgci_srp[pgcnum - 1].pgc;
232  AVRational framerate = { 0, 0 };
233  int64_t duration = 0;
234  uint64_t size = 0;
235  int interleaved = 0;
236  int start_cell = cur_pgc->program_map[pgn - 1] - 1;
237  int end_cell = 0;
238 
239  if (pgn < cur_pgc->nr_of_programs && !full_title)
240  {
241  end_cell = cur_pgc->program_map[pgn] - 1;
242  }
243  else
244  {
245  end_cell = cur_pgc->nr_of_cells;
246  }
247 
248  interleaved = cur_pgc->cell_playback[start_cell].interleaved;
249  framerate = dvd_frame_rate(&cur_pgc->cell_playback[start_cell].playback_time.frame_u);
250 
251  bool has_angles = false;
252 
253  for (int cell_no = start_cell; cell_no < end_cell; cell_no++)
254  {
255  cell_playback_t *cell_playback = &cur_pgc->cell_playback[cell_no];
256 
257  // Only count normal cells and the first of an angle to avoid duplicate sizes
258  if (cell_playback->block_mode == BLOCK_MODE_NOT_IN_BLOCK || cell_playback->block_mode == BLOCK_MODE_FIRST_CELL)
259  {
260  size += (cell_playback->last_sector - cell_playback->first_sector) * 2048;
261  duration += BCDtime(&cell_playback->playback_time);
262  }
263 
264  if (cell_playback->block_type == BLOCK_TYPE_ANGLE_BLOCK)
265  {
266  has_angles = true;
267  }
268  }
269 
270  if (duration < params.m_min_dvd_chapter_duration * AV_TIME_BASE)
271  {
272  Logging::debug(nullptr, "Skipping short DVD chapter.");
273  return true;
274  }
275 
276  if (!has_angles)
277  {
278  // If this chapter has no angle cells, reset angles to 1
279  angles = 1;
280  }
281 
282  // Split file if chapter has several angles
283  for (int angle_idx = 0; angle_idx < angles; angle_idx++)
284  {
285  char title_buf[PATH_MAX + 1];
286  std::string origfile;
287  struct stat stbuf;
288  int angle_no = angle_idx + 1;
289 
290  // can safely assume this a video
291  if (!full_title)
292  {
293  // Single chapter
294  if (angles > 1)
295  {
296  sprintf(title_buf, "%02d. Chapter %03d (Angle %d) [%s].%s",
297  title_no,
298  chapter_no,
299  angle_no,
300  replace_all(format_duration(duration), ":", "-").c_str(),
301  params.m_format[0].real_desttype().c_str());
302  }
303  else
304  {
305  sprintf(title_buf, "%02d. Chapter %03d [%s].%s",
306  title_no,
307  chapter_no,
308  replace_all(format_duration(duration), ":", "-").c_str(),
309  params.m_format[0].real_desttype().c_str());
310  }
311  }
312  else
313  {
314  // Full title
315  if (angles > 1)
316  {
317  sprintf(title_buf, "%02d. Title (Angle %d) [%s].%s",
318  title_no,
319  angle_no,
320  replace_all(format_duration(duration), ":", "-").c_str(),
321  params.m_format[0].real_desttype().c_str());
322  }
323  else
324  {
325  sprintf(title_buf, "%02d. Title [%s].%s",
326  title_no,
327  replace_all(format_duration(duration), ":", "-").c_str(),
328  params.m_format[0].real_desttype().c_str());
329  }
330  }
331 
332  std::string filename(title_buf);
333 
334  origfile = path + filename;
335 
336  memcpy(&stbuf, statbuf, sizeof(struct stat));
337 
338  stbuf.st_size = static_cast<__off_t>(size);
339  stbuf.st_blocks = (stbuf.st_size + 512 - 1) / 512;
340 
341  //init_stat(&st, size, false);
342 
343  if (buf != nullptr && filler(buf, filename.c_str(), &stbuf, 0))
344  {
345  // break;
346  }
347 
348  LPVIRTUALFILE virtualfile = insert_file(VIRTUALTYPE_DVD, path + filename, origfile, &stbuf);
349 
350  // DVD is video format anyway
351  virtualfile->m_format_idx = 0;
352  // Mark title/chapter/angle
353  virtualfile->m_full_title = full_title;
354  virtualfile->m_dvd.m_title_no = title_no;
355  virtualfile->m_dvd.m_chapter_no = chapter_no;
356  virtualfile->m_dvd.m_angle_no = angle_no;
357 
358  if (!transcoder_cached_filesize(virtualfile, &stbuf))
359  {
360  virtualfile->m_duration = duration;
361 
362  BITRATE video_bit_rate = video_settings.m_video_bit_rate;
363  if (duration)
364  {
365  video_bit_rate = static_cast<BITRATE>(size * 8LL * AV_TIME_BASE / static_cast<uint64_t>(duration)); // calculate bitrate in bps
366  }
367 
368  Logging::debug(virtualfile->m_origfile, "Video %1 %2x%3@%<%5.2f>4%5 fps %6 [%7]", format_bitrate(video_settings.m_video_bit_rate).c_str(), video_settings.m_width, video_settings.m_height, av_q2d(framerate), interleaved ? "i" : "p", format_size(size).c_str(), format_duration(duration).c_str());
369  if (audio_stream > -1)
370  {
371  Logging::debug(virtualfile->m_origfile, "Audio %1 Channels %2", audio_settings.m_channels, audio_settings.m_sample_rate);
372  }
373 
374  transcoder_set_filesize(virtualfile, duration, audio_settings.m_audio_bit_rate, audio_settings.m_channels, audio_settings.m_sample_rate, video_bit_rate, video_settings.m_width, video_settings.m_height, interleaved, framerate);
375  }
376  }
377 
378  return true;

381 static int parse_dvd(const std::string & path, const struct stat *statbuf, void *buf, fuse_fill_dir_t filler)
382 {
383  dvd_reader_t *dvd;
384  ifo_handle_t *ifo_file;
385  tt_srpt_t *tt_srpt;
386  int titles;
387  bool success = true;
388 
389  Logging::debug(path, "Parsing DVD.");
390 
391  dvd = DVDOpen(path.c_str());
392  if (!dvd)
393  {
394  Logging::error(path, "Couldn't open DVD.");
395  return ENOENT;
396  }
397 
398  ifo_file = ifoOpen(dvd, 0);
399  if (!ifo_file)
400  {
401  Logging::error(path, "Can't open VMG info for DVD.");
402  DVDClose(dvd);
403  return -EINVAL;
404  }
405  tt_srpt = ifo_file->tt_srpt;
406 
407  titles = tt_srpt->nr_of_srpts;
408 
409  Logging::debug(path, "There are %1 titles on this DVD.", titles);
410 
411  for (int title_idx = 0; title_idx < titles && success; ++title_idx)
412  {
413  ifo_handle_t *vts_file;
414  int vtsnum = tt_srpt->title[title_idx].title_set_nr;
415  int ttnnum = tt_srpt->title[title_idx].vts_ttn;
416  int chapters = tt_srpt->title[title_idx].nr_of_ptts;
417  int angles = tt_srpt->title[title_idx].nr_of_angles;
418 
419  Logging::trace(path, "Title: %1 VTS: %2 TTN: %3", title_idx + 1, vtsnum, ttnnum);
420  Logging::trace(path, "DVD title has %1 chapters and %2 angles.", chapters, angles);
421 
422  vts_file = ifoOpen(dvd, vtsnum);
423  if (!vts_file)
424  {
425  Logging::error(path, "Can't open info file for title %1.", vtsnum);
426  DVDClose(dvd);
427  return -EINVAL;
428  }
429 
430  // Set reasonable defaults
431  AUDIO_SETTINGS audio_settings;
432  audio_settings.m_audio_bit_rate = 256000;
433  audio_settings.m_channels = 2;
434  audio_settings.m_sample_rate = 48000;
435  int audio_stream = 0;
436 
437  VIDEO_SETTINGS video_settings;
438  video_settings.m_video_bit_rate = 8000000;
439  video_settings.m_width = 720;
440  video_settings.m_height = 576;
441 
442  if (vts_file->vtsi_mat)
443  {
444  audio_stream = dvd_find_best_audio_stream(vts_file->vtsi_mat, &audio_settings.m_channels, &audio_settings.m_sample_rate);
445 
446  video_settings.m_height = (vts_file->vtsi_mat->vts_video_attr.video_format != 0) ? 576 : 480;
447 
448  switch(vts_file->vtsi_mat->vts_video_attr.picture_size)
449  {
450  case 0:
451  {
452  video_settings.m_width = 720;
453  break;
454  }
455  case 1:
456  {
457  video_settings.m_width = 704;
458  break;
459  }
460  case 2:
461  {
462  video_settings.m_width = 352;
463  break;
464  }
465  case 3:
466  {
467  video_settings.m_width = 352;
468  video_settings.m_height /= 2;
469  break;
470  }
471  default:
472  {
473  Logging::warning(path, "DVD video contains invalid picture size attribute.");
474  }
475  }
476  }
477 
478  // Check if chapter is valid
479  c_adt_t *c_adt = vts_file->menu_c_adt;
480  size_t info_length = 0;
481 
482  if (c_adt != nullptr)
483  {
484  info_length = c_adt->last_byte + 1 - C_ADT_SIZE;
485  }
486 
487  bool skip = false;
488 
489  for (unsigned int n = 0; n < info_length / sizeof(cell_adr_t) && !skip; n++)
490  {
491  skip = (c_adt->cell_adr_table[n].start_sector >= c_adt->cell_adr_table[n].last_sector);
492  }
493 
494  if (skip)
495  {
496  Logging::info(path, "Title %1 has invalid size, ignoring.", title_idx + 1);
497  }
498  else
499  {
500  // Add separate chapters
501  for (int chapter_idx = 0; chapter_idx < chapters && success; ++chapter_idx)
502  {
503  success = create_dvd_virtualfile(vts_file, path, statbuf, buf, filler, false, title_idx, chapter_idx, angles, ttnnum, audio_stream, audio_settings, video_settings);
504  }
505 
506  if (success && chapters > 1)
507  {
508  // If more than 1 chapter, add full title as well
509  success = create_dvd_virtualfile(vts_file, path, statbuf, buf, filler, true, title_idx, 0, 1, ttnnum, audio_stream, audio_settings, video_settings);
510  }
511  }
512 
513  ifoClose(vts_file);
514  }
515 
516  ifoClose(ifo_file);
517  DVDClose(dvd);
518 
519  if (success)
520  {
521  return titles; // Number of titles on disk
522  }
523  else
524  {
525  return -errno;
526  }
527 }
