54void thread_pool::loop_function()
55{
56 unsigned int thread_no = ++m_cur_threads;
57
58 while (true)
59 {
60 THREADINFO info;
61 {
62 std::unique_lock<std::mutex> lock_queue_mutex(m_queue_mutex);
63 m_queue_cond.wait(lock_queue_mutex, [this]{ return (!m_thread_queue.empty() || m_queue_shutdown); });
64
65 if (m_queue_shutdown)
66 {
67 lock_queue_mutex.unlock();
68 break;
69 }
70
71 Logging::trace(nullptr, "Starting job taking pool thread no. %1 with id 0x%<%" FFMPEGFS_FORMAT_PTHREAD_T ">2.", thread_no, pthread_self());
72 info = m_thread_queue.front();
73 m_thread_queue.pop();
74 }
75
76 int ret = info.m_thread_func(info.m_opaque);
77
78 Logging::trace(nullptr, "The job using pool thread no. %1 with id 0x%<%" FFMPEGFS_FORMAT_PTHREAD_T ">2 has exited with return code %3.", thread_no, pthread_self(), ret);
79 }
80}

82bool thread_pool::schedule_thread(int (*thread_func)(void *), void *opaque)
83{
84 if (!m_queue_shutdown)
85 {
86 Logging::trace(nullptr, "Queuing up a new thread. There are %1 threads in the pool.", m_thread_pool.size());
87
88 {
89 std::lock_guard<std::mutex> lock_queue_mutex(m_queue_mutex);
90
91 THREADINFO info;
92
93 info.m_thread_func = thread_func;
94 info.m_opaque = opaque;
95 m_thread_queue.push(info);
96 }
97
98 m_queue_cond.notify_one();
99
100 return true;
101 }
102 else
103 {
104 return false;
105 }
106}

125void thread_pool::init(unsigned int num_threads /*= 0*/)
126{
127 if (num_threads)
128 {
129 m_num_threads = num_threads;
130 }
131
132 Logging::info(nullptr, "The thread pool is being initialised with a maximum of %1 threads.", m_num_threads);
133
134 for(unsigned int i = 0; i < m_num_threads; i++)
135 {
136 m_thread_pool.emplace_back(std::thread(&thread_pool::loop_function_starter, std::ref(*this)));
137 }

140void thread_pool::tear_down(bool silent)
141{
142 if (!silent)
143 {
144 Logging::debug(nullptr, "Tearing down the thread pool. There are %1 threads still in the pool.", m_thread_queue.size());
145 }
146
147 m_queue_mutex.lock();
148 m_queue_shutdown = true;
149 m_queue_mutex.unlock();
150 m_queue_cond.notify_all();
151
152 while (!m_thread_pool.empty())
153 {
154 m_thread_pool.back().join();
155 m_thread_pool.pop_back();
156 }
157}
