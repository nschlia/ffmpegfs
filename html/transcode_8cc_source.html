81 static bool transcode_until(Cache_Entry* cache_entry, size_t offset, size_t len, uint32_t segment_no)
82 {
83  size_t end = offset + len; // Cast OK: offset will never be < 0.
84  bool success = true;
85 
86  if (cache_entry->is_finished() || cache_entry->m_buffer->is_segment_finished(segment_no) || cache_entry->m_buffer->tell(segment_no) >= end)
87  {
88  return true;
89  }
90 
91  try
92  {
93  // Wait until decoder thread has reached the desired position
94  if (cache_entry->m_is_decoding)
95  {
96  bool reported = false;
97  while (!(cache_entry->is_finished() || cache_entry->m_buffer->is_segment_finished(segment_no) || cache_entry->m_buffer->tell(segment_no) >= end) && !cache_entry->m_cache_info.m_error)
98  {
99  if (fuse_interrupted())
100  {
101  Logging::info(cache_entry->virtname(), "The client has gone away.");
102  errno = 0; // No error
103  break;
104  }
105 
106  if (thread_exit)
107  {
108  Logging::warning(cache_entry->virtname(), "Thread exit was received.");
109  errno = EINTR;
110  throw false;
111  }
112 
113  if (!reported)
114  {
115  if (!segment_no)
116  {
117  Logging::trace(cache_entry->virtname(), "Cache miss at offset %1 with length %2.", offset, len);
118  }
119  else
120  {
121  Logging::trace(cache_entry->virtname(), "Cache miss at offset %1 with length %2 for segment no. %3.", offset, len, segment_no);
122  }
123  reported = true;
124  }
125  mssleep(250);
126  }
127 
128  if (reported)
129  {
130  if (!segment_no)
131  {
132  Logging::trace(cache_entry->virtname(), "Cache hit at offset %1 with length %2.", offset, len);
133  }
134  else
135  {
136  Logging::trace(cache_entry->virtname(), "Cache hit at offset %1 with length %2 for segment no. %3.", offset, len, segment_no);
137  }
138  }
139  success = !cache_entry->m_cache_info.m_error;
140  }
141  }
142  catch (bool _success)
143  {
144  success = _success;
145  }
146 
147  return success;
148 }

156 static int transcode_finish(Cache_Entry* cache_entry, FFmpeg_Transcoder &transcoder)
157 {
158  int res = transcoder.encode_finish();
159  if (res < 0)
160  {
161  return res;
162  }
163 
164  // Check encoded buffer size. Does not affect HLS segments.
165  cache_entry->m_cache_info.m_duration = transcoder.duration();
166  cache_entry->m_cache_info.m_encoded_filesize = cache_entry->m_buffer->buffer_watermark();
167  cache_entry->m_cache_info.m_video_frame_count = transcoder.video_frame_count();
168  cache_entry->m_cache_info.m_segment_count = transcoder.segment_count();
169  cache_entry->m_cache_info.m_result = !transcoder.have_seeked() ? RESULTCODE_FINISHED_SUCCESS : RESULTCODE_FINISHED_INCOMPLETE;
170  cache_entry->m_is_decoding = false;
171  cache_entry->m_cache_info.m_errno = 0;
172  cache_entry->m_cache_info.m_averror = 0;
173 
174  Logging::debug(transcoder.virtname(), "Finishing file.");
175 
176  if (!cache_entry->m_buffer->reserve(cache_entry->m_cache_info.m_encoded_filesize))
177  {
178  Logging::debug(transcoder.virtname(), "Unable to truncate the buffer.");
179  }
180 
181  if (!transcoder.is_multiformat())
182  {
183  Logging::debug(transcoder.virtname(), "Predicted size: %1 Final: %2 Diff: %3 (%4%).",
184  format_size_ex(cache_entry->m_cache_info.m_predicted_filesize).c_str(),
185  format_size_ex(cache_entry->m_cache_info.m_encoded_filesize).c_str(),
186  format_result_size_ex(cache_entry->m_cache_info.m_encoded_filesize, cache_entry->m_cache_info.m_predicted_filesize).c_str(),
187  ((static_cast<double>(cache_entry->m_cache_info.m_encoded_filesize) * 1000 / (static_cast<double>(cache_entry->m_cache_info.m_predicted_filesize) + 1)) + 5) / 10);
188  }
189 
190  cache_entry->flush();
191 
192  return 0;

355 Cache_Entry* transcoder_new(LPVIRTUALFILE virtualfile, bool begin_transcode)
356 {
357  // Allocate transcoder structure
358  Cache_Entry* cache_entry = cache->openio(virtualfile);
359  if (cache_entry == nullptr)
360  {
361  return nullptr;
362  }
363 
364  Logging::trace(cache_entry->filename(), "Creating transcoder object.");
365 
366  try
367  {
368  cache_entry->lock();
369 
370  if (!cache_entry->openio(begin_transcode))
371  {
372  throw static_cast<int>(errno);
373  }
374 
375  if (params.m_disable_cache)
376  {
377  // Disable cache
378  cache_entry->clear();
379  }
380  else if (!cache_entry->m_is_decoding && cache_entry->outdated())
381  {
382  cache_entry->clear();
383  }
384 
385  if (cache_entry->m_cache_info.m_duration)
386  {
387  virtualfile->m_duration = cache_entry->m_cache_info.m_duration;
388  }
389  if (cache_entry->m_cache_info.m_predicted_filesize)
390  {
391  virtualfile->m_predicted_size = cache_entry->m_cache_info.m_predicted_filesize;
392  }
393  if (cache_entry->m_cache_info.m_video_frame_count)
394  {
395  virtualfile->m_video_frame_count = cache_entry->m_cache_info.m_video_frame_count;
396  }
397 
398  if (!cache_entry->m_is_decoding && !cache_entry->is_finished_success())
399  {
400  if (begin_transcode)
401  {
402  Logging::debug(cache_entry->filename(), "Starting transcoder thread.");
403 
404  // Clear cache to remove any older remains
405  cache_entry->clear();
406 
407  // Must decode the file, otherwise simply use cache
408  cache_entry->m_is_decoding = true;
409 
410  THREAD_DATA* thread_data = new(std::nothrow) THREAD_DATA;
411 
412  thread_data->m_initialised = false;
413  thread_data->m_arg = cache_entry;
414  thread_data->m_thread_running_lock_guard = false;
415 
416  {
417  std::unique_lock<std::mutex> lock_thread_running_mutex(thread_data->m_thread_running_mutex);
418 
419  tp->schedule_thread(&transcoder_thread, thread_data);
420 
421  // Let decoder get into gear before returning from open
422  while (!thread_data->m_thread_running_lock_guard)
423  {
424  thread_data->m_thread_running_cond.wait(lock_thread_running_mutex);
425  }
426  }
427 
428  if (cache_entry->m_cache_info.m_error)
429  {
430  int ret;
431 
432  Logging::trace(cache_entry->filename(), "Transcoder error!");
433 
434  ret = cache_entry->m_cache_info.m_errno;
435  if (!ret)
436  {
437  ret = EIO; // Must return something, be it a simple I/O error...
438  }
439  throw ret;
440  }
441 
442  Logging::debug(cache_entry->filename(), "Transcoder thread is running.");
443  }
444  else if (!cache_entry->m_cache_info.m_predicted_filesize)
445  {
446  if (!transcoder_predict_filesize(virtualfile, cache_entry))
447  {
448  throw static_cast<int>(errno);
449  }
450  }
451  }
452  else if (begin_transcode)
453  {
454  Logging::trace(cache_entry->virtname(), "Reading file from cache.");
455  }
456 
457  cache_entry->unlock();
458  }
459  catch (int orgerrno)
460  {
461  cache_entry->m_is_decoding = false;
462  cache_entry->unlock();
463  cache->closeio(&cache_entry, CACHE_CLOSE_DELETE);
464  cache_entry = nullptr; // Make sure to return NULL here even if the cache could not be deleted now (still in use)
465  errno = orgerrno; // Restore last errno
466  }
467 
468  return cache_entry;
469 }

471 bool transcoder_read(Cache_Entry* cache_entry, char* buff, size_t offset, size_t len, int * bytes_read, uint32_t segment_no)
472 {
473  bool success = true;
474 
475  if (!segment_no)
476  {
477  Logging::trace(cache_entry->virtname(), "Reading %1 bytes from offset %2 to %3.", len, offset, len + offset);
478  }
479  else
480  {
481  Logging::trace(cache_entry->virtname(), "Reading %1 bytes from offset %2 to %3 for segment no. %4.", len, offset, len + offset, segment_no);
482  }
483 
484  // Store access time
485  cache_entry->update_access();
486 
487  // Update read counter
488  cache_entry->update_read_count();
489 
490  try
491  {
492  // Try to read requested segment, stack a seek to if if this fails.
493  // No seek if not HLS (segment_no) and not required if < MIN_SEGMENT
494  if (segment_no && !cache_entry->m_buffer->segment_exists(segment_no))
495  {
496  cache_entry->m_seek_to_no = segment_no;
497  }
498 
499  // Open for reading if necessary
500  if (!cache_entry->m_buffer->open_file(segment_no, CACHE_FLAG_RO))
501  {
502  throw false;
503  }
504 
505  if (!cache_entry->is_finished_success())
506  {
507  switch (params.current_format(cache_entry->virtualfile())->filetype())
508  {
509  case FILETYPE_MP3:
510  {
511  // If we are encoding to MP3 and the requested data overlaps the ID3v1 tag
512  // at the end of the file, do not encode data first up to that position.
513  // This optimises the case where applications read the end of the file
514  // first to read the ID3v1 tag.
515  if ((offset > cache_entry->m_buffer->tell(segment_no)) &&
516  (offset + len >= (cache_entry->size() - ID3V1_TAG_LENGTH)))
517  {
518  // Stuff buffer with garbage, apps won't try to play that chunk anyway.
519  memset(buff, 0xFF, len);
520  if (cache_entry->size() - offset == ID3V1_TAG_LENGTH)
521  {
522  memcpy(buff, &cache_entry->m_id3v1, std::min(len, ID3V1_TAG_LENGTH));
523  }
524 
525  errno = 0;
526 
527  throw true; // OK
528  }
529  break;
530  }
531  default:
532  {
533  break;
534  }
535  }
536 
537  // Windows seems to access the files on Samba drives starting at the last 64K segment simply when
538  // the file is opened. Setting win_smb_fix=1 will ignore these attempts (not decode the file up
539  // to this point).
540  // Access will only be ignored if occurring at the second access.
541  if (params.m_win_smb_fix && cache_entry->read_count() == 2)
542  {
543  if ((offset > cache_entry->m_buffer->tell(segment_no)) &&
544  (len <= 65536) &&
545  check_ignore(cache_entry->size(), offset) &&
546  ((offset + len) > (cache_entry->size())))
547  {
548  Logging::warning(cache_entry->virtname(), "Ignoring Windows' groundless access to the last 8K boundary of the file.");
549 
550  errno = 0;
551  *bytes_read = 0; // We've read nothing
552  len = 0;
553 
554  throw true; // OK
555  }
556  }
557  }
558 
559  success = transcode_until(cache_entry, offset, len, segment_no);
560 
561  if (!success)
562  {
563  errno = cache_entry->m_cache_info.m_errno ? cache_entry->m_cache_info.m_errno : EIO;
564  throw false;
565  }
566 
567  // truncate if we didn't actually get len
568  if (cache_entry->m_buffer->buffer_watermark(segment_no) < offset)
569  {
570  len = 0;
571  }
572  else if (cache_entry->m_buffer->buffer_watermark(segment_no) < offset + len)
573  {
574  len = cache_entry->m_buffer->buffer_watermark(segment_no) - offset;
575  }
576 
577  if (!cache_entry->m_buffer->copy(reinterpret_cast<uint8_t*>(buff), offset, len, segment_no))
578  {
579  len = 0;
580  // We already capped len to not overread the buffer, so it is an error if we end here.
581  throw false;
582  }
583 
584  if (cache_entry->m_cache_info.m_error)
585  {
586  errno = cache_entry->m_cache_info.m_errno ? cache_entry->m_cache_info.m_errno : EIO;
587  throw false;
588  }
589 
590  errno = 0;
591  }
592  catch (bool _success)
593  {
594  success = _success;
595  }
596 
597  *bytes_read = static_cast<int>(len);
598 
599  return success;
600 }

602 bool transcoder_read_frame(Cache_Entry* cache_entry, char* buff, size_t offset, size_t len, uint32_t frame_no, int * bytes_read, LPVIRTUALFILE virtualfile)
603 {
604  bool success = false;
605 
606  Logging::trace(cache_entry->virtname(), "Reading %1 bytes from offset %2 to %3 for frame no. %4.", len, offset, len + offset, frame_no);
607 
608  // Store access time
609  cache_entry->update_access();
610 
611  // Update read counter
612  cache_entry->update_read_count();
613 
614  try
615  {
616  // Open for reading if necessary
617  if (!cache_entry->m_buffer->open_file(0, CACHE_FLAG_RO))
618  {
619  throw false;
620  }
621 
622  std::vector<uint8_t> data;
623 
624  // Wait until decoder thread has the requested frame available
625  if (cache_entry->m_is_decoding || cache_entry->m_suspend_timeout)
626  {
627  // Try to read requested frame, stack a seek to if if this fails.
628  if (!cache_entry->m_buffer->read_frame(&data, frame_no))
629  {
630  bool reported = false;
631 
632  cache_entry->m_seek_to_no = frame_no;
633 
634  int retries = TOTAL_RETRIES;
635  while (!cache_entry->m_buffer->read_frame(&data, frame_no) && !thread_exit)
636  {
637  if (errno != EAGAIN)
638  {
639  Logging::error(cache_entry->virtname(), "Reading image frame no. %1: (%2) %3", frame_no, errno, strerror(errno));
640  throw false;
641  }
642 
643  if (!cache_entry->m_suspend_timeout)
644  {
645  if (errno == EAGAIN && !--retries)
646  {
647  errno = ETIMEDOUT;
648  Logging::error(cache_entry->virtname(), "Timeout reading image frame no. %1: (%2) %3", frame_no, errno, strerror(errno));
649  throw false;
650  }
651  }
652  else
653  {
654  retries = TOTAL_RETRIES;
655  }
656 
657  if (thread_exit)
658  {
659  Logging::warning(cache_entry->virtname(), "Thread exit was received.");
660  errno = EINTR;
661  throw false;
662  }
663 
664  if (!reported)
665  {
666  Logging::trace(cache_entry->virtname(), "Frame no. %1: Cache miss at offset %<%11zu>2 (length %<%6u>3).", frame_no, offset, len);
667  reported = true;
668  }
669 
670  mssleep(GRANULARITY);
671  }
672  }
673  else
674  {
675  Logging::trace(cache_entry->virtname(), "Frame no. %1: Cache hit at offset %<%11zu>2 (length %<%6u>3).", frame_no, offset, len);
676  }
677  success = !cache_entry->m_cache_info.m_error;
678  }
679  else
680  {
681  success = cache_entry->m_buffer->read_frame(&data, frame_no);
682  if (!success)
683  {
684  Logging::error(cache_entry->virtname(), "Reading image frame no. %1: (%2) %3", frame_no, errno, strerror(errno));
685  throw false;
686  }
687  }
688 
689  if (success)
690  {
691  if (data.size() < offset)
692  {
693  len = 0;
694  }
695  else if (data.size() < offset + len)
696  {
697  len = data.size() - offset;
698  }
699 
700  if (len)
701  {
702  memcpy(buff, data.data() + offset, len);
703  }
704 
705  stat_set_size(&virtualfile->m_st, data.size());
706 
707  *bytes_read = static_cast<int>(len);
708  }
709  }
710  catch (bool _success)
711  {
712  success = _success;
713  }
714 
715  return success;
716 }

775 static bool transcode(THREAD_DATA *thread_data, Cache_Entry *cache_entry, FFmpeg_Transcoder & transcoder, bool *timeout)
776 {
777  int averror = 0;
778  int syserror = 0;
779  bool success = true;
780 
781  // Clear cache to remove any older remains
782  cache_entry->clear();
783 
784  // Must decode the file, otherwise simply use cache
785  cache_entry->m_is_decoding = true;
786 
787  try
788  {
789  bool unlocked = false;
790 
791  Logging::info(cache_entry->filename(), "Transcoding to %1.", params.current_format(cache_entry->virtualfile())->desttype().c_str());
792 
793  if (!cache_entry->openio())
794  {
795  throw (static_cast<int>(errno));
796  }
797 
798  averror = transcoder.open_input_file(cache_entry->virtualfile());
799  if (averror < 0)
800  {
801  throw (static_cast<int>(errno));
802  }
803 
804  if (!cache_entry->m_cache_info.m_duration)
805  {
806  cache_entry->m_cache_info.m_duration = transcoder.duration();
807  }
808 
809  if (!cache_entry->m_cache_info.m_predicted_filesize)
810  {
811  cache_entry->m_cache_info.m_predicted_filesize = transcoder.predicted_filesize();
812  }
813 
814  if (!cache_entry->m_cache_info.m_video_frame_count)
815  {
816  cache_entry->m_cache_info.m_video_frame_count = transcoder.video_frame_count();
817  }
818 
819  if (!cache_entry->m_cache_info.m_segment_count)
820  {
821  cache_entry->m_cache_info.m_segment_count = transcoder.segment_count();
822  }
823 
824  if (cache != nullptr && !cache->maintenance(transcoder.predicted_filesize()))
825  {
826  throw (static_cast<int>(errno));
827  }
828 
829  averror = transcoder.open_output_file(cache_entry->m_buffer);
830  if (averror < 0)
831  {
832  throw (static_cast<int>(errno));
833  }
834 
835  memcpy(&cache_entry->m_id3v1, transcoder.id3v1tag(), sizeof(ID3v1));
836 
837  thread_data->m_initialised = true;
838 
839  unlocked = false;
840  if ((!params.m_prebuffer_size && !params.m_prebuffer_time) || transcoder.is_frameset())
841  {
842  // Unlock frame set from beginning
843  unlocked = true;
844  thread_data->m_thread_running_lock_guard = true;
845  thread_data->m_thread_running_cond.notify_all(); // signal that we are running
846  }
847  else
848  {
849  if (params.m_prebuffer_time)
850  {
851  Logging::debug(cache_entry->virtname(), "Pre-buffering up to %1.", format_time(params.m_prebuffer_time).c_str());
852  }
853  if (params.m_prebuffer_size)
854  {
855  Logging::debug(cache_entry->virtname(), "Pre-buffering up to %1.", format_size(params.m_prebuffer_size).c_str());
856  }
857  }
858 
859  cache_entry->m_suspend_timeout = false;
860 
861  while (!cache_entry->is_finished() && !(*timeout = cache_entry->decode_timeout()) && !thread_exit)
862  {
863  DECODER_STATUS status = DECODER_SUCCESS;
864 
865  if (cache_entry->ref_count() > 1)
866  {
867  // Set last access time
868  cache_entry->update_access(false);
869  }
870 
871  if (transcoder.is_frameset())
872  {
873  uint32_t frame_no = cache_entry->m_seek_to_no;
874  if (frame_no)
875  {
876  cache_entry->m_seek_to_no = 0;
877 
878  averror = transcoder.stack_seek_frame(frame_no);
879  if (averror < 0)
880  {
881  throw (static_cast<int>(errno));
882  }
883  }
884  }
885  else if (transcoder.is_hls())
886  {
887  uint32_t segment_no = cache_entry->m_seek_to_no;
888  if (segment_no)
889  {
890  cache_entry->m_seek_to_no = 0;
891 
892  averror = transcoder.stack_seek_segment(segment_no);
893  if (averror < 0)
894  {
895  throw (static_cast<int>(errno));
896  }
897  }
898  }
899 
900  averror = transcoder.process_single_fr(&status);
901 
902  if (status == DECODER_ERROR)
903  {
904  errno = EIO;
905  throw (static_cast<int>(errno));
906  }
907  else if (status == DECODER_EOF)
908  {
909  cache_entry->m_suspend_timeout = true; // Suspend read_frame time out until transcoder is reopened.
910 
911  averror = transcode_finish(cache_entry, transcoder);
912 
913  if (averror < 0)
914  {
915  errno = EIO;
916  throw (static_cast<int>(errno));
917  }
918  }
919 
920  if (!unlocked && cache_entry->m_buffer->buffer_watermark() > params.m_prebuffer_size && transcoder.pts() > static_cast<int64_t>(params.m_prebuffer_time) * AV_TIME_BASE)
921  {
922  unlocked = true;
923  Logging::debug(cache_entry->virtname(), "Pre-buffer limit reached.");
924 
925  thread_data->m_thread_running_lock_guard = true;
926  thread_data->m_thread_running_cond.notify_all(); // signal that we are running
927  }
928 
929  if (cache_entry->ref_count() <= 1 && cache_entry->suspend_timeout())
930  {
931  if (!unlocked && (params.m_prebuffer_size || params.m_prebuffer_time))
932  {
933  unlocked = true;
934  thread_data->m_thread_running_lock_guard = true;
935  thread_data->m_thread_running_cond.notify_all(); // signal that we are running
936  }
937 
938  Logging::info(cache_entry->virtname(), "Suspend timeout. Transcoding suspended after %1 seconds inactivity.", params.m_max_inactive_suspend);
939 
940  while (cache_entry->suspend_timeout() && !(*timeout = cache_entry->decode_timeout()) && !thread_exit)
941  {
942  mssleep(GRANULARITY);
943  }
944 
945  if (timeout)
946  {
947  break;
948  }
949 
950  Logging::info(cache_entry->virtname(), "Transcoding resumed.");
951  }
952  }
953 
954  if (!unlocked && (params.m_prebuffer_size || params.m_prebuffer_time))
955  {
956  Logging::debug(cache_entry->virtname(), "File transcode complete, releasing buffer early: Size %1.", cache_entry->m_buffer->buffer_watermark());
957  thread_data->m_thread_running_lock_guard = true;
958  thread_data->m_thread_running_cond.notify_all(); // signal that we are running
959  }
960  }
961  catch (int _syserror)
962  {
963  success = false;
964  syserror = _syserror;
965 
966  if (!syserror && averror > -512)
967  {
968  // If no system error reported explicitly, and averror is a POSIX error
969  // (we simply assume that if averror < 512, I think averrors are all higher values)
970  syserror = AVUNERROR(averror);
971  }
972 
973  cache_entry->m_cache_info.m_error = true;
974  if (!syserror)
975  {
976  // If system error is still zero, set to EIO to return at least anything else than success.
977  syserror = EIO;
978  }
979 
980  thread_data->m_thread_running_lock_guard = true;
981  thread_data->m_thread_running_cond.notify_all(); // unlock main thread
982  }
983 
984  cache_entry->m_cache_info.m_errno = syserror; // Preserve errno
985  cache_entry->m_cache_info.m_averror = averror; // Preserve averror
986 
987  transcoder.closeio();
988 
989  return success;
990 }

997 static int transcoder_thread(void *arg)
998 {
999  THREAD_DATA *thread_data = static_cast<THREAD_DATA*>(arg);
1000  Cache_Entry *cache_entry = static_cast<Cache_Entry *>(thread_data->m_arg);
1001  FFmpeg_Transcoder transcoder;
1002  bool timeout = false;
1003  bool success = true;
1004 
1005  std::unique_lock<std::recursive_mutex> lock_active_mutex(cache_entry->m_active_mutex);
1006  std::unique_lock<std::recursive_mutex> lock_restart_mutex(cache_entry->m_restart_mutex);
1007 
1008  uint32_t seek_frame = 0;
1009 
1010  do
1011  {
1012  if (seek_frame)
1013  {
1014  Logging::error(transcoder.virtname(), "Transcoder completed with last seek frame to %1. Transcoder is being restarted.", seek_frame);
1015  }
1016 
1017  success = transcode(thread_data, cache_entry, transcoder, &timeout);
1018 
1019  seek_frame = cache_entry->m_seek_to_no != 0 ? cache_entry->m_seek_to_no.load() : transcoder.last_seek_frame_no();
1020  }
1021  while (success && !thread_exit && cache != nullptr && seek_frame);
1022 
1023  cache_entry->m_is_decoding = false;
1024 
1025  if (timeout || thread_exit || transcoder.have_seeked())
1026  {
1027  if (!transcoder.have_seeked())
1028  {
1029  cache_entry->m_cache_info.m_error = true;
1030  cache_entry->m_cache_info.m_errno = EIO; // Report I/O error
1031 
1032  if (timeout)
1033  {
1034  Logging::warning(cache_entry->virtname(), "Timeout! Transcoding aborted after %1 seconds inactivity.", params.m_max_inactive_abort);
1035  }
1036  else if (thread_exit)
1037  {
1038  Logging::info(cache_entry->virtname(), "Thread exit! Transcoding aborted.");
1039  }
1040  else
1041  {
1042  Logging::info(cache_entry->virtname(), "Transcoding aborted.");
1043  }
1044  }
1045  else
1046  {
1047  // Must restart from scratch, but this is not an error.
1048  cache_entry->m_cache_info.m_error = false;
1049  cache_entry->m_cache_info.m_errno = 0;
1050  cache_entry->m_cache_info.m_averror = 0;
1051  }
1052  }
1053  else
1054  {
1055  cache_entry->m_cache_info.m_error = !success;
1056 
1057  if (success)
1058  {
1059  Logging::info(cache_entry->virtname(), "Transcoding completed successfully.");
1060 
1061  cache_entry->m_cache_info.m_errno = 0;
1062  cache_entry->m_cache_info.m_averror = 0;
1063  }
1064  else
1065  {
1066  Logging::error(cache_entry->virtname(), "Transcoding exited with error.");
1067  if (cache_entry->m_cache_info.m_errno)
1068  {
1069  Logging::error(cache_entry->virtname(), "System error: (%1) %2", cache_entry->m_cache_info.m_errno, strerror(cache_entry->m_cache_info.m_errno));
1070  }
1071 
1072  if (cache_entry->m_cache_info.m_averror)
1073  {
1074  Logging::error(cache_entry->virtname(), "FFMpeg error: (%1) %2", cache_entry->m_cache_info.m_averror, ffmpeg_geterror(cache_entry->m_cache_info.m_averror).c_str());
1075  }
1076  }
1077  }
1078 
1079  int _errno = cache_entry->m_cache_info.m_errno;
1080 
1081  if (cache != nullptr)
1082  {
1083  cache->closeio(&cache_entry, timeout ? CACHE_CLOSE_DELETE : CACHE_CLOSE_NOOPT);
1084  }
1085 
1086  delete thread_data;
1087 
1088  errno = _errno;
1089 
1090  return errno;
1091 }
