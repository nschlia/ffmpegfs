433 static bool create_bluray_virtualfile(BLURAY *bd, const BLURAY_TITLE_INFO* ti, const std::string & path, const struct stat * statbuf, void * buf, fuse_fill_dir_t filler, bool is_main_title, bool full_title, uint32_t title_idx, uint32_t chapter_idx)
434 {
435  BLURAY_CLIP_INFO *clip = &ti->clips[0];
436  BLURAY_TITLE_CHAPTER *chapter = &ti->chapters[chapter_idx];
437  char title_buf[PATH_MAX + 1];
438  int64_t duration;
439 
440  if (full_title)
441  {
442  duration = static_cast<int64_t>(ti->duration) * AV_TIME_BASE / 90000;
443 
444  if (duration < AV_TIME_BASE)
445  {
446  Logging::trace(path, "Title %1: skipping empty title", title_idx + 1);
447  return true;
448  }
449 
450  snprintf(title_buf, sizeof(title_buf) - 1, "%02u. Title [%s]%s.%s",
451  title_idx + 1,
452  replace_all(format_duration(duration), ":", "-").c_str(),
453  is_main_title ? "+" : "",
454  ffmpeg_format[0].fileext().c_str()); // can safely assume this is a video format
455  }
456  else
457  {
458  duration = static_cast<int64_t>(chapter->duration) * AV_TIME_BASE / 90000;
459 
460  if (duration < AV_TIME_BASE)
461  {
462  Logging::trace(path, "Title %1 Chapter %2: skipping empty chapter", title_idx + 1, chapter_idx + 1);
463  return true;
464  }
465 
466  snprintf(title_buf, sizeof(title_buf) - 1, "%02u. Chapter %03u [%s]%s.%s",
467  title_idx + 1,
468  chapter_idx + 1,
469  replace_all(format_duration(duration), ":", "-").c_str(),
470  is_main_title ? "+" : "",
471  ffmpeg_format[0].fileext().c_str()); // can safely assume this is a video format
472 
473  }
474 
475  std::string filename(title_buf);
476 
477  LPVIRTUALFILE virtualfile = nullptr;
478  if (!ffmpeg_format[0].is_multiformat())
479  {
480  virtualfile = insert_file(VIRTUALTYPE_BLURAY, path + filename, statbuf);
481  }
482  else
483  {
484  virtualfile = insert_dir(VIRTUALTYPE_BLURAY, path + filename, statbuf);
485  }
486 
487  if (virtualfile == nullptr)
488  {
489  Logging::error(path, "Failed to create virtual path: %1", (path + filename).c_str());
490  errno = EIO;
491  return false;
492  }
493 
494  if (add_fuse_entry(buf, filler, title_buf, &virtualfile->m_st, 0))
495  {
496  // break;
497  }
498 
499  // Bluray is video format anyway
500  virtualfile->m_format_idx = 0;
501  // Mark title/chapter/angle
502  virtualfile->m_full_title = full_title;
503  virtualfile->m_bluray.m_title_no = title_idx + 1;
504  virtualfile->m_bluray.m_playlist_no = ti->playlist;
505  virtualfile->m_bluray.m_chapter_no = chapter_idx + 1;
506  virtualfile->m_bluray.m_angle_no = 1;
507 
508  if (!transcoder_cached_filesize(virtualfile, &virtualfile->m_st))
509  {
510  BITRATE video_bit_rate = 29*1024*1024; // In case the real bitrate cannot be calculated later, assume 20 Mbit video bitrate
511  BITRATE audio_bit_rate = 256*1024; // In case the real bitrate cannot be calculated later, assume 256 kBit audio bitrate
512 
513  bool audio = false;
514 
515  int interleaved = 0;
516 
517  if (!bd_select_title(bd, title_idx))
518  {
519  Logging::error(path, "Failed to open bluray title %1", title_idx);
520  errno = EIO;
521  return false;
522  }
523 
524  uint64_t size = bd_get_title_size(bd);
525 
526  virtualfile->m_duration = duration;
527 
528  if (duration)
529  {
535  video_bit_rate = static_cast<BITRATE>(size * 8LL * AV_TIME_BASE / static_cast<uint64_t>(duration)); // calculate bitrate in bps
536  }
537 
538  // Get details
539  if (clip->audio_stream_count)
540  {
541  audio_stream_info(path, &clip->audio_streams[parse_find_best_audio_stream()], &virtualfile->m_channels, &virtualfile->m_sample_rate);
542  }
543  if (clip->video_stream_count)
544  {
545  audio = video_stream_info(path, &clip->video_streams[parse_find_best_video_stream()], &virtualfile->m_width, &virtualfile->m_height, &virtualfile->m_framerate, &interleaved);
546  }
547 
548  Logging::trace(virtualfile->m_destfile, "Video %1 %2x%3@%<%5.2f>4%5 fps %6 [%7]", format_bitrate(video_bit_rate).c_str(), virtualfile->m_width, virtualfile->m_height, av_q2d(virtualfile->m_framerate), interleaved ? "i" : "p", format_size(size).c_str(), format_duration(duration).c_str());
549  if (audio)
550  {
551  Logging::trace(virtualfile->m_destfile, "Audio %1 channels %2", virtualfile->m_channels, format_samplerate(virtualfile->m_sample_rate).c_str());
552  }
553 
554  transcoder_set_filesize(virtualfile, duration, audio_bit_rate, virtualfile->m_channels, virtualfile->m_sample_rate, AV_SAMPLE_FMT_NONE, video_bit_rate, virtualfile->m_width, virtualfile->m_height, interleaved, virtualfile->m_framerate);
555 
556  virtualfile->m_video_frame_count = static_cast<uint32_t>(av_rescale_q(duration, av_get_time_base_q(), av_inv_q(virtualfile->m_framerate)));
557  virtualfile->m_predicted_size = static_cast<size_t>(size);
558  }
559 
560  return true;
561 }

571 static int parse_bluray(const std::string & path, const struct stat * statbuf, void * buf, fuse_fill_dir_t filler)
572 {
573  BLURAY *bd;
574  uint32_t title_count;
575  int main_title;
576  unsigned int seconds = 0;
577  uint8_t flags = TITLES_RELEVANT;
578  const char *bd_dir = nullptr;
579  bool success = true;
580 
581  bd_dir = path.c_str();
582 
583  Logging::debug(path, "Parsing Bluray.");
584 
585  bd = bd_open(bd_dir, nullptr);
586 
587  title_count = bd_get_titles(bd, flags, seconds);
588  main_title = bd_get_main_title(bd);
589  if (main_title >= 0)
590  {
591  Logging::trace(path, "Main title: %1", main_title + 1);
592  }
593 
594  for (uint32_t title_idx = 0; title_idx < title_count && success; title_idx++)
595  {
596  BLURAY_TITLE_INFO* ti = bd_get_title_info(bd, title_idx, 0);
597  bool is_main_title = (main_title >= 0 && title_idx == static_cast<uint32_t>(main_title));
598 
599  // Add separate chapters
600  for (uint32_t chapter_idx = 0; chapter_idx < ti->chapter_count && success; chapter_idx++)
601  {
602  success = create_bluray_virtualfile(bd, ti, path, statbuf, buf, filler, is_main_title, false, title_idx, chapter_idx);
603  }
604 
605  if (success && ti->chapter_count > 1)
606  {
607  // If more than 1 chapter, add full title as well
608  success = create_bluray_virtualfile(bd, ti, path, statbuf, buf, filler, is_main_title, true, title_idx, 0);
609  }
610 
611  bd_free_title_info(ti);
612  }
613 
614  bd_close(bd);
615 
616  if (success)
617  {
618  return static_cast<int>(title_count);
619  }
620  else
621  {
622  return -errno;
623  }
624 }

626 int check_bluray(const std::string & path, void *buf, fuse_fill_dir_t filler)
627 {
628  std::string _path(path);
629  struct stat stbuf;
630  int res = 0;
631 
632  append_sep(&_path);
633 
634  if (stat((_path + "BDMV/index.bdmv").c_str(), &stbuf) == 0)
635  {
636  if (!check_path(_path))
637  {
638  Logging::trace(_path, "Bluray detected.");
639  res = parse_bluray(_path, &stbuf, buf, filler);
640  Logging::trace(_path, "Found %1 titles.", res);
641  }
642  else
643  {
644  res = load_path(_path, &stbuf, buf, filler);
645  }
646 
647  add_dotdot(buf, filler, &stbuf, 0);
648  }
649  return res;
650 }
