78 bool Buffer::open_file(uint32_t segment_no, uint32_t flags)
79 {
80  uint32_t index = segment_no;
81  if (index)
82  {
83  index--;
84  }
85 
86  m_ci[index].m_flags |= flags;
87 
88  if (m_ci[index].m_fd != -1)
89  {
90  Logging::trace(m_ci[index].m_cachefile, "Cache file already open, no need to open again.");
91  // Already open
92  return true;
93  }
94 
95  if (flags & CACHE_FLAG_RW)
96  {
97  Logging::debug(m_ci[index].m_cachefile, "Writing cache file.");
98  }
99  else
100  {
101  Logging::debug(m_ci[index].m_cachefile, "Reading cache file.");
102  }
103 
104  size_t filesize = 0;
105  bool isdefaultsize = false;
106  uint8_t *p = nullptr;
107 
108  if (!map_file(m_ci[index].m_cachefile, &m_ci[index].m_fd, &p, &filesize, &isdefaultsize, 0, (flags & CACHE_FLAG_RW) ? true : false))
109  {
110  return false;
111  }
112 
113  if (!isdefaultsize)
114  {
115  m_ci[index].m_buffer_pos = m_ci[index].m_buffer_watermark = filesize;
116  }
117 
118  m_ci[index].m_buffer_size = filesize;
119  m_ci[index].m_buffer = static_cast<uint8_t*>(p);
120 
121  ++m_cur_open; // track open files
122 
123  return true;
124 }

126 bool Buffer::close_file(uint32_t segment_no, uint32_t flags)
127 {
128  uint32_t index = segment_no;
129  if (index)
130  {
131  index--;
132  }
133 
134  m_ci[index].m_flags &= ~flags;
135 
136  if (m_ci[index].m_flags)
137  {
138  Logging::trace(m_ci[index].m_cachefile, "Cache file still in use while trying to close. Currently open: %1", m_cur_open);
139  return true;
140  }
141 
142  if (m_ci[index].m_fd == -1)
143  {
144  // Already closed
145  Logging::trace(m_ci[index].m_cachefile, "No need to close unopened cache file. Currenly open: %1", m_cur_open);
146  return true;
147  }
148 
149  Logging::trace(m_ci[index].m_cachefile, "Closing cache file.");
150 
151  bool success = unmap_file(m_ci[index].m_cachefile, &m_ci[index].m_fd, &m_ci[index].m_buffer, &m_ci[index].m_buffer_watermark);
152 
153  m_ci[index].m_buffer_pos = 0;
154  m_ci[index].m_buffer_size = 0;
155 
156  if (success && m_cur_open > 0)
157  {
158  --m_cur_open; // track open files
159  }
160 
161  return success;
162 }

164 bool Buffer::init(bool erase_cache)
165 {
166  std::lock_guard<std::recursive_mutex> lck (m_mutex);
167 
168  if (is_open())
169  {
170  return true;
171  }
172 
173  bool success = true;
174 
175  try
176  {
177  if ((virtualfile()->m_flags & VIRTUALFLAG_HLS))
178  {
179  // HLS format: create several segments
180  if (virtualfile()->get_segment_count())
181  {
182  m_ci.resize(virtualfile()->get_segment_count());
183 
184  for (uint32_t segment_no = 1; segment_no <= virtualfile()->get_segment_count(); segment_no++)
185  {
186  make_cachefile_name(m_ci[segment_no - 1].m_cachefile, filename() + "." + make_filename(segment_no, params.current_format(virtualfile())->fileext()), params.current_format(virtualfile())->fileext(), false);
187  }
188  }
189  else
190  {
191  Logging::error(filename(), "INTERNAL ERROR: Segment count is 0!");
192  errno = EINVAL;
193  throw false;
194  }
195  }
196  else
197  {
198  // All other formats: create just a single segment.
199  m_ci.resize(1);
200 
201  make_cachefile_name(m_ci[0].m_cachefile, filename(), params.current_format(virtualfile())->fileext(), false);
202  if ((virtualfile()->m_flags & VIRTUALFLAG_FRAME))
203  {
204  // Create extra index cash for frame sets only
205  make_cachefile_name(m_ci[0].m_cachefile_idx, filename(), params.current_format(virtualfile())->fileext(), true);
206  }
207  }
208 
209  // Set current segment
210  m_cur_ci = &m_ci[0];
211 
212  // Create the path to the cache file. All paths are the same, so this is required only once.
213  char *cachefile = new_strdup(m_ci[0].m_cachefile);
214 
215  if (cachefile == nullptr)
216  {
217  Logging::error(m_ci[0].m_cachefile, "Error opening cache file: Out of memory");
218  errno = ENOMEM;
219  throw false;
220  }
221 
222  if (mktree(dirname(cachefile), S_IRWXU | S_IRWXG | S_IROTH | S_IXOTH) && errno != EEXIST)
223  {
224  Logging::error(m_ci[0].m_cachefile, "Error creating cache directory: (%1) %2", errno, strerror(errno));
225  delete [] cachefile;
226  throw false;
227  }
228  errno = 0; // reset EEXIST, error can safely be ignored here
229 
230  delete [] cachefile;
231 
232  for (uint32_t index = 0; index < segment_count(); index++)
233  {
234  m_ci[index].m_fd = -1;
235  m_ci[index].m_buffer = nullptr;
236  m_ci[index].m_buffer_pos = 0;
237  m_ci[index].m_buffer_watermark = 0;
238  m_ci[index].m_buffer_size = 0;
239 
240  if (erase_cache)
241  {
242  remove_cachefile(index + 1);
243  errno = 0; // ignore this error
244  }
245  }
246 
247  // Index only required for frame sets and there is only one.
248  if (!m_ci[0].m_cachefile_idx.empty())
249  {
250  if (virtualfile()->m_video_frame_count == 0)
251  {
252  errno = EINVAL;
253  Logging::error(m_ci[0].m_cachefile, "INTERNAL ERROR! Frame count is zero (%1) %2", errno, strerror(errno));
254  throw false;
255  }
256  assert(sizeof(IMAGE_FRAME) == 32);
257 
258  size_t filesize = 0;
259  bool isdefaultsize = false;
260  uint8_t *p = nullptr;
261 
262  if (!map_file(m_ci[0].m_cachefile_idx, &m_ci[0].m_fd_idx, &p, &filesize, &isdefaultsize, static_cast<off_t>(sizeof(IMAGE_FRAME)) * virtualfile()->m_video_frame_count, false))
263  {
264  throw false;
265  }
266 
267  m_ci[0].m_buffer_size_idx = filesize;
268  m_ci[0].m_buffer_idx = static_cast<uint8_t*>(p);
269  }
270  }
271  catch (bool _success)
272  {
273  success = _success;
274 
275  if (!success)
276  {
277  for (uint32_t index = 0; index < segment_count(); index++)
278  {
279  m_ci[index].m_fd = -1;
280  m_ci[index].m_buffer = nullptr;
281  m_ci[index].m_buffer_pos = 0;
282  m_ci[index].m_buffer_watermark = 0;
283  m_ci[index].m_buffer_size = 0;
284  }
285  }
286  }
287 
288  return success;
289 }

345 bool Buffer::map_file(const std::string & filename, int *fd, uint8_t **p, size_t *filesize, bool *isdefaultsize, off_t defaultsize, bool truncate) const
346 {
347  bool success = true;
348 
349  Logging::trace(filename, "Mapping cache file.");
350 
351  try
352  {
353  struct stat sb;
354 
355  *fd = ::open(filename.c_str(), O_CREAT | O_RDWR | (truncate ? O_TRUNC : 0), static_cast<mode_t>(0644));
356  if (*fd == -1)
357  {
358  Logging::error(filename, "Error opening cache file: (%1) %2", errno, strerror(errno));
359  throw false;
360  }
361 
362  if (fstat(*fd, &sb) == -1)
363  {
364  Logging::error(filename, "File stat failed: (%1) %2 (fd = %3)", errno, strerror(errno), *fd);
365  throw false;
366  }
367 
368  if (!S_ISREG(sb.st_mode))
369  {
370  Logging::error(filename, "Not a file.");
371  throw false;
372  }
373 
374  if (!sb.st_size || *isdefaultsize)
375  {
376  // If file is empty or did not exist set file size to default
377 
378  if (!defaultsize)
379  {
380  defaultsize = sysconf(_SC_PAGESIZE);
381  }
382 
383  if (ftruncate(*fd, defaultsize) == -1)
384  {
385  Logging::error(filename, "Error calling ftruncate() to 'stretch' the file: (%1) %2 (fd = %3)", errno, strerror(errno), *fd);
386  throw false;
387  }
388 
389  *filesize = static_cast<size_t>(defaultsize);
390  *isdefaultsize = true;
391  }
392  else
393  {
394  // Keep size
395  *filesize = static_cast<size_t>(sb.st_size);
396  *isdefaultsize = false;
397  }
398 
399  *p = static_cast<uint8_t *>(mmap(nullptr, *filesize, PROT_READ | PROT_WRITE, MAP_SHARED, *fd, 0));
400  if (*p == MAP_FAILED)
401  {
402  Logging::error(filename, "File mapping failed: (%1) %2 (fd = %3)", errno, strerror(errno), *fd);
403  throw false;
404  }
405  }
406  catch (bool _success)
407  {
408  success = _success;
409 
410  if (!success && *fd != -1)
411  {
412  ::close(*fd);
413  *fd = -1;
414  }
415  }
416 
417  return success;
418 }

420 bool Buffer::unmap_file(const std::string &filename, int * fd, uint8_t **p, size_t * filesize) const
421 {
422  bool success = true;
423 
424  Logging::trace(filename, "Unmapping cache file.");
425 
426  void * __p = *p;
427  size_t __filesize = *filesize;
428  int __fd = *fd;
429 
430  // Clear all variables
431  *p = nullptr;
432  *filesize = 0;
433  *fd = -1;
434 
435  if (__p != nullptr)
436  {
437  if (munmap(__p, __filesize ? __filesize : static_cast<size_t>(sysconf(_SC_PAGESIZE))) == -1) // Make sure we do not unmap a zero size file (spitzs EINVBAL error)
438  {
439  Logging::error(filename, "Unmapping cache file failed: (%1) %2 %3", errno, strerror(errno), __filesize);
440  success = false;
441  }
442  }
443 
444  if (__fd != -1)
445  {
446  if (__filesize)
447  {
448  if (ftruncate(__fd, static_cast<off_t>(__filesize)) == -1)
449  {
450  Logging::error(filename, "Error calling ftruncate() to resize and close the cache file: (%1) %2 (fd = %3)", errno, strerror(errno), __fd);
451  success = false;
452  }
453  ::close(__fd);
454  }
455  else
456  {
457  ::close(__fd);
458 
459  if (unlink(filename.c_str()))
460  {
461  Logging::error(filename, "Error removing the cache file: (%1) %2 (fd = %3)", errno, strerror(errno), __fd);
462  success = false;
463  }
464  }
465  }
466 
467  return success;
468 }

470 bool Buffer::release(int flags /*= CACHE_CLOSE_NOOPT*/)
471 {
472  std::lock_guard<std::recursive_mutex> lck (m_mutex);
473 
474  bool success = true;
475 
476  if (!is_open())
477  {
478  if (CACHE_CHECK_BIT(CACHE_CLOSE_DELETE, flags))
479  {
480  for (uint32_t n = 1; n <= segment_count(); n++)
481  {
482  remove_cachefile(n);
483  }
484  errno = 0; // ignore this error
485  }
486 
487  return true;
488  }
489 
490  // Write active cache to disk
491  flush();
492 
493  // Close anything that's still open
494  for (uint32_t index = 0; index < segment_count(); index++)
495  {
496  if (!close_file(index + 1, CACHE_FLAG_RO | CACHE_FLAG_RW))
497  {
498  success = false;
499  }
500 
501  if (CACHE_CHECK_BIT(CACHE_CLOSE_DELETE, flags))
502  {
503  remove_cachefile(index + 1);
504  errno = 0; // ignore this error
505  }
506  }
507 
508  // Remove index for frame sets. There is only one.
509  if (!m_ci[0].m_cachefile_idx.empty())
510  {
511  if (!unmap_file(m_ci[0].m_cachefile_idx, &m_ci[0].m_fd_idx, &m_ci[0].m_buffer_idx, &m_ci[0].m_buffer_size_idx))
512  {
513  success = false;
514  }
515  }
516 
517  return success;
518 }

659 size_t Buffer::write_frame(const uint8_t *data, size_t length, uint32_t frame_no)
660 {
661  std::lock_guard<std::recursive_mutex> lck (m_mutex);
662 
663  if (data == nullptr || m_cur_ci == nullptr || m_cur_ci->m_buffer_idx == nullptr || frame_no < 1 || frame_no > virtualfile()->m_video_frame_count)
664  {
665  // Invalid parameter
666  errno = EINVAL;
667  return 0;
668  }
669 
670  LPIMAGE_FRAME old_image_frame;
671  IMAGE_FRAME new_image_frame;
672  size_t bytes_written;
673  size_t start = static_cast<size_t>(frame_no - 1) * sizeof(IMAGE_FRAME);
674 
675  old_image_frame = reinterpret_cast<LPIMAGE_FRAME>(m_cur_ci->m_buffer_idx + start);
676 
677  if (old_image_frame->m_frame_no && (old_image_frame->m_size <= static_cast<uint32_t>(length)))
678  {
679  // Frame already exists and has enough space
680  old_image_frame->m_size = static_cast<uint32_t>(length);
681 
682  // Write image
683  seek(static_cast<long>(old_image_frame->m_offset), SEEK_SET);
684  bytes_written = write(data, old_image_frame->m_size);
685  if (bytes_written != old_image_frame->m_size)
686  {
687  return 0;
688  }
689  }
690  else
691  {
692  // Create new frame if not existing or not enough space
693  memset(&new_image_frame, 0xFF, sizeof(new_image_frame));
694  memcpy(new_image_frame.m_tag, IMAGE_FRAME_TAG, sizeof(new_image_frame.m_tag));
695  new_image_frame.m_frame_no = frame_no;
696  new_image_frame.m_offset = buffer_watermark();
697  new_image_frame.m_size = static_cast<uint32_t>(length);
698 
699  // Write image
700  seek(static_cast<long>(new_image_frame.m_offset), SEEK_SET);
701  bytes_written = write(data, new_image_frame.m_size);
702  if (bytes_written != new_image_frame.m_size)
703  {
704  return 0;
705  }
706 
707  memcpy(reinterpret_cast<void *>(m_cur_ci->m_buffer_idx + start), &new_image_frame, sizeof(IMAGE_FRAME));
708  }
709 
710  return bytes_written;
711 }

741 {
742  LPCACHEINFO ci = cacheinfo(segment_no);
743

745  {

748  }

751 
752  switch (whence)
753  {
754  case SEEK_SET:
755  {
756  seek_pos = offset;
757  break;
758  }
759  case SEEK_CUR:
760  {

762  break;
763  }
764  case SEEK_END:
765  {
766  seek_pos = static_cast<off_t>(size(segment_no)) + offset;
767  break;
768  }
769  default:
770  {
771  errno = EINVAL;
772  return (EOF);
773  }
774  }

777  {
778  ci->m_buffer_pos = size(segment_no); // Cannot go beyond EOF. Set position to end, leave errno untouched.

783  {
784  errno = EINVAL;

786  }
787
