46uint32_t VIRTUALFILE::get_segment_count() const
47{
48 if (m_duration && params.m_segment_duration)
49 {
50 return static_cast<uint32_t>((m_duration - 200000) / params.m_segment_duration) + 1;
51 }
52 else
53 {
54 return 0;

97void FileIO::set_virtualfile(LPVIRTUALFILE virtualfile)
98{
99 m_virtualfile = virtualfile;
100
101 if (virtualfile != nullptr)
102 {
103 // Store path to original file without file name for fast access
104 m_path = m_virtualfile->m_origfile;
105
106 remove_filename(&m_path);

109 {
110 m_path.clear();
