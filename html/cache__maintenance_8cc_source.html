90 static bool start_timer(time_t interval)
91 {
92  struct sigevent sev;
93  struct itimerspec its;
94  long long freq_nanosecs;
95  struct sigaction sa;
96 
97  freq_nanosecs = interval * 1000000000LL;
98 
99  Logging::trace(nullptr, "Starting maintenance timer with %1period.", format_time(interval).c_str());
100 
101  // Establish maintenance_handler for timer signal
102  sa.sa_flags = SA_SIGINFO;
103  sa.sa_sigaction = maintenance_handler;
104  sigemptyset(&sa.sa_mask);
105  if (sigaction(SIGMAINT, &sa, nullptr) == -1)
106  {
107  Logging::error(nullptr, "start_timer(): sigaction failed: (%1) %2", errno, strerror(errno));
108  return false;
109  }
110 
111  // Block timer signal temporarily
112  sigemptyset(&mask);
113  sigaddset(&mask, SIGMAINT);
114  if (sigprocmask(SIG_SETMASK, &mask, nullptr) == -1)
115  {
116  Logging::error(nullptr, "start_timer(): sigprocmask(SIG_SETMASK) failed: (%1) %2", errno, strerror(errno));
117  return false;
118  }
119 
120  // Create the timer
121  sev.sigev_notify = SIGEV_SIGNAL;
122  sev.sigev_signo = SIGMAINT;
123  sev.sigev_value.sival_ptr = &timerid;
124  if (timer_create(CLOCKID, &sev, &timerid) == -1)
125  {
126  Logging::error(nullptr, "start_timer(): timer_create failed: (%1) %2", errno, strerror(errno));
127  return false;
128  }
129 
130  // Start the timer
131  its.it_value.tv_sec = static_cast<time_t>(freq_nanosecs / 1000000000);
132  its.it_value.tv_nsec = static_cast<long>(freq_nanosecs % 1000000000);
133  its.it_interval.tv_sec = its.it_value.tv_sec;
134  its.it_interval.tv_nsec = its.it_value.tv_nsec;
135 
136  if (timer_settime(timerid, 0, &its, nullptr) == -1)
137  {
138  Logging::error(nullptr, "start_timer(): timer_settime failed: (%1) %2", errno, strerror(errno));
139  return false;
140  }
141 
142  if (sigprocmask(SIG_UNBLOCK, &mask, nullptr) == -1)
143  {
144  Logging::error(nullptr, "start_timer(): sigprocmask(SIG_UNBLOCK) failed: (%1) %2", errno, strerror(errno));
145  }
146 
147  Logging::trace(nullptr, "The maintenance timer started successfully.");
148 
149  return true;
150 }

173 static bool link_up()
174 {
175  key_t shmkey;
176 
177  Logging::debug(nullptr, "Activating " PACKAGE " inter-process link.");
178 
179  // initialise a shared variable in shared memory
180  shmkey = ftok ("/dev/null", 5); // valid directory name and a number
181 
182  if (shmkey == -1)
183  {
184  Logging::error(nullptr, "link_up(): ftok error (%1) %2", errno, strerror(errno));
185  return false;
186  }
187 
188  // First try to open existing memory.
189  shmid = shmget (shmkey, sizeof (pid_t), S_IRUSR | S_IWUSR | S_IRGRP | S_IWGRP);
190  if (shmid != -1)
191  {
192  // Shared memory already exists, seems we are client.
193  master = false;
194  }
195  else
196  {
197  // Ignore error at first, try to create memory.
198  shmid = shmget (shmkey, sizeof (pid_t), IPC_CREAT | S_IRUSR | S_IWUSR | S_IRGRP | S_IWGRP);
199  if (shmid != -1)
200  {
201  // Shared memory freshly created, seems we are master.
202  master = true;
203  }
204  else
205  {
206  Logging::error(nullptr, "link_up(): shmget error (%1) %2", errno, strerror(errno));
207  return false;
208  }
209  }
210 
211  pid_master = static_cast<pid_t *>(shmat (shmid, nullptr, 0)); // attach pid_master to shared memory
212 
213  if (master)
214  {
215  *pid_master = getpid();
216  Logging::info(nullptr, "The process with PID %1 is now master.", *pid_master);
217  }
218  else
219  {
220  Logging::info(nullptr, "The process with PID %1 is now a client, and PID %2 is the master.", getpid(), *pid_master);
221  }
222 
223  // Also create inter-process semaphore.
224  sem = sem_open(const_cast<const char *>(SEM_OPEN_FILE), O_CREAT, 0777, 1);
225 
226  if (sem == SEM_FAILED)
227  {
228  Logging::error(nullptr, "link_up(): sem_open error (%1) %2", errno, strerror(errno));
229  link_down();
230  return false;
231  }
232 
233  return true;
234 }

239 static void master_check()
240 {
241  pid_t pid_self = getpid();
242 
243  if (*pid_master == pid_self)
244  {
245  Logging::trace(nullptr, "The process with PID %1 is already master.", pid_self);
246  return;
247  }
248 
249  sem_wait(sem);
250 
251  // Check if master process still exists
252  int master_running = (getpgid(*pid_master) >= 0);
253 
254  Logging::trace(nullptr, "Master with PID %1 is %2 running.", *pid_master, master_running ? "still" : "NOT");
255 
256  if (!master_running)
257  {
258  Logging::info(nullptr, "Master with PID %1 is gone. PID %2 taking over as new master.", *pid_master, pid_self);
259 
260  // Register us as master
261  *pid_master = pid_self;
262  master = true;
263  }
264 
265  sem_post(sem);
266 }

272 static bool link_down()
273 {
274  struct shmid_ds buf;
275  bool success = true;
276 
277  Logging::info(nullptr, "Shutting " PACKAGE " inter-process link down.");
278 
279  if (sem != nullptr && sem_close(sem))
280  {
281  Logging::error(nullptr, "link_down(): sem_close error (%1) %2", errno, strerror(errno));
282  success = false;
283  }
284 
285  // shared memory detach
286  if (shmdt (pid_master))
287  {
288  Logging::error(nullptr, "link_down(): shmdt error (%1) %2", errno, strerror(errno));
289  success = false;
290  }
291 
292  if (shmctl(shmid, IPC_STAT, &buf))
293  {
294  Logging::error(nullptr, "link_down(): shmctl error (%1) %2", errno, strerror(errno));
295  success = false;
296  }
297  else
298  {
299  if (!buf.shm_nattch)
300  {
301  if (shmctl (shmid, IPC_RMID, nullptr))
302  {
303  Logging::error(nullptr, "link_down(): shmctl error (%1) %2", errno, strerror(errno));
304  success = false;
305  }
306 
307  // unlink prevents the semaphore existing forever
308  // if a crash occurs during the execution
309  if (sem_unlink(SEM_OPEN_FILE))
310  {
311  Logging::error(nullptr, "link_down(): sem_unlink error (%1) %2", errno, strerror(errno));
312  success = false;
313  }
314  }
315  }
316 
317  return success;
318 }
