529 bool Cache::prune_expired()
530 {
531  if (params.m_expiry_time <= 0)
532  {
533  // There's no limit.
534  return true;
535  }
536 
537  std::vector<cache_key_t> keys;
538  sqlite3_stmt * stmt;
539  time_t now = time(nullptr);
540  char sql[1024];
541 
542  Logging::trace(m_cacheidx_file, "Pruning expired cache entries older than %1...", format_time(params.m_expiry_time).c_str());
543 
544  sprintf(sql, "SELECT filename, desttype, strftime('%%s', access_time) FROM cache_entry WHERE strftime('%%s', access_time) + %" FFMPEGFS_FORMAT_TIME_T " < %" FFMPEGFS_FORMAT_TIME_T ";\n", params.m_expiry_time, now);
545 
546  std::lock_guard<std::recursive_mutex> lck (m_mutex);
547 
548  sqlite3_prepare(m_cacheidx_db, sql, -1, &stmt, nullptr);
549 
550  int ret = 0;
551  while ((ret = sqlite3_step(stmt)) == SQLITE_ROW)
552  {
553  const char *filename = reinterpret_cast<const char *>(sqlite3_column_text(stmt, 0));
554  const char *desttype = reinterpret_cast<const char *>(sqlite3_column_text(stmt, 1));
555 
556  keys.push_back(std::make_pair(filename, desttype));
557 
558  Logging::trace(filename, "Found %1 old entries.", format_time(now - static_cast<time_t>(sqlite3_column_int64(stmt, 2))).c_str());
559  }
560 
561  Logging::trace(m_cacheidx_file, "%1 expired cache entries found.", keys.size());
562 
563  if (ret == SQLITE_DONE)
564  {
565  for (std::vector<cache_key_t>::const_iterator it = keys.begin(); it != keys.end(); it++)
566  {
567  const cache_key_t & key = *it;
568  Logging::trace(m_cacheidx_file, "Pruning '%1' - Type: %2", key.first.c_str(), key.second.c_str());
569 
570  cache_t::iterator p = m_cache.find(key);
571  if (p != m_cache.end())
572  {
573  delete_entry(&p->second, CLOSE_CACHE_DELETE);
574  }
575 
576  if (delete_info(key.first, key.second))
577  {
578  remove_cachefile(key.first, key.second);
579  }
580  }
581  }
582  else
583  {
584  Logging::error(m_cacheidx_file, "Failed to execute select: %1\n%2\n%3", ret, sqlite3_errmsg(m_cacheidx_db), expanded_sql(stmt).c_str());
585  }
586 
587  sqlite3_finalize(stmt);
588 
589  return true;
590 }
