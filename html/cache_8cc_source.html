575 bool Cache::load_index()
576 {
577  bool success = true;
578 
579  try
580  {
581  char *errmsg = nullptr;
582  int ret;
583  bool new_database = false;
584  bool need_upate = false;
585 
586  transcoder_cache_path(m_cacheidx_file);
587 
588  if (mktree(m_cacheidx_file, S_IRWXU | S_IRWXG | S_IROTH | S_IXOTH) && errno != EEXIST)
589  {
590  Logging::error(m_cacheidx_file, "Error creating cache directory: (%1) %2\n%3", errno, strerror(errno), m_cacheidx_file.c_str());
591  throw false;
592  }
593 
594  append_filename(&m_cacheidx_file, "cacheidx.sqlite");
595 
596  // initialise engine
597  if (SQLITE_OK != (ret = sqlite3_initialize()))
598  {
599  Logging::error(m_cacheidx_file, "Failed to initialise SQLite3 library: (%1) %2", ret, sqlite3_errstr(ret));
600  throw false;
601  }
602 
603  // open connection to a DB
604  if (SQLITE_OK != (ret = sqlite3_open_v2(m_cacheidx_file.c_str(), &m_cacheidx_db, SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE | SQLITE_OPEN_SHAREDCACHE, nullptr)))
605  {
606  Logging::error(m_cacheidx_file, "Failed to initialise SQLite3 connection: (%1) %2", ret, sqlite3_errmsg(m_cacheidx_db));
607  throw false;
608  }
609 
610  if (SQLITE_OK != (ret = sqlite3_busy_timeout(m_cacheidx_db, 1000)))
611  {
612  Logging::error(m_cacheidx_file, "Failed to set SQLite3 busy timeout: (%1) %2", ret, sqlite3_errmsg(m_cacheidx_db));
613  throw false;
614  }
615 
616  // Beginning with version 3.7.0 (2010-07-21), a new "Write-Ahead Log" option
617  // We support Sqlite from 3.7.13 anyway
618  if (SQLITE_OK != (ret = sqlite3_exec(m_cacheidx_db, "pragma journal_mode = WAL", nullptr, nullptr, nullptr)))
619  {
620  Logging::error(m_cacheidx_file, "Failed to set SQLite3 WAL mode: (%1) %2", ret, sqlite3_errmsg(m_cacheidx_db));
621  throw false;
622  }
623 
624  // Very strange: Compare operations with =, > etc. are case sensitive, while LIKE by default ignores upper/lowercase.
625  // Produces strange results when reading from a Samba drive and different cases are used...
626  if (SQLITE_OK != (ret = sqlite3_exec(m_cacheidx_db, "PRAGMA case_sensitive_like = 1;", nullptr, nullptr, nullptr)))
627  {
628  Logging::error(m_cacheidx_file, "Failed to set SQLite3 case_sensitive_like = 1: (%1) %2", ret, sqlite3_errmsg(m_cacheidx_db));
629  throw false;
630  }
631 
632  // Make sure the next changes are either all successfull or rolled back
633  if (!begin_transaction())
634  {
635  throw false;
636  }
637 
638  // Check if we got a new, empty database and create necessary tables
639
