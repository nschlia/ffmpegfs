54 void thread_pool::loop_function()
55 {
56  unsigned int thread_no = ++m_cur_threads;
57 
58  Logging::trace(nullptr, "Starting pool thread no. %1 with id 0x%<%" FFMPEGFS_FORMAT_PTHREAD_T ">2.", thread_no, pthread_self());
59 
60  while (true)
61  {
62  THREADINFO info;
63  {
64  std::unique_lock<std::mutex> lock(m_queue_mutex);
65  m_queue_condition.wait(lock, [this]{ return (!m_thread_queue.empty() || m_queue_shutdown); });
66 
67  if (m_queue_shutdown)
68  {
69  lock.unlock();
70  break;
71  }
72 
73  Logging::trace(nullptr, "Starting job using pool thread no. %1 with id 0x%<%" FFMPEGFS_FORMAT_PTHREAD_T ">2.", thread_no, pthread_self());
74  info = m_thread_queue.front();
75  m_thread_queue.pop();
76  }
77 
78  info.m_thread_func(info.m_opaque);
79  }
80 
81  Logging::trace(nullptr, "Exiting pool thread no. %1 with id 0x%<%" FFMPEGFS_FORMAT_PTHREAD_T ">2.", thread_no, pthread_self());
82 }

143 {
144  if (!silent)

147  }
