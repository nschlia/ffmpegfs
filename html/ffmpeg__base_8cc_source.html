69 void FFmpeg_Base::video_stream_setup(AVCodecContext *output_codec_ctx, AVStream* output_stream, AVCodecContext *input_codec_ctx, AVRational framerate, AVPixelFormat enc_hw_pix_fmt) const
70 {
71  AVRational time_base_tbn;
72  AVRational time_base_tbc;
73 
74  if (!framerate.num || !framerate.den)
75  {
76  framerate.num = 25;
77  framerate.den = 1;
78  Logging::warning(nullptr, "No information about the input framerate is available. Falling back to a default value of 25fps for the output stream.");
79  }
80 
81  // timebase: This is the fundamental unit of time (in seconds) in terms
82  // of which frame timestamps are represented. For fixed-fps content,
83  // timebase should be 1/framerate and timestamp increments should be
84  // identical to 1.
85  //time_base = m_in.m_pVideo_stream->time_base;
86 
87  // tbn: must be set differently for the target format. Otherwise produces strange results.
88  switch (output_codec_ctx->codec_id)
89  {
90  case AV_CODEC_ID_THEORA: // ogg
91  case AV_CODEC_ID_MPEG1VIDEO:
92  case AV_CODEC_ID_MPEG2VIDEO:
93  {
94  time_base_tbn = av_inv_q(framerate);
95  time_base_tbc = time_base_tbn;
96  break;
97  }
98  case AV_CODEC_ID_VP9: // webm
99  {
100  time_base_tbn.num = 1;
101  time_base_tbn.den = 1000;
102  time_base_tbc = time_base_tbn;
103  break;
104  }
105  case AV_CODEC_ID_H264: // h264
106  case AV_CODEC_ID_H265: // h265
107  {
108  time_base_tbn.num = 1;
109  time_base_tbn.den = 90000;
110  time_base_tbc = av_inv_q(framerate);
111  break;
112  }
113  default: // mp4 and all others
114  {
115  time_base_tbn.num = 1;
116  time_base_tbn.den = 90000;
117  time_base_tbc = time_base_tbn;
118  break;
119  }
120  }
121 
122  // tbn
123  output_stream->time_base = time_base_tbn;
124  // tbc
125  output_codec_ctx->time_base = time_base_tbc;
126 
127  // tbr
128  // output_stream->r_frame_rate = m_in.m_pVideo_stream->r_frame_rate;
129  output_stream->r_frame_rate = framerate;
130 
131  // fps
132  output_stream->avg_frame_rate = framerate;
133  // output_codec_ctx->framerate = framerate;
134 
135  if (enc_hw_pix_fmt == AV_PIX_FMT_NONE)
136  {
137  // Automatic pix_fmt selection
138  int loss = 0;
139 
140  AVPixelFormat src_pix_fmt = input_codec_ctx->pix_fmt;
141  if (output_codec_ctx->codec->pix_fmts != nullptr)
142  {
143  int alpha = 0;
144  enc_hw_pix_fmt = avcodec_find_best_pix_fmt_of_list(output_codec_ctx->codec->pix_fmts, src_pix_fmt, alpha, &loss);
145  }
146 
147  if (enc_hw_pix_fmt == AV_PIX_FMT_NONE)
148  {
149  // Fail safe if avcodec_find_best_pix_fmt_of_list has no idea what to use.
150  switch (output_codec_ctx->codec_id)
151  {
152  case AV_CODEC_ID_PRORES: // mov/prores
153  {
154  // yuva444p10le
155  // ProRes 4:4:4 if the source is RGB and ProRes 4:2:2 if the source is YUV.
156  enc_hw_pix_fmt = AV_PIX_FMT_YUV422P10LE;
157  break;
158  }
159  default: // all others
160  {
161  // At this moment the output format must be AV_PIX_FMT_YUV420P;
162  enc_hw_pix_fmt = AV_PIX_FMT_YUV420P;
163  break;
164  }
165  }
166  }
167  }
168 
169  output_codec_ctx->pix_fmt = enc_hw_pix_fmt;
170  output_codec_ctx->gop_size = 12; // emit one intra frame every twelve frames at most
171 }

219 void FFmpeg_Base::video_info(bool out_file, const AVFormatContext *format_ctx, const AVStream *stream) const
220 {
221  if (stream != nullptr && stream->codecpar != nullptr)
222  {
223  int64_t duration = AV_NOPTS_VALUE;
224 
225  if (stream->duration != AV_NOPTS_VALUE)
226  {
227  duration = ffmpeg_rescale_q_rnd(stream->duration, stream->time_base);
228  }
229 
230  Logging::debug(out_file ? virtname() : filename(), "Video %1 #%2: %3@%4 [%5]",
231  out_file ? "out" : "in",
232  stream->index,
233  get_codec_name(stream->codecpar->codec_id, false),
234  format_bitrate((stream->codecpar->bit_rate != 0) ? stream->codecpar->bit_rate : format_ctx->bit_rate).c_str(),
235  format_duration(duration).c_str());
236  }
237  else
238  {
239  Logging::debug(out_file ? virtname() : filename(), "Video %1: invalid stream",
240  out_file ? "out" : "in");
241  }

314 uint32_t FFmpeg_Base::pts_to_frame(AVStream* stream, int64_t pts) const
315 {
316  if (pts == AV_NOPTS_VALUE)
317  {
318  return 0;
319  }
320  int64_t start_time = (stream->start_time != AV_NOPTS_VALUE) ? stream->start_time : 0;
321  AVRational factor = av_mul_q(stream->r_frame_rate, stream->time_base);
322  return static_cast<uint32_t>(av_rescale(pts - start_time, factor.num, factor.den) + 1);
323 }

325 int64_t FFmpeg_Base::frame_to_pts(AVStream* stream, uint32_t frame_no) const
326 {
327  int64_t start_time = (stream->start_time != AV_NOPTS_VALUE) ? stream->start_time : 0;
328  AVRational factor = av_mul_q(stream->r_frame_rate, stream->time_base);
329  return static_cast<uint32_t>(av_rescale(frame_no - 1, factor.den, factor.num) + start_time);
330 }
