340 template <typename T, typename... Args>
341 static std::string format_helper(
342 const std::string &string_to_search,
343 const size_t index_to_replace,
344 T &&val,
345 Args &&...args)
346 {
347 // Match %# exactly (e.g. %12 and %123 literally)
348
349 std::regex exp("%(<([^>]+)>)*" + std::to_string(index_to_replace) + "(?=[^0-9]|$)");
350 std::smatch res;
351 std::string string_to_update(string_to_search);
352 std::string::const_iterator searchStart(string_to_search.cbegin());
353 size_t offset = 0;
354
355 while (std::regex_search(searchStart, string_to_search.cend(), res, exp))
356 {
357 std::ostringstream ostr;
358
359 if (res[2].length())
360 {
361 // Found match with printf format in res[2]
362 std::vector<char> fmt;
363 std::string ftmspec = res[2].str();
364
365 if (ftmspec.front() != '%')
366 {
367 fmt.push_back('%');
368 }
369
370
371 fmt.insert(fmt.end(), ftmspec.begin(), ftmspec.end());
372 fmt.push_back('\0');
373
374 size_t size = static_cast<size_t>(std::snprintf(nullptr, 0, fmt.data(), fix_std_string(val))) + 1;
375 std::vector<char> buffer;
376 buffer.resize(size);
377 std::snprintf(buffer.data(), size, fmt.data(), fix_std_string(val));
378 ostr << buffer.data();
379 }
380 else
381 {
382 // No printf format, replace literally
383 ostr << convert(val);
384 }
385
386 string_to_update.replace(static_cast<size_t>(res.position()) + offset, static_cast<size_t>(res[0].length()), ostr.str());
387
388 offset += static_cast<size_t>(res.position()) + ostr.str().length();
389
390 searchStart = res.suffix().first;
391 }
392
393 return format_helper(
394 string_to_update,
395 index_to_replace + 1,
396 std::forward<Args>(args)...);
397 }
